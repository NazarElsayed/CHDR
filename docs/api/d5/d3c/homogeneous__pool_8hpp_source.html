84 HOT uint8_t* expand(size_t _size, size_t _alignment) noexcept {
85
86 uint8_t* result = nullptr;
87
88 try {
89 const auto allocate_size = utils::max(m_block_width, _alignment);
90
91 result = static_cast<uint8_t*>(operator new(allocate_size, static_cast<std::align_val_t>(_alignment)));
92
93 m_blocks.emplace_back(
94 allocate_size,
95 result
96 );
97
98 const auto result_num = reinterpret_cast<uintptr_t>(result);
99
100 // Alignment calculations:
101 const auto aligned_chunk_size = ( _size + _alignment - 1U) & ~(_alignment - 1U);
102 const auto aligned_base = (result_num + _alignment - 1U) & ~(_alignment - 1U);
103
104 // Divide into aligned chunks and distribute:
105 if (const auto num_chunks = (allocate_size - (aligned_base - result_num)) / aligned_chunk_size;
106 num_chunks > 1U
107 ) {
108 m_free.reserve(m_free.size() + num_chunks - 1U);
109
110 IVDEP
111 for (size_t i = 1U; i < num_chunks; ++i) {
112 m_free.emplace_back(result + (aligned_chunk_size * i));
113 }
114 }
115
116 m_block_width = utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size);
117 }
118 catch (...) {
119
120 /* Catch any errors that occur during allocation. */
121
122 // ReSharper disable once CppDFAConstantConditions
123 if (result != nullptr) {
124 // ReSharper disable once CppDFAUnreachableCode
125 operator delete(result, static_cast<std::align_val_t>(_alignment));
126 result = nullptr;
127
128 m_blocks.pop_back();
129 }
130 }
131
132 return result;
133 }

135 HOT uint8_t* allocate_from_free() noexcept {
136
137 uint8_t* result;
138
139 if (!m_free.empty()) {
140 result = m_free.back();
141 m_free.pop_back();
142 }
143 else {
144 result = nullptr;
145 }
146
147 return result;
148 }

191 [[nodiscard]] HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
192 assert(_bytes > 0U && "Allocation size must be greater than zero.");
193
194 m_alignment = _alignment;
195
196 // Attempt to allocate from the stack block:
197 if (m_stack_write < StackSize) {
198
199 const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
200
201 if (aligned_bytes < MaxStackAllocationSize && m_stack_write + aligned_bytes <= s_stack_block_size) {
202
203 auto* aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
204 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
205
206 return aligned_ptr;
207 }
208 }
209
210 // Attempt to find a free block, or create one otherwise:
211 auto* aligned_ptr = allocate_from_free();
212 if (aligned_ptr == nullptr) {
213 aligned_ptr = expand(_bytes, _alignment);
214 }
215
216 if (aligned_ptr == nullptr) {
217 throw std::bad_alloc();
218 }
219
220 return aligned_ptr;
221 }

370 HOT void reset() noexcept {
371
372 m_stack_write = 0U;
373 m_block_width = m_initial_block_width;
374
375 try {
376 m_free.clear();
377
378 // Reuse memory from existing blocks to repopulate `free`:
379 for (auto& [size, data] : m_blocks) {
380
381 // Divide the block into chunks; all chunks are guaranteed aligned.
382 const auto chunk_size = size / m_block_width;
383
384 const auto current_size = m_free.size();
385 m_free.resize(current_size + chunk_size);
386
387 IVDEP
388 for (size_t j = 0U; j < chunk_size; ++j) {
389 m_free[j + current_size] = data + j * m_block_width;
390 }
391 }
392 }
393 catch (...) {
394 m_blocks.clear();
395 m_free.clear();
396 }
397 }

414 void release() noexcept {
415

418
419 {
420 cleanup();
421 decltype(m_blocks) temp{};
422 m_blocks = std::move(temp);
423 }
424 {

426 }
