101 HOT uint8_t* expand(size_t _bytes, size_t _alignment) noexcept {
102
103 assert(_bytes > 0U && "Allocation size must be greater than zero.");
104 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
105 assert((m_alignment == 0U || _alignment == m_alignment) && "Alignment mismatch.");
106
107 uint8_t* result { nullptr };
108
109 try {
110 const auto allocate_bytes = utils::max(m_block_width, _alignment);
111
112 result = static_cast<uint8_t*>(operator new(allocate_bytes, static_cast<std::align_val_t>(_alignment)));
113
114 m_blocks.emplace_back(
115 allocate_bytes,
116 result
117 );
118
119 const auto result_num = reinterpret_cast<uintptr_t>(result);
120
121 // Alignment calculations:
122 const auto aligned_chunk_bytes = ( _bytes + _alignment - 1U) & ~(_alignment - 1U);
123 const auto aligned_base = (result_num + _alignment - 1U) & ~(_alignment - 1U);
124
125 // Divide into aligned chunks and distribute:
126 if (const auto num_chunks = (allocate_bytes - (aligned_base - result_num)) / aligned_chunk_bytes;
127 num_chunks > 1U
128 ) {
129 size_t free_size;
130 m_free.resize((free_size = (m_free.size() + num_chunks - 1U)), {});
131
132 IVDEP
133 for (size_t i = 1U; i < num_chunks; ++i) {
134 m_free[free_size - i] = result + (aligned_chunk_bytes * (num_chunks - i));
135 }
136 }
137
138 m_block_width = utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size);
139 }
140 catch (...) {
141
142 /* Catch any errors that occur during allocation. */
143
144 // ReSharper disable once CppDFAConstantConditions
145 if (result != nullptr) {
146 // ReSharper disable once CppDFAUnreachableCode
147 operator delete(result, static_cast<std::align_val_t>(_alignment));
148 result = nullptr;
149
150 m_blocks.pop_back();
151 }
152 }
153
154 return result;
155 }
156
169 void cleanup() noexcept {
170 for (const auto& item : m_blocks) {
171 operator delete(item.data, static_cast<std::align_val_t>(m_alignment));
172 }
173 }

206 [[nodiscard]] virtual HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
207 assert(_bytes > 0U && "Allocation size must be greater than zero.");
208 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
209 assert((m_alignment == 0U || _alignment == m_alignment) && "Alignment mismatch.");
210
211 m_alignment = _alignment;
212
213 uint8_t* aligned_ptr { nullptr };
214
215 // Attempt to allocate from the stack block:
216 if (m_stack_write < StackSize) {
217
218 if (const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
219 aligned_bytes < MaxStackAllocationSize && m_stack_write + aligned_bytes <= s_stack_block_size
220 ) {
221 aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
222 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
223 }
224 }
225
226 if (aligned_ptr == nullptr) {
227
228 // Attempt to find a free block, or create one otherwise:
229 if (LIKELY(!m_free.empty())) {
230 aligned_ptr = m_free.back();
231 m_free.pop_back();
232 }
233 else {
234 aligned_ptr = expand(_bytes, _alignment);
235 }
236 }
237
238 // ReSharper disable once CppDFAConstantConditions
239 if (UNLIKELY(aligned_ptr == nullptr)) {
240 // ReSharper disable once CppDFAUnreachableCode
241 throw std::bad_alloc();
242 }
243
244 PREFETCH(aligned_ptr, _MM_HINT_T0);
245
246#if CHDR_DIAGNOSTICS == 1
247
248 __diagnostic_data. num_allocated += _bytes;
249 __diagnostic_data.peak_allocated = utils::max(__diagnostic_data.peak_allocated, __diagnostic_data.num_allocated);
250
251#endif //CHDR_DIAGNOSTICS == 1
252
253 return aligned_ptr;
254 }

423 HOT void reset() noexcept {
424
425 m_stack_write = 0U;
426 m_block_width = m_initial_block_width;
427
428 try {
429 m_free.clear();
430
431 // Reuse memory from existing blocks to repopulate `free`:
432 for (auto& [size, data] : m_blocks) {
433
434 // Divide the block into chunks; all chunks are guaranteed aligned.
435 const auto chunk_size = size / m_block_width;
436
437 const auto current_size = m_free.size();
438 m_free.resize(current_size + chunk_size);
439
440 IVDEP
441 for (size_t j = 0U; j < chunk_size; ++j) {
442 m_free[j + current_size] = data + j * m_block_width;
443 }
444 }
445 }
446 catch (...) {
447 m_blocks.clear();
448 m_free.clear();
449 }
