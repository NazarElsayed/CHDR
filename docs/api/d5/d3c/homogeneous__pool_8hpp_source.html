105 HOT uint8_t* expand(size_t _bytes, size_t _alignment) noexcept {
106
107 assert(_bytes > 0U && "Allocation size must be greater than zero.");
108 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
109 assert((m_alignment == 0U || _alignment == m_alignment) && "Alignment mismatch.");
110
111 uint8_t* result { nullptr };
112
113 try {
114 const auto allocate_bytes = utils::max(m_block_width, utils::max(_bytes, _alignment));
115
116 result = static_cast<uint8_t*>(operator new(allocate_bytes, static_cast<std::align_val_t>(_alignment)));
117
118 m_blocks.emplace_back(
119 allocate_bytes,
120 result
121 );
122
123 const auto result_num = reinterpret_cast<uintptr_t>(result);
124
125 // Alignment calculations:
126 const auto aligned_chunk_bytes = ( _bytes + _alignment - 1U) & ~(_alignment - 1U);
127 const auto aligned_base = (result_num + _alignment - 1U) & ~(_alignment - 1U);
128
129 // Divide into aligned chunks and distribute:
130 if (const auto num_chunks = (allocate_bytes - (aligned_base - result_num)) / aligned_chunk_bytes;
131 num_chunks > 1U
132 ) {
133 size_t free_size;
134 m_free.resize((free_size = (m_free.size() + num_chunks - 1U)), {});
135
136 IVDEP
137 for (size_t i = 1U; i < num_chunks; ++i) {
138 m_free[free_size - i] = result + (aligned_chunk_bytes * (num_chunks - i));
139 }
140 }
141
142 m_block_width = utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size);
143 }
144 catch (...) {
145
146 /* Catch any errors that occur during allocation. */
147
148 // ReSharper disable once CppDFAConstantConditions
149 if (result != nullptr) {
150 // ReSharper disable once CppDFAUnreachableCode
151 operator delete(result, static_cast<std::align_val_t>(_alignment));
152 result = nullptr;
153
154 m_blocks.pop_back();
155 }
156 }
157
158 return result;
159 }

173 void cleanup() noexcept {
174 for (const auto& item : m_blocks) {
175 operator delete(item.data, static_cast<std::align_val_t>(m_alignment));
176 }
177 }

210 [[nodiscard]] virtual HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
211
212 assert(_bytes <= std::min(m_initial_block_width, s_max_heap_block_size) &&
213 "Allocation size must not exceed the initial block width or maximum block size.");
214
215 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
216 assert((m_alignment == 0U || _alignment == m_alignment) && "Alignment mismatch.");
217
218 m_alignment = _alignment;
219
220 if (_bytes == 0U) {
221 return nullptr; // No-op. (See C++17 standard.)
222 }
223 else {
224 uint8_t* aligned_ptr { nullptr };
225
226 // Attempt to allocate from the stack block:
227 if (m_stack_write < StackSize) {
228
229 if (const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
230 aligned_bytes < MaxStackAllocationSize && m_stack_write + aligned_bytes <= s_stack_block_size
231 ) {
232 aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
233 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
234 }
235 }
236
237 if (aligned_ptr == nullptr) {
238
239 // Attempt to find a free block, or create one otherwise:
240 if (LIKELY(!m_free.empty())) {
241 aligned_ptr = m_free.back();
242 m_free.pop_back();
243 }
244 else {
245 aligned_ptr = expand(_bytes, _alignment);
246 }
247 }
248
249 // ReSharper disable once CppDFAConstantConditions
250 if (UNLIKELY(aligned_ptr == nullptr)) {
251 // ReSharper disable once CppDFAUnreachableCode
252 throw std::bad_alloc();
253 }
254
255 PREFETCH(aligned_ptr, 1, 1);
256
257#if CHDR_DIAGNOSTICS == 1
258
259 __diagnostic_data. num_allocated += _bytes;
260 __diagnostic_data.peak_allocated = utils::max(__diagnostic_data.peak_allocated, __diagnostic_data.num_allocated);
261
262#endif //CHDR_DIAGNOSTICS == 1
263
264 return aligned_ptr;
265 }
266 }

442 HOT void reset() noexcept {
443
444 m_stack_write = 0U;
445 m_block_width = m_initial_block_width;
446
447 try {
448 m_free.clear();
449
450 // Reuse memory from existing blocks to repopulate `free`:
451 for (auto& [size, data] : m_blocks) {
452
453 // Divide the block into chunks; all chunks are guaranteed aligned.
454 const auto chunk_size = size / m_block_width;
455
456 const auto current_size = m_free.size();
457 m_free.resize(current_size + chunk_size);
458
459 IVDEP
460 for (size_t j = 0U; j < chunk_size; ++j) {
461 m_free[j + current_size] = data + j * m_block_width;
462 }
463 }
464 }
465 catch (...) {
466 m_blocks.clear();
467 m_free.clear();
468 }

472 __diagnostic_data.peak_allocated = 0U;
473#endif //CHDR_DIAGNOSTICS == 1
474 }
