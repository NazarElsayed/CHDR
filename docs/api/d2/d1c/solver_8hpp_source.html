541 template<typename node_t, typename coord_t>
542 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
543
544 std::vector<coord_t> result{};
545
546 {
547 size_t depth = 0U;
548 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
549 ++depth;
550 }
551
552 result.resize(depth);
553 }
554
555 size_t i = 0U;
556 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
557 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
558 ++i;
559 }
560
561 return result;
562 }

779 [[maybe_unused, nodiscard]]
780#if __cplusplus >= 202003L
781 constexpr
782#endif // __cplusplus >= 202003L
783 static auto solve(const params_t& _params) {
784
785 std::vector<typename params_t::coord_type> result{};
786
787 std::exception_ptr exception { nullptr };
788
789 /*
790 * Run solver:
791 */
792
793 try {
794 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
795 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
796
797 if (_params.maze.contains(s) && _params.maze[s].is_active() &&
798 _params.maze.contains(e) && _params.maze[e].is_active()
799 ) {
800
801 result = s != e ?
802 solver_t<params_t>::invoke(_params) :
803 std::vector<typename params_t::coord_type> { _params.end };
804 }
805 }
806 catch (...) { // NOLINT(*-empty-catch)
807
808 // ReSharper disable once CppDFAUnusedValue
809 exception = std::current_exception();
810 }
811
812 // Reset memory resources:
813 if constexpr (!params_t::no_cleanup::value) {
814 try {
815 solver_utils::reset_resources(_params);
816 }
817 catch (...) { // NOLINT(*-empty-catch)
818 if (exception == nullptr) { exception = std::current_exception(); }
819 }
820 }
821
822 // Report first exception to occur:
823 if (exception != nullptr) {
824 std::rethrow_exception(exception);
825 }
826
827 return result;
828 }
