542 template<typename node_t, typename coord_t>
543 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
544
545 std::vector<coord_t> result{};
546
547 {
548 size_t depth = 0U;
549 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
550 ++depth;
551 }
552
553 result.resize(depth);
554 }
555
556 size_t i = 0U;
557 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
558 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
559 ++i;
560 }
561
562 return result;
563 }

862 [[maybe_unused, nodiscard]]
863#if __cplusplus >= 202003L
864 constexpr
865#endif // __cplusplus >= 202003L
866 static auto solve(const params_t& _params) {
867
868 std::vector<typename params_t::coord_type> result{};
869
870 std::exception_ptr exception { nullptr };
871
872 /*
873 * Run solver:
874 */
875
876 try {
877 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
878 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
879
880 if (_params.maze.contains(s) && _params.maze[s].is_active() &&
881 _params.maze.contains(e) && _params.maze[e].is_active()
882 ) {
883
884 result = s != e ?
885 solver_t<params_t>::invoke(_params) :
886 std::vector<typename params_t::coord_type> { _params.end };
887 }
888 }
889 catch (...) { // NOLINT(*-empty-catch)
890
891 // ReSharper disable once CppDFAUnusedValue
892 exception = std::current_exception();
893 }
894
895 // Reset memory resources:
896 if constexpr (!params_t::no_cleanup::value) {
897 try {
898 solver_utils::reset_resources(_params);
899 }
900 catch (...) { // NOLINT(*-empty-catch)
901 if (exception == nullptr) { exception = std::current_exception(); }
902 }
903 }
904
905 // Report first exception to occur:
906 if (exception != nullptr) {
907 std::rethrow_exception(exception);
908 }
909
910 return result;
911 }
