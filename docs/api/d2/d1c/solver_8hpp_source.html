77 template<typename node_t, typename coord_t>
78 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
79
80 std::vector<coord_t> result;
81
82 {
83 size_t depth = 0U;
84 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++depth) {}
85
86 result.resize(depth);
87 }
88
89 size_t i = 0U;
90 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++i) {
91 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
92 }
93
94 return result;
95 }

111 [[nodiscard]] static constexpr auto ibacktrack(const open_set_t& _open, const coord_t& _size) {
112
113 std::vector<coord_t> result;
114 result.reserve(_open.size());
115
116 for (auto it = _open.rbegin(); it != _open.rend(); ++it) {
117 result.emplace_back(utils::to_nd(it->m_index, _size));

235 static constexpr node_data get_data(const maze_neighbour_t& _n, const params_t& _params) noexcept {
236
237 if constexpr (is_graph<decltype(_params.maze)>::value) {
238
239 // _params.maze is a graph...
240 const auto& [nIndex, nDistance](_n);
241
242 return {
243 true,
244 nIndex,
245 utils::to_nd(nIndex, _params.size),
246 nDistance
247 };
248 }
249 else {
250
251 // _params.maze is a grid...
252 const auto& [nActive, nCoord](_n);
253
254 const auto nIndex = static_cast<typename params_t::index_type>(
255 nActive ? utils::to_1d(nCoord, _params.size) : typename params_t::index_type{}
256 );
257
258 constexpr auto nDistance = static_cast<typename params_t::scalar_type>(1);
259
260 return {
261 nActive,
262 nIndex,
263 nCoord,
264 nDistance
265 };
266 }

360 [[maybe_unused, nodiscard]]
361 static
362#if __cplusplus >= 202003L
363 constexpr
364#endif // __cplusplus >= 202003L
365 auto operator()(const params_t& _params) {
366
367 using solver_t = Derived<params_t>;
368
369 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
370 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
371
372 if (_params.maze.contains(s) && _params.maze.at(s).is_active() &&
373 _params.maze.contains(e) && _params.maze.at(e).is_active()
374 ) {
375 auto result = s != e ? solver_t::execute(_params) : std::vector<typename params_t::coord_type> { _params.end };
376
377 if constexpr (resettable<std::remove_reference_t<decltype(*_params.monotonic_pmr)>>::value) {
378 if (_params.monotonic_pmr != nullptr) { _params.monotonic_pmr->reset();}
379 }
380 if constexpr (resettable<std::remove_reference_t<decltype(*_params.polytonic_pmr)>>::value) {
381 if (_params.polytonic_pmr != nullptr) { _params.polytonic_pmr->reset(); }
382 }
383 if constexpr (resettable<std::remove_reference_t<decltype(*_params.pool_pmr)>>::value) {
384 if (_params.pool_pmr != nullptr) { _params.pool_pmr->reset(); }
385 }
386
387 return result;
388 }
389
390 return std::vector<typename params_t::coord_type>{};
391 }
