127 template <typename maze_neighbour_t>
128 static constexpr node_data get_data(maze_neighbour_t& _n, const params_t& _params) noexcept {
129
130 if constexpr (is_graph<decltype(_params.maze)>::value) {
131
132 // _params.maze is a graph...
133 const auto& [nIndex, nDistance](_n);
134
135 return {
136 true,
137 nIndex,
138 utils::to_nd(nIndex, _params.size),
139 nDistance
140 };
141 }
142 else {
143
144 // _params.maze is a grid...
145 const auto& [nActive, nCoord](_n);
146
147 const auto nIndex = static_cast<typename params_t::index_type>(
148 nActive ? utils::to_1d(nCoord, _params.size) : typename params_t::index_type{}
149 );
150
151 constexpr auto nDistance = static_cast<typename params_t::scalar_type>(1);
152
153 return {
154 nActive,
155 nIndex,
156 nCoord,
157 nDistance
158 };
159 }

162 static constexpr size_t determine_capacity(const params_t& _params) noexcept {
163
164 if constexpr (is_graph<decltype(_params.maze)>::value) {
165
166 return static_cast<size_t>(
167 _params.capacity != 0U ?
168 _params.capacity :
169 utils::max(_params.maze.count() / 10U, static_cast<size_t>(1U))
170 );
171 }
172 else {
173 return utils::max(
174 _params.capacity,
175 utils::max(
176 static_cast<size_t>(utils::to_1d(_params.start, _params.size)),
177 static_cast<size_t>(utils::to_1d(_params.end, _params.size))
178 )
179 );
180 }

196 [[maybe_unused, nodiscard]]
197#if __cplusplus >= 202003L
198 constexpr
199#endif // __cplusplus >= 202003L
200 auto operator()(const params_t& _params) const {
201
202 using solver_t = Derived<params_t>;
203
204 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
205 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
206
207 if (_params.maze.contains(s) && _params.maze.at(s).is_active() &&
208 _params.maze.contains(e) && _params.maze.at(e).is_active()
209 ) {
210 auto result = s != e ? solver_t::execute(_params) : std::vector<typename params_t::coord_type> { _params.end };
211
212 if constexpr (resettable<std::remove_reference_t<decltype(*_params.monotonic_pmr)>>::value) {
213 if (_params.monotonic_pmr != nullptr) { _params.monotonic_pmr->reset();}
214 }
215 if constexpr (resettable<std::remove_reference_t<decltype(*_params.polytonic_pmr)>>::value) {
216 if (_params.polytonic_pmr != nullptr) { _params.polytonic_pmr->reset(); }
217 }
218 if constexpr (resettable<std::remove_reference_t<decltype(*_params.pool_pmr)>>::value) {
219 if (_params.pool_pmr != nullptr) { _params.pool_pmr->reset(); }
220 }
221
222 return result;
223 }
224
225 return std::vector<typename params_t::coord_type>{};
226 }

248 template<typename node_t, typename coord_t>
249 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
250
251 std::vector<coord_t> result;
252
253 {
254 size_t depth = 0U;
255 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++depth) {}
256
257 result.resize(depth);
258 }
259
260 size_t i = 0U;
261 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++i) {
262 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
263 }
264
265 return result;
266 }

282 [[nodiscard]] static constexpr auto ibacktrack(const open_set_t& _open, const coord_t& _size) {
283
284 std::vector<coord_t> result;
285 result.reserve(_open.size());
286
287 for (auto it = _open.rbegin(); it != _open.rend(); ++it) {
288 result.emplace_back(utils::to_nd(it->m_index, _size));
