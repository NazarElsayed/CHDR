334 template<typename node_t, typename coord_t>
335 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
336
337 std::vector<coord_t> result;
338
339 {
340 size_t depth = 0U;
341 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++depth) {}
342
343 result.resize(depth);
344 }
345
346 size_t i = 0U;
347 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++i) {
348 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
349 }
350
351 return result;
352 }

467 if constexpr (solver_utils::template is_graph_v<decltype(_params.maze)>) {

488 constexpr auto nDistance = static_cast<typename params_t::scalar_type>(1);

491 nActive,
492 nIndex,

597 [[maybe_unused, nodiscard]]
598 static
599#if __cplusplus >= 202003L
600 constexpr
601#endif // __cplusplus >= 202003L
602 auto operator()(const params_t& _params) {
603
604 using solver_t = Derived<params_t>;
605
606 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
607 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
608
609 if (_params.maze.contains(s) && _params.maze.at(s).is_active() &&
610 _params.maze.contains(e) && _params.maze.at(e).is_active()
611 ) {
612 auto result = s != e ? solver_t::execute(_params) : std::vector<typename params_t::coord_type> { _params.end };
613
614 if constexpr (solver_utils::template has_method_reset_v<decltype(*_params.monotonic_pmr)>) {
615 if (_params.monotonic_pmr != nullptr) { _params.monotonic_pmr->reset(); }
616 }
617 if constexpr (solver_utils::template has_method_reset_v<decltype(*_params.polytonic_pmr)>) {
618 if (_params.polytonic_pmr != nullptr) { _params.polytonic_pmr->reset(); }
619 }
620 if constexpr (solver_utils::template has_method_reset_v<decltype(*_params.pool_pmr)>) {
621 if (_params.pool_pmr != nullptr) { _params.pool_pmr->reset(); }
622 }
623
624 return result;
625 }
626
627 return std::vector<typename params_t::coord_type>{};
628 }
