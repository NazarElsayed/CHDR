354 template<typename node_t, typename coord_t>
355 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
356
357 std::vector<coord_t> result{};
358
359 {
360 size_t depth = 0U;
361 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
362 ++depth;
363 }
364
365 result.resize(depth);
366 }
367
368 size_t i = 0U;
369 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
370 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
371 ++i;
372 }
373
374 return result;

592 [[maybe_unused, nodiscard]]
593#if __cplusplus >= 202003L
594 constexpr
595#endif // __cplusplus >= 202003L
596 static auto solve(const params_t& _params) {
597
598 std::vector<typename params_t::coord_type> result{};
599
600 std::exception_ptr exception { nullptr };
601
602 /*
603 * Run solver:
604 */
605
606 try {
607 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
608 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
609
610 if (_params.maze.contains(s) && _params.maze[s].is_active() &&
611 _params.maze.contains(e) && _params.maze[e].is_active()
612 ) {
613 result = s != e ?
614 solver_t<params_t>::invoke(_params) :
615 std::vector<typename params_t::coord_type> { _params.end };
616 }
617 }
618 catch (...) { // NOLINT(*-empty-catch)
619
620 // ReSharper disable once CppDFAUnusedValue
621 exception = std::current_exception();
622 }
623
624 /*
625 * Release resources with deferred exception handling:
626 */
627
628 try {
629 if constexpr (solver_utils::template has_method_reset_v<std::remove_pointer_t<std::decay_t<decltype(_params.monotonic_pmr)>>>) {
630 if (_params.monotonic_pmr != nullptr) { _params.monotonic_pmr->reset(); }
631 }
632 }
633 catch (...) { // NOLINT(*-empty-catch)
634 if (exception == nullptr) { exception = std::current_exception(); }
635 }
636
637 try {
638 if constexpr (solver_utils::template has_method_reset_v<std::remove_pointer_t<std::decay_t<decltype(_params.heterogeneous_pmr)>>>) {
639 if (_params.heterogeneous_pmr != nullptr) { _params.heterogeneous_pmr->reset(); }
640 }
641 }
642 catch (...) { // NOLINT(*-empty-catch)
643 if (exception == nullptr) { exception = std::current_exception(); }
644 }
645
646 try {
647 if constexpr (solver_utils::template has_method_reset_v<std::remove_pointer_t<std::decay_t<decltype(_params.homogeneous_pmr)>>>) {
648 if (_params.homogeneous_pmr != nullptr) { _params.homogeneous_pmr->reset(); }
649 }
650 }
651 catch (...) { // NOLINT(*-empty-catch)
652 if (exception == nullptr) { exception = std::current_exception(); }
653 }
654
655 // Report first exception to occur:
656 if (exception != nullptr) {
657 std::rethrow_exception(exception);
658 }
659
660 return result;
661 }
