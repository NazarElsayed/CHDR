339 template<typename node_t, typename coord_t>
340 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
341
342 std::vector<coord_t> result;
343
344 {
345 size_t depth = 0U;
346 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++depth) {}
347
348 result.resize(depth);
349 }
350
351 size_t i = 0U;
352 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent), ++i) {
353 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
354 }
355
356 return result;
357 }

604#if __cplusplus >= 202003L
605 constexpr
606#endif // __cplusplus >= 202003L
607 auto operator()(const params_t& _params) {
608
609 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
610 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
611
612 if (_params.maze.contains(s) && _params.maze.at(s).is_active() &&
613 _params.maze.contains(e) && _params.maze.at(e).is_active()
614 ) {
615 auto result = s != e ? solver_t<params_t>::invoke(_params) : std::vector<typename params_t::coord_type> { _params.end };
616
617 if constexpr (solver_utils::template has_method_reset_v<decltype(*_params.monotonic_pmr)>) {
618 if (_params.monotonic_pmr != nullptr) { _params.monotonic_pmr->reset(); }
619 }
620 if constexpr (solver_utils::template has_method_reset_v<decltype(*_params.polytonic_pmr)>) {
621 if (_params.polytonic_pmr != nullptr) { _params.polytonic_pmr->reset(); }
622 }
623 if constexpr (solver_utils::template has_method_reset_v<decltype(*_params.pool_pmr)>) {
624 if (_params.pool_pmr != nullptr) { _params.pool_pmr->reset(); }
625 }
626
627 return result;
628 }
629
630 return std::vector<typename params_t::coord_type>{};
631 }
