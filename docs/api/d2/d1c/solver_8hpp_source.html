547 template<typename node_t, typename coord_t>
548 static constexpr auto rbacktrack(const node_t& _node, const coord_t& _size) {
549
550 std::vector<coord_t> result{};
551
552 {
553 size_t depth = 0U;
554 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
555 ++depth;
556 }
557
558 result.resize(depth);
559 }
560
561 size_t i = 0U;
562 for (const auto* RESTRICT t = &_node; t->m_parent != nullptr; t = static_cast<const node_t*>(t->m_parent)) {
563 result[(result.size() - 1U) - i] = utils::to_nd(t->m_index, _size);
564 ++i;
565 }
566
567 return result;
568 }

867 [[maybe_unused, nodiscard]]
868#if __cplusplus >= 202003L
869 constexpr
870#endif // __cplusplus >= 202003L
871 static auto solve(const params_t& _params) {
872
873 std::vector<typename params_t::coord_type> result{};
874
875 std::exception_ptr exception { nullptr };
876
877 /*
878 * Run solver:
879 */
880
881 try {
882 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
883 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
884
885 if (_params.maze.contains(s) && _params.maze[s].is_active() &&
886 _params.maze.contains(e) && _params.maze[e].is_active()
887 ) {
888
889 result = s != e ?
890 solver_t<params_t>::invoke(_params) :
891 std::vector<typename params_t::coord_type> { _params.end };
892 }
893 }
894 catch (...) { // NOLINT(*-empty-catch)
895
896 // ReSharper disable once CppDFAUnusedValue
897 exception = std::current_exception();
898 }
899
900 // Reset memory resources:
901 if constexpr (!params_t::no_cleanup::value) {
902 try {
903 solver_utils::reset_resources(_params);
904 }
905 catch (...) { // NOLINT(*-empty-catch)
906 if (exception == nullptr) { exception = std::current_exception(); }
907 }
908 }
909
910 // Report first exception to occur:
911 if (exception != nullptr) {
912 std::rethrow_exception(exception);
913 }
914
915 return result;
916 }
