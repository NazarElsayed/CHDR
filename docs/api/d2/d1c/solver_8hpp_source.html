585 [[maybe_unused, nodiscard]]
586#if __cplusplus >= 202003L
587 constexpr
588#endif // __cplusplus >= 202003L
589 static auto solve(const params_t& _params) {
590
591 std::exception_ptr exception;
592
593 try {
594 const auto s = static_cast<typename params_t::index_type>(utils::to_1d(_params.start, _params.size));
595 const auto e = static_cast<typename params_t::index_type>(utils::to_1d(_params.end, _params.size));
596
597 if (_params.maze.contains(s) && _params.maze.at(s).is_active() &&
598 _params.maze.contains(e) && _params.maze.at(e).is_active()
599 ) {
600 return s != e ? solver_t<params_t>::invoke(_params) : std::vector<typename params_t::coord_type> { _params.end };
601 }
602 }
603 catch (...) { // NOLINT(*-empty-catch)
604 exception = std::current_exception();
605 }
606
607 try {
608 if constexpr (solver_utils::template has_method_reset_v<std::remove_pointer_t<std::decay_t<decltype(_params.monotonic_pmr)>>>) {
609 if (_params.monotonic_pmr != nullptr) { _params.monotonic_pmr->reset(); }
610 }
611 if constexpr (solver_utils::template has_method_reset_v<std::remove_pointer_t<std::decay_t<decltype(_params.polytonic_pmr)>>>) {
612 if (_params.polytonic_pmr != nullptr) { _params.polytonic_pmr->reset(); }
613 }
614 if constexpr (solver_utils::template has_method_reset_v<std::remove_pointer_t<std::decay_t<decltype(_params.homogeneous_pmr)>>>) {
615 if (_params.homogeneous_pmr != nullptr) { _params.homogeneous_pmr->reset(); }
616 }
617 }
618 catch (...) { // NOLINT(*-empty-catch)
619 if (!exception) {
620 exception = std::current_exception();
621 }
622
623 std::rethrow_exception(exception);
624 }
625
626 return std::vector<typename params_t::coord_type>{};
627 }
