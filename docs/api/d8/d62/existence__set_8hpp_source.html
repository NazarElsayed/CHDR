55 constexpr void enable(const size_t& _hash) {
56
57 if (_hash >= c.size()) {
58 resize(_hash + 1U);
59 }
60
61#if defined(GCC)
62#pragma GCC diagnostic push
63#pragma GCC diagnostic ignored "-Wstringop-overflow"
64#endif
65
66 c[_hash] = static_cast<boolean_t>(true);
67
68#if defined(GCC)
69#pragma GCC diagnostic pop
70#endif
71 }
72
73 constexpr void disable(const size_t& _hash) noexcept {
74
75 if (_hash < c.size()) {
76 c[_hash] = static_cast<boolean_t>(false);
77 }
78 }

104 [[maybe_unused]] constexpr existence_set(const std::initializer_list<size_t>& _items, const size_t& _capacity = 0U) {
105
106 size_t autoCapacity = _capacity;
107
108 if (autoCapacity < 1U) {
109 autoCapacity = utils::max<size_t>(_items.size(), 1U);

112 reserve(autoCapacity);

114 for (const auto& item : _items) {
