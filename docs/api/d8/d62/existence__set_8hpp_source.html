108 constexpr void enable(size_t _hash) {
109
110 if (_hash >= c.size()) {
111 resize(_hash + 1U);
112 }
113
114 /*
115 * Suppress GCC-specific warnings about potential string overflow in this section of code.
116 *
117 * As 'width_t' can be a valid alias for 'char' type, linting tools occasionally
118 * misidentify this code as string manipulation.
119 */
120#if defined(GCC)
121#pragma GCC diagnostic push
122#pragma GCC diagnostic ignored "-Wstringop-overflow"
123#endif
124
125 c[_hash] = static_cast<width_t>(true);
126
127#if defined(GCC)
128#pragma GCC diagnostic pop
129#endif
130 }
131
138 constexpr void disable(size_t _hash) noexcept {
139 if (_hash < c.size()) {
140 c[_hash] = static_cast<width_t>(false);
