58 constexpr void enable(const size_t& _hash) {
59
60 if (_hash >= c.size()) {
61 resize(_hash + 1U);
62 }
63
64#if defined(GCC)
65#pragma GCC diagnostic push
66#pragma GCC diagnostic ignored "-Wstringop-overflow"
67#endif
68
69 c[_hash] = static_cast<boolean_t>(true);
70
71#if defined(GCC)
72#pragma GCC diagnostic pop
73#endif
74 }
75
76 constexpr void disable(const size_t& _hash) noexcept {
77
78 if (_hash < c.size()) {
79 c[_hash] = static_cast<boolean_t>(false);
80 }
81 }

107 [[maybe_unused]] constexpr existence_set(const std::initializer_list<size_t>& _items, const size_t& _capacity = 0U) {
108
109 size_t autoCapacity = _capacity;
110
111 if (autoCapacity < 1U) {
112 autoCapacity = utils::max<size_t>(_items.size(), 1U);

115 reserve(autoCapacity);

117 for (const auto& item : _items) {
