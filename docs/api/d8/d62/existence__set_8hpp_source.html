53 template <typename layout_t = high_performance>
54 class existence_set {
55
56 // Validate the template parameter.
57 static_assert(
58 std::is_same_v<layout_t, low_memory_usage> ||
59 std::is_same_v<layout_t, high_performance>,
60 "layout_t must be one of the following: low_memory_usage, high_performance"

69 using boolean_t = std::conditional_t<std::is_same_v<layout_t, low_memory_usage>, bool, char>;
70
71 private:
80 std::pmr::vector<boolean_t> c;
81
88 constexpr void enable(size_t _hash) {
89
90 if (_hash >= c.size()) {
91 resize(_hash + 1U);
92 }
93
94 /*
95 * Suppress GCC-specific warnings about potential string overflow in this section of code.
96 *
97 * As 'boolean_t' can be a valid alias for 'char' type, linting tools occasionally
98 * misidentify this code as string manipulation.
99 */
100#if defined(GCC)
101#pragma GCC diagnostic push
102#pragma GCC diagnostic ignored "-Wstringop-overflow"
103#endif
104
105 c[_hash] = static_cast<boolean_t>(true);
106
107#if defined(GCC)
108#pragma GCC diagnostic pop
109#endif
110 }
111
118 constexpr void disable(size_t _hash) noexcept {
119 if (_hash < c.size()) {
120 c[_hash] = static_cast<boolean_t>(false);
121 }
122 }

263 [[maybe_unused]] constexpr void trim() {
264
265 auto it = c.rbegin();
266 while (it != c.rend() && !static_cast<boolean_t>(*it)) {
267 ++it;

269 c.erase(it.base(), c.end());
