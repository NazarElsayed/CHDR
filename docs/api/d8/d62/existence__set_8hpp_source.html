98 constexpr void enable(size_t _hash) {
99
100 if (_hash >= c.size()) {
101 resize(_hash + 1U);
102 }
103
104 /*
105 * Suppress GCC-specific warnings about potential string overflow in this section of code.
106 *
107 * As 'boolean_t' can be a valid alias for 'char' type, linting tools occasionally
108 * misidentify this code as string manipulation.
109 */
110#if defined(GCC)
111#pragma GCC diagnostic push
112#pragma GCC diagnostic ignored "-Wstringop-overflow"
113#endif
114
115 c[_hash] = static_cast<boolean_t>(true);
116
117#if defined(GCC)
118#pragma GCC diagnostic pop
119#endif
120 }
121
128 constexpr void disable(size_t _hash) noexcept {
129 if (_hash < c.size()) {
130 c[_hash] = static_cast<boolean_t>(false);
131 }
132 }

234 template <typename T>
235 HOT constexpr void push(const T& _item) {
236
237 static_assert(std::is_same_v<decltype(std::hash<T>{}(std::declval<T>())), size_t>,
238 "Type T must be supported by std::hash and return size_t");
239
240 enable(std::hash<T>{}(_item));

288 [[maybe_unused]] constexpr void trim() {
289
