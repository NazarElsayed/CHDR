165
166 bool result = false;
167 if (!_expunct.empty()) { // LOSSLESS:
168
169 _params.homogeneous_pmr->deallocate(std::move(_expunct.top()), sizeof(node), alignof(node)); _dynamic_allocations--;
170 _expunct.pop();
171 }
172 else if (!_open.empty()) { // LOSSY:
173
174 auto worst_node = std::prev(_open.end());
175 bitwise_regression(worst_node->m_parent, _closed);
176 _closed.erase(worst_node->m_index);
177 _open.erase(worst_node);
178 }
179 else {
180 result = true;
181 }

186 template <typename open_set_t, typename closed_set_t>
187 [[maybe_unused, nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
188
189 constexpr bool optimising = true;
190 constexpr bool do_reverse = !solver_t::solver_utils::template is_graph_v<decltype(_params.maze)> && params_t::reverse_equivalence::value;
191
192 const auto& end = do_reverse ? _params.start : _params.end;
193 const auto& start = do_reverse ? _params.end : _params.start;
194
195 const auto s = utils::to_1d(start, _params.size);
196 const auto e = utils::to_1d(end, _params.size);
197
198 _open.emplace(s, static_cast<scalar_t>(0), _params.h(start, end) * _params.weight);
199 _closed.emplace(s);
200
201 size_t dynamic_allocations(0U);
202 size_t closed_allocations(1U);
203
204 // Define heuristic for quantifying memory usage:
205 const auto memory_usage = [&_open, &closed_allocations, &dynamic_allocations]() ALWAYS_INLINE {
206 return _open.size() + closed_allocations + dynamic_allocations;
207 };
208
209 stack<node*> expunct(_params.heterogeneous_pmr);
210
211 std::optional<node> best_solution;
212
213 // Main loop:
214 while (LIKELY(!_open.empty())) {
215
216 auto curr(_open.extract(_open.begin()).value());
217
218 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
219
220 node* RESTRICT curr_ptr(nullptr);
221
222 if (!best_solution.has_value() || curr.m_gScore < best_solution->m_gScore) {
223
224 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
225
226 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
227
228 // Check if node is not already visited:
229 if (!_closed.contains(n.index)) {
230
231 // Attempt to resolve the issue of memory saturation.
232 bool full = memory_usage() >= _params.memory_limit;
233 if (full) {
234 full = desaturate(_open, _closed, expunct, dynamic_allocations, _params);
235 }
236
237 if (!full) {
238 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
239
240 if (curr_ptr == nullptr) {
241 curr_ptr = new (_params.homogeneous_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr)); dynamic_allocations++;
242 }
243
244 _open.emplace(n.index, curr_ptr->m_gScore + n.distance, _params.h(n.coord, end) * _params.weight, curr_ptr);
245 }
246 else {
247 // Memory saturated. Backup losslessly...
248 bitwise_regression(curr.m_parent, _closed);
249 break;
250 }
251 }
252 }
253 }
254 }
255
256 if (curr_ptr == nullptr) {
257
258 if (auto* p = curr.forget_one(); p != nullptr) {
259 expunct.emplace(static_cast<node*>(p));
260 }
261 }
262 }
263 else { // SOLUTION REACHED...
264
265 if (!best_solution.has_value() || curr.m_gScore < best_solution->m_gScore) {
266 best_solution.emplace(std::move(curr));
267
268 if constexpr (optimising) { // Optimising mechanism:
269
270 bool full = memory_usage() >= _params.memory_limit;
271
272 // ReSharper disable once CppUsingResultOfAssignmentAsCondition
273 if (full && ((full = !_open.empty()))) {
274 _open.clear();
275 }
276
277 if (!full) {
278 _open.emplace(s, static_cast<scalar_t>(0), _params.h(start, end) * _params.weight);
279 }
280 }
281 }
282 }
283 }
284
285 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
286 _open.wipe();
287 }
288 else {
289 _open = open_set_t{};
290 }
291 _closed = closed_set_t{};
292
293 if (best_solution.has_value()) {
294
295 if constexpr (params_t::reverse_equivalence::value) {
296 return solver_t::solver_utils::rbacktrack(best_solution.value(), _params.size, best_solution->m_gScore);
297 }
298 else {
299 return solver_t::solver_utils::rbacktrack_noreverse(best_solution.value(), _params.size, best_solution->m_gScore);
300 }
301 }
302 else {
303 return std::vector<coord_t>{};
304 }
305 }

307 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
308
309 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
310
311 existence_set closed(_params.monotonic_pmr);
312 closed.reserve(capacity);
313
314 std::pmr::multiset<node> open(_params.heterogeneous_pmr);

316 return solve_internal(open, closed, capacity, _params);
317 }
