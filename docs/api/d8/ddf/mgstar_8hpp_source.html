105 HOT static void bitwise_regression(const managed_node<index_t, node>* _parent, closed_set_t& _closed) {
106 auto* p = _parent;
107 while (p != nullptr && _closed.contains(p->m_index)) {
108 _closed.erase(p->m_index);
109 p = p->m_parent;

114 [[nodiscard]] HOT static auto desaturate(open_set_t& _open, closed_set_t& _closed, expunct_set_t& _expunct, size_t& _dynamic_allocations, const params_t& _params) {
115
116 bool result = false;
117 if (!_expunct.empty()) { // LOSSLESS:
118
119 _params.homogeneous_pmr->deallocate(std::move(_expunct.top()), sizeof(node), alignof(node)); _dynamic_allocations--;
120 _expunct.pop();
121 }
122 else if (!_open.empty()) { // LOSSY:
123
124 auto worst_node = std::prev(_open.end());
125 bitwise_regression(worst_node->m_parent, _closed);
126 _closed.erase(worst_node->m_index);
127 _open.erase(worst_node);
128 }
129 else {
130 result = true;
131 }

136 template <typename open_set_t, typename closed_set_t>
137 [[maybe_unused, nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
138
139 const auto s = utils::to_1d(_params.start, _params.size);
140 const auto e = utils::to_1d(_params.end, _params.size);
141
142 _open.emplace(s, static_cast<scalar_t>(0), _params.h(_params.start, _params.end) * _params.weight);
143 _closed.emplace(s);
144
145 size_t dynamic_allocations(0U);
146 size_t closed_allocations(1U);
147
148 // Define heuristic for quantifying memory usage:
149 const auto memory_usage = [&_open, &closed_allocations, &dynamic_allocations]() ALWAYS_INLINE {
150 return _open.size() + closed_allocations + dynamic_allocations;
151 };
152
153 stack<node*> expunct;
154
155 std::optional<node> best_solution;
156
157 // Main loop:
158 while (LIKELY(!_open.empty())) {
159
160 auto curr(_open.extract(_open.begin()).value());
161
162 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
163
164 node* RESTRICT curr_ptr(nullptr);
165
166 if (!best_solution.has_value() || curr.m_gScore < best_solution->m_gScore) {
167
168 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
169
170 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
171
172 // Check if node is not already visited:
173 if (!_closed.contains(n.index)) {
174
175 // Attempt to resolve the issue of memory saturation.
176 bool full = memory_usage() >= _params.memory_limit;
177 if (full) {
178 full = desaturate(_open, _closed, expunct, dynamic_allocations, _params);
179 }
180
181 if (!full) {
182 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
183
184 if (curr_ptr == nullptr) {
185 curr_ptr = new (_params.homogeneous_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr)); dynamic_allocations++;
186 }
187
188 _open.emplace(n.index, curr_ptr->m_gScore + n.distance, _params.h(n.coord, _params.end) * _params.weight, curr_ptr);
189 }
190 else {
191 // Memory saturated. Backup losslessly...
192 bitwise_regression(curr.m_parent, _closed);
193 break;
194 }
195 }
196 }
197 }
198 }
199
200 if (curr_ptr == nullptr) {
201
202 if (auto* p = curr.forget_one(); p != nullptr) {
203 expunct.emplace(static_cast<node*>(p));
204 }
205 }
206 }
207 else { // SOLUTION REACHED...
208
209 if (!best_solution.has_value() || curr.m_gScore < best_solution->m_gScore) {
210 best_solution.emplace(std::move(curr));
211
212 bool full = memory_usage() >= _params.memory_limit;
213 if (full && (full = !_open.empty())) {
214 _open.clear();
215 }
216
217 if (!full) {
218 _open.emplace(s, static_cast<scalar_t>(0), _params.h(_params.start, _params.end) * _params.weight);
219 }
220 }
221 }
222 }
223
224 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
225 _open.wipe();
226 }
227 else {
228 _open = open_set_t{};
229 }
230 _closed = closed_set_t{};
231
232 return best_solution.has_value() ?
233 solver_t::solver_utils::rbacktrack(best_solution.value(), _params.size, best_solution.value().m_gScore) :
234 std::vector<coord_t>{};
235 }

237 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
238
239 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
240
241 existence_set closed(_params.monotonic_pmr);
242 closed.reserve(capacity);
243
244 std::pmr::multiset<node> open(_params.heterogeneous_pmr);
245
246 return solve_internal(open, closed, capacity, _params);
247 }
