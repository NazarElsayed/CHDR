132 template <typename open_set_t>
133 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, const params_t& _params) {
134
135 using neighbours_t = decltype(_params.maze.get_neighbours(std::declval<index_t>()));
136
137 const auto& end = params_t::reverse_equivalence::value ? _params.start : _params.end;
138 const auto& start = params_t::reverse_equivalence::value ? _params.end : _params.start;
139
140 const auto s = utils::to_1d(start, _params.size);
141 const auto e = utils::to_1d(end, _params.size);
142
143 auto bound = _params.h(start, end) * _params.weight;
144
145 _open.emplace_back(s, static_cast<scalar_t>(0), bound);
146
147 stack<state<neighbours_t>> stack(_params.heterogeneous_pmr);
148
149 transposition_table_t transposition_table(_params.heterogeneous_pmr);
150
151 do {
152 stack.emplace(_open.back(), _params);
153 transposition_table[_open.back().m_index] = bound;
154
155 auto min = std::numeric_limits<scalar_t>::max();
156
157 // Main loop:
158 while (!stack.empty()) {
159
160 auto& _ = stack.top();
161 auto& curr = _.curr;
162
163 if (curr.m_fScore <= bound) {
164
165 if (_.neighbours_idx != _.neighbours.size()) {
166
167 if (const auto& n = solver_t::get_data(_.neighbours[(_.neighbours.size() - 1U) - (_.neighbours_idx++)], _params); n.active) {
168
169 auto g = curr.m_gScore + n.distance;
170 auto f = g + (_params.h(n.coord, end) * _params.weight);
171
172 auto search = transposition_table.find(n.index);
173 if (!(search != transposition_table.end() && f >= search->second)) {
174 transposition_table[n.index] = f;
175
176 _open.emplace_back(n.index, g, f);
177
178 if (n.index != e) { // SEARCH FOR SOLUTION...
179 stack.emplace(_open.back(), _params);
180 }
181 else { // SOLUTION REACHED...
182
183 // ReSharper disable once CppDFAUnusedValue
184 transposition_table = {};
185
186 if constexpr (params_t::reverse_equivalence::value) {
187 return solver_t::solver_utils::ibacktrack(_open, _params.size);
188 }
189 else {
190 return solver_t::solver_utils::ibacktrack_noreverse(_open, _params.size);
191 }
192 }
193 }
194 }
195 }
196 else {
197 if (_open.size() > 1U) {
198 _open.pop_back();
199 }
200 stack.pop();
201 }
202 }
203 else {
204 min = utils::min(min, curr.m_fScore);
205 if (_open.size() > 1U) {
206 _open.pop_back();
207 }
208 stack.pop();
209 }
210 }
211
212 bound = min;
213
214 stack.clear();
215 transposition_table.clear();
216 }
217 while (bound != std::numeric_limits<scalar_t>::max());
218
219 return std::vector<coord_t>{};
220 }

222 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
223
224 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
225
226 std::pmr::vector<node> open(_params.heterogeneous_pmr);
227 try {
228 open.reserve(capacity / 8U);
229 }
230 catch (...) {} // NOLINT(*-empty-catch)
231
232 return solve_internal(open, _params);
233 }
