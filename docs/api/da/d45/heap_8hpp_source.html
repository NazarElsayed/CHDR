362 [[maybe_unused]] constexpr void erase(const T& _item) noexcept {
363
364 assert(!empty() && "Heap is empty");
365
366 auto i = index_of(_item);
367
368 assert(i < size() && "(Out of Bounds) Item does not exist in Heap.");
369
370 if (i < size()) {
371
372 if (i == size() - 1U) {
373 c.pop_back(); // If the item to remove is the last item, just remove it.
374 }
375 else {
376 c[i] = std::move(c.back());
377 c.pop_back();
378
379 // Restore heap property:
380 if (size() > 1U) {
381
382 if (i > 1U && comp(c[i], c[i / Kd])) {
383 sort_up(c[i]);
384 }
385 else {
386 sort_down(c[i]);
387 }
388 }
389 }
390 }

469 HOT constexpr void sort_up(const T& _item) noexcept {
470
471 if (size() > 1U) {
472
473 auto i = index_of(_item);
474
475 assert(i < size() && "(Out of Bounds) Item does not exist in Heap.");
476
477 while (i > 1U) {
478 const auto p = i / Kd;
479
480 if (comp(c[p], c[i])) {
481 std::swap(c[i], c[p]);
482 i = p;
483 }
484 else {
485 break;
486 }
487 }
488 }

500 HOT constexpr void sort_down(const T& _item) noexcept {
501
502 if (size() > 1U) {
503
504 auto i = index_of(_item);
505
506 assert(i < size() && "(Out of Bounds) Item does not exist in Heap.");
507
508 while (i > 1U) {
509 const auto c0 = i * Kd;
510 const auto cn = c0 + (Kd - 1U);
511
512 if (cn < c.size()) {
513 size_t min{};
514
515 if constexpr (Kd == 2U) {
516 min = (cn < c.size() && comp(c[c0], c[cn])) ? cn : c0;
517 }
518 else {
519 min = i;
520
521 for (auto j = c0; j <= cn && j < c.size(); ++j) {
522 if (comp(c[min], c[j])) {
523 min = j;
524 }
525 }
526 }
527
528 if (comp(c[i], c[min])) {
529 std::swap(c[i], c[min]);
530 i = min;
531 }
532 else {
533 break;
534 }
535 }
536 else {
537 break;
538 }
539 }

693 [[maybe_unused, nodiscard]] constexpr iterator_t lower_bounds(const T& _value) noexcept {
694
695 if (!empty()) {
696
697 for (size_t i = 1U; i < c.size();) { // Start from index 1 to skip the super element.
698
699 if (!comp(c[i], _value)) {
700 return c.begin() + i; // Return the first element not less than the given value.
701 }
702
703 // Iterate through child nodes based on the Kd dimension.
704 size_t child_start = i * Kd;
705 size_t child_end = std::min(child_start + Kd, c.size());
706 bool traversed = false;
707
708 for (size_t j = child_start; j < child_end; ++j) {
709 if (j < c.size() && !comp(c[j], _value)) {
710 i = j; // Traverse to the first valid child node.
711 traversed = true;
712 break;
713 }
714 }
715
716 if (!traversed) {
717 break; // Exit loop if no valid traversal path exists.
718 }
719 }
720 }
721
722 return c.end(); // Return end if no matching element is found.
723 }

759 constexpr iterator_t erase(iterator_t _position) {
760
761 auto distance_from_start = std::distance(c.begin(), _position);
762 auto size_before = size();
763
764 if (distance_from_start < size() + 1U) { // Ensure valid offset for super element
765 if (distance_from_start == size()) {
766 c.pop_back(); // If last element, simply remove
767 }
768 else {
769 c[distance_from_start] = std::move(c.back());
770 c.pop_back();
771 if (size_before > 1U) {
772 if (distance_from_start > 1U && comp(c[distance_from_start], c[distance_from_start / Kd])) {
773 sort_up(c[distance_from_start]);
774 }
775 else {
776 sort_down(c[distance_from_start]);
777 }
778 }
779 }
780 }
781
782 return c.begin() + distance_from_start; // Return iterator to next element after erasure
783 }
