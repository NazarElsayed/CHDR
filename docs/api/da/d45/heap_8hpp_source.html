420 [[maybe_unused]] constexpr void erase(const T& _item) noexcept {
421
422 assert(!empty() && "Heap is empty");
423
424 auto i = index_of(_item);
425
426 assert(i < size() && "(Out of Bounds) Item does not exist in Heap.");
427
428 if (i < size()) {
429
430 if (i == size() - 1U) {
431 c.pop_back(); // If the item to remove is the last item, just remove it.
432 }
433 else {
434 c[i] = std::move(c.back());
435 c.pop_back();
436
437 // Restore heap property:
438 if (size() > 1U) {
439
440 if (i > 1U && comp(c[i], c[i / Kd])) {
441 sort_up(c[i]);
442 }
443 else {
444 sort_down(c[i]);
445 }
446 }
447 }
448 }

526 HOT constexpr void sort_up(const T& _item) noexcept {
527
528 if (size() > 1U) {
529
530 auto i = index_of(_item);
531
532 assert(i < size() && "(Out of Bounds) Item does not exist in Heap.");
533
534 auto value_to_insert = std::move(c[i]);
535
536 while (i > 1U) {
537 const auto p = i / Kd;
538
539 if (comp(c[p], value_to_insert)) {
540 c[i] = std::move(c[p]);
541 i = p;
542 }
543 else {
544 break;
545 }
546 }
547
548 c[i] = std::move(value_to_insert);
549 }
550 }

561 HOT constexpr void sort_down(const T& _item) noexcept {
562
563 if (size() > 1U) {
564
565 auto i = index_of(_item);
566 assert(i < size() && "(Out of Bounds) Item does not exist in Heap.");
567
568 auto value_to_insert = std::move(c[i]);
569
570 const auto c0 = i * Kd;
571 const auto cn = c0 + (Kd - 1U);
572
573 while (i > 1U) {
574
575 if (c0 < c.size()) {
576 size_t max = c0;
577
578 if constexpr (Kd == 2U) {
579 max = (cn < c.size() && comp(c[c0], c[cn])) ? cn : c0;
580 }
581 else {
582 for (auto j = c0; j <= cn && j < c.size(); ++j) {
583 if (comp(c[max], c[j])) {
584 max = j;
585 }
586 }
587 }
588
589 if (comp(value_to_insert, c[max])) {
590 c[i] = std::move(c[max]);
591 i = max;
592 }
593 else {
594 break;
595 }
596 }
597 else {
598 break;
599 }
600 }
601
602 c[i] = std::move(value_to_insert);
603 }

756 [[maybe_unused, nodiscard]] constexpr iterator_t lower_bounds(const T& _value) noexcept {
757
758 if (!empty()) {
759
760 for (size_t i = 1U; i < c.size();) { // Start from index 1 to skip the super element.
761
762 if (!comp(c[i], _value)) {
763 return c.begin() + i; // Return the first element not less than the given value.
764 }
765
766 // Iterate through child nodes based on the Kd dimension.
767 size_t child_start = i * Kd;
768 size_t child_end = std::min(child_start + Kd, c.size());
769 bool traversed = false;
770
771 for (size_t j = child_start; j < child_end; ++j) {
772 if (j < c.size() && !comp(c[j], _value)) {
773 i = j; // Traverse to the first valid child node.
774 traversed = true;
775 break;
776 }
777 }
778
779 if (!traversed) {
780 break; // Exit loop if no valid traversal path exists.
781 }
782 }
783 }
784
785 return c.end(); // Return end if no matching element is found.
786 }

822 constexpr iterator_t erase(iterator_t _position) {
823
824 auto distance_from_start = std::distance(c.begin(), _position);
825 auto size_before = size();
826
827 if (distance_from_start < size() + 1U) { // Ensure valid offset for super element
828 if (distance_from_start == size()) {
829 c.pop_back(); // If last element, simply remove
830 }
831 else {
832 c[distance_from_start] = std::move(c.back());
833 c.pop_back();
834 if (size_before > 1U) {
835 if (distance_from_start > 1U && comp(c[distance_from_start], c[distance_from_start / Kd])) {
836 sort_up(c[distance_from_start]);
837 }
838 else {
839 sort_down(c[distance_from_start]);
840 }
841 }
842 }
843 }
844
845 return c.begin() + distance_from_start; // Return iterator to next element after erasure
846 }
