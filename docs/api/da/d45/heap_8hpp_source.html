424 [[maybe_unused]] constexpr void erase(const T& _item) noexcept {
425
426 assert(!empty() && "Heap is empty");
427
428 auto i = index_of(_item);
429
430 assert(i < c.size() && "(Out of Bounds) Item does not exist in Heap.");
431
432 if (i < size()) {
433
434 if (i == size() - 1U) {
435 c.pop_back(); // If the item to remove is the last item, just remove it.
436 }
437 else {
438 c[i] = std::move(c.back());
439 c.pop_back();
440
441 // Restore heap property:
442 if (size() > 1U) {
443
444 if (i > 1U && comp(c[i], c[i / Kd])) {
445 sort_up(c[i]);
446 }
447 else {
448 sort_down(c[i]);
449 }
450 }
451 }

528 HOT constexpr void sort_up(const T& _item) noexcept {
529
530 if (size() > 1U) {
531
532 auto i = index_of(_item);
533
534 assert(i < c.size() && "(Out of Bounds) Item does not exist in Heap.");
535
536 auto value_to_insert = std::move(c[i]);
537
538 while (i > 1U) {
539 const auto p = i / Kd;
540
541 if (comp(c[p], value_to_insert)) {
542 c[i] = std::move(c[p]);
543 i = p;
544 }
545 else {
546 break;
547 }
548 }
549
550 c[i] = std::move(value_to_insert);
551 }
552 }

563 HOT constexpr void sort_down(const T& _item) noexcept {
564
565 if (size() > 1U) {
566
567 auto i = index_of(_item);
568 assert(i < c.size() && "(Out of Bounds) Item does not exist in Heap.");
569
570 auto value_to_insert = std::move(c[i]); // Save value to reposition.
571
572 while (true) {
573
574 const size_t c0 = i * Kd; // First child index
575 const size_t cn = c0 + (Kd - 1U); // Last child index in range
576
577 if (c0 < c.size()) {
578 PREFETCH(&c[c0], _MM_HINT_NTA);
579 PREFETCH(&c[cn], _MM_HINT_NTA);
580
581 // Find best candidate child to swap with:
582 size_t best_child = c0;
583 for (size_t j = c0 + 1U; j <= cn && j < c.size(); ++j) {
584 if (comp(c[best_child], c[j])) {
585 best_child = j;
586 }
587 }
588
589 // If value to insert is less favorable than the best child:
590 if (comp(value_to_insert, c[best_child])) {
591 c[i] = std::move(c[best_child]);
592 i = best_child;
593 continue;
594 }
595 }
596
597 c[i] = std::move(value_to_insert);
598 break;
599 }
600 }

749 [[maybe_unused, nodiscard]] constexpr iterator_t lower_bounds(const T& _value) noexcept {
750
751 if (!empty()) {
752
753 for (size_t i = 1U; i < c.size();) { // Start from index 1 to skip the super element.
754
755 if (!comp(c[i], _value)) {
756 return c.begin() + i; // Return the first element not less than the given value.
757 }
758
759 // Iterate through child nodes based on the Kd dimension.
760 const size_t child_start = i * Kd;
761 const size_t child_end = std::min(child_start + Kd, c.size());
762
763 bool traversed = false;
764 for (size_t j = child_start; j < child_end; ++j) {
765 if (j < c.size() && !comp(c[j], _value)) {
766 i = j; // Traverse to the first valid child node.
767 traversed = true;
768 break;
769 }
770 }
771
772 if (!traversed) {
773 break; // Exit loop if no valid traversal path exists.
774 }
775 }
776 }
777
778 return c.end(); // Return end if no matching element is found.
779 }

860 constexpr iterator_t erase(iterator_t _position) {
861
862 const auto distance_from_start = static_cast<unsigned>(std::distance(c.begin(), _position));
863 const auto size_before = size();
864
865 if (distance_from_start < size() + 1U) { // Offset for super element.
866
867 if (distance_from_start == size()) {
868 c.pop_back(); // Remove if last element.
869 }
870 else {
871
872 c[distance_from_start] = std::move(c.back());
873 c.pop_back();
874 if (size_before > 1U) {
875 if (distance_from_start > 1U && comp(c[distance_from_start], c[distance_from_start / Kd])) {
876 sort_up(c[distance_from_start]);
877 }
878 else {
879 sort_down(c[distance_from_start]);
880 }
881 }
882 }
883 }
884
885 return c.begin() + distance_from_start; // Return iterator to next element after erasure
