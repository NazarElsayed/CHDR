79 template <typename open_set_t, typename closed_set_t>
80 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
81
82 constexpr bool do_reverse = !solver_t::solver_utils::template is_graph_v<decltype(_params.maze)> && params_t::reverse_equivalence::value;
83
84 const auto& end = do_reverse ? _params.start : _params.end;
85 const auto& start = do_reverse ? _params.end : _params.start;
86
87 const auto s = utils::to_1d(start, _params.size);
88 const auto e = utils::to_1d(end, _params.size);
89
90 _open.emplace(s);
91 _closed.emplace(s);
92
93 // Main loop:
94 while (LIKELY(!_open.empty())) {
95
96 auto curr(std::move(_open.top()));
97 _open.pop();
98
99 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
100
101 node* RESTRICT curr_ptr(nullptr);
102
103 for (const auto& n_data : _params.maze.template get_neighbours<params_t::octile_neighbours::value>(curr.m_index)) {
104
105 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
106
107 // Check if node is not already visited:
108 if (!_closed.contains(n.index)) {
109 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
110
111 if (curr_ptr == nullptr) {
112 curr_ptr = new (_params.monotonic_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
113 }
114
115 _open.emplace(n.index, curr_ptr);
116 }
117 }
118 }
119 }
120 else { // SOLUTION REACHED...
121
122 _open = open_set_t{};
123 _closed = closed_set_t{};
124
125 if constexpr (params_t::reverse_equivalence::value) {
126 return solver_t::solver_utils::rbacktrack(curr, _params.size);
127 }
128 else {
129 return solver_t::solver_utils::rbacktrack_noreverse(curr, _params.size);
130 }
131 }
132 }
133
134 return std::vector<coord_t>{};
135 }
136
137 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
138
139 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
140
141 existence_set closed(_params.monotonic_pmr);
142 closed.reserve(capacity);
143
144 queue<node> open(_params.heterogeneous_pmr);
145
146 return solve_internal(open, closed, capacity, _params);
147 }
148 };
