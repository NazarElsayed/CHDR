69 static constexpr size_t s_stack_block_size { StackSize };
70 static constexpr size_t s_default_heap_block_size { 4096U };
71 static constexpr size_t s_max_heap_block_size { MaxHeapBlockSize };

73 // Fixed stack memory block:
74 alignas(max_align_t) uint8_t m_stack_block[s_stack_block_size]; //NOLINT(*-avoid-c-arrays)
75
76 size_t m_stack_write; // Current write position for the stack block.
77 size_t m_block_write; // Current write position in the active block.
78 size_t m_active_block_index; // Index of the current active block.
79 size_t m_initial_block_width; // Width of the first allocated block.
80 size_t m_block_width; // Size of the current block.

82 std::vector<block> m_blocks;

84 HOT uint8_t* expand(size_t _bytes, size_t _alignment) {
85
86 uint8_t* result { nullptr };
87
88 try {
89
90 if (m_active_block_index + 1U < m_blocks.size()) { // Reuse an existing block:
91
92 auto& recycled_block = m_blocks[++m_active_block_index];
93
94 m_block_width = recycled_block.size;
95 result = recycled_block.data;
96 }
97 else { // Allocate a new, larger block:
98
99 m_block_width = utils::max(
100 m_initial_block_width,
101 utils::max(utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size), _bytes)
102 );
103
104 result = static_cast<uint8_t*>(operator new(m_block_width, static_cast<std::align_val_t>(_alignment)));
105
106 m_blocks.emplace_back(
107 m_block_width,
108 _alignment,
109 result
110 );
111
112 m_active_block_index = m_blocks.size() - 1U;
113 }
114
115 m_block_write = 0U; // Reset write head to beginning.
116 }
117 catch (...) {
118
119 /* Catch any errors that occur during allocation. */
120
121 // ReSharper disable once CppDFAConstantConditions
122 if (result != nullptr) {
123 // ReSharper disable once CppDFAUnreachableCode
124 operator delete(result, static_cast<std::align_val_t>(_alignment));
125 result = nullptr;
126
127 m_blocks.pop_back();
128 }
129 }
130
131 return result;
132 }
133
146 void cleanup() noexcept {
147 for (const auto& item : m_blocks) {
148 operator delete(item.data, static_cast<std::align_val_t>(item.alignment));
149 }
150 }

170 [[nodiscard]] virtual HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
171 assert(_bytes > 0U && "Allocation size must be greater than zero.");
172
173 uint8_t* aligned_ptr { nullptr };
174
175 // Attempt to allocate from the stack block:
176 if (m_stack_write < StackSize) {
177
178 if (const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
179 aligned_bytes < MaxStackAllocationSize &&
180 m_stack_write + aligned_bytes <= s_stack_block_size
181 ) {
182 aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
183 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
184 }
185 }
186
187 if (aligned_ptr == nullptr) {
188
189 // If the stack block is exhausted, fall back to dynamic blocks:
190 if (!m_blocks.empty()) {
191
192 auto* raw_ptr = static_cast<void*>(static_cast<uint8_t*>(m_blocks[m_active_block_index].data) + m_block_write);
193 size_t space = m_blocks[m_active_block_index].size - m_block_write;
194
195 aligned_ptr = static_cast<uint8_t*>(std::align(_alignment, _bytes, raw_ptr, space));
196 if (aligned_ptr != nullptr) {
197 m_block_write = static_cast<size_t>(static_cast<uint8_t*>(aligned_ptr) - static_cast<uint8_t*>(m_blocks[m_active_block_index].data)) + _bytes;
198 }
199 }
200 else {
201 aligned_ptr = nullptr;
202 }
203
204 // Expand if no valid candidate for allocation was found.
205 if (aligned_ptr == nullptr) {
206 aligned_ptr = expand(_bytes, _alignment);
207 }
208
209 // Update write position.
210 m_block_write += static_cast<size_t>(aligned_ptr + _bytes - (m_blocks[m_active_block_index].data + m_block_write));
211 }
212
213 // ReSharper disable once CppDFAConstantConditions
214 if (aligned_ptr == nullptr) {
215 // ReSharper disable once CppDFAUnreachableCode
216 throw std::bad_alloc();
217 }
218
219 return aligned_ptr;
220 }
