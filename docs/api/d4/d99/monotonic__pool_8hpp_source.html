48 struct block final {
49
50 size_t size;
51 size_t alignment;

54 [[nodiscard]] HOT constexpr block(size_t _size, size_t _alignment, uint8_t* _data) noexcept :
55 size (_size ),

57 data (_data ) {}

64 static constexpr size_t s_stack_block_size { StackSize };
65 static constexpr size_t s_default_heap_block_size { 4096U };
66 static constexpr size_t s_max_heap_block_size { MaxHeapBlockSize };
67
68 alignas(std::max_align_t) uint8_t m_stack_block[s_stack_block_size]; // Fixed stack memory block.
69
70 size_t m_stack_write; // Current write position for the stack block.
71 size_t m_block_write; // Current write position in the active block.
72 size_t m_active_block_index; // Index of the current active block.
73 size_t m_initial_block_width; // Width of the first allocated block.
74 size_t m_block_width; // Size of the current block.
75
76 std::vector<block> m_blocks;
77
78 HOT uint8_t* expand(size_t _bytes, size_t _alignment) {
79
80 uint8_t* result = nullptr;
81
82 try {
83
84 if (m_active_block_index + 1U < m_blocks.size()) { // Reuse an existing block:
85
86 auto& recycled_block = m_blocks[++m_active_block_index];
87
88 m_block_width = recycled_block.size;
89 result = recycled_block.data;
90 }
91 else { // Allocate a new, larger block:
92
93 m_block_width = utils::max(
94 m_initial_block_width,
95 utils::max(utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size), _bytes)
96 );
97
98 result = static_cast<uint8_t*>(operator new(m_block_width, static_cast<std::align_val_t>(_alignment)));
99
100 m_blocks.emplace_back(
101 m_block_width,
102 _alignment,
103 result
104 );
105
106 m_active_block_index = m_blocks.size() - 1U;
107 }
108
109 m_block_write = 0U; // Reset write head to beginning.
110 }
111 catch (...) {
112
113 /* Catch any errors that occur during allocation. */
114
115 // ReSharper disable once CppDFAConstantConditions
116 if (result != nullptr) {
117 // ReSharper disable once CppDFAUnreachableCode
118 ::operator delete(result, static_cast<std::align_val_t>(_alignment));
119 result = nullptr;
120
121 m_blocks.pop_back();
122 }
123 }
124
125 return result;
126 }
127
140 void cleanup() noexcept {
141 for (const auto& item : m_blocks) {
142 ::operator delete(item.data, static_cast<std::align_val_t>(item.alignment));
143 }
144 }

164 [[nodiscard]] HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
165 assert(_bytes > 0U && "Allocation size must be greater than zero.");
166
167 // Attempt to allocate from the stack block:
168 if (m_stack_write < StackSize) {
169
170 if (const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
171 aligned_bytes < MaxStackAllocationSize &&
172 m_stack_write + aligned_bytes <= s_stack_block_size
173 ) {
174
175 auto* aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
176 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
177
178 return aligned_ptr;
179 }
180 }
181
182 uint8_t* aligned_ptr;
183
184 // If stack block is exhausted, fall back to dynamic blocks:
185 if (!m_blocks.empty()) {
186
187 aligned_ptr = reinterpret_cast<uint8_t*>(
188 (reinterpret_cast<uintptr_t>(m_blocks[m_active_block_index].data + m_block_write) + _alignment - 1U) & ~(_alignment - 1U)
189 );
190
191 // Invalidate if the current block cannot fit the allocation.
192 if (aligned_ptr + _bytes > m_blocks[m_active_block_index].data + m_blocks[m_active_block_index].size) {
193 aligned_ptr = nullptr;
194 }
195 }
196 else {
197 aligned_ptr = nullptr;
198 }
199
200 // Expand if no valid candidate for allocation was found.
201 if (aligned_ptr == nullptr) {
202 aligned_ptr = expand(_bytes, _alignment);
203 }
204
205 // Update write position.
206 m_block_write += static_cast<size_t>(aligned_ptr + _bytes - (m_blocks[m_active_block_index].data + m_block_write));
207
208 // ReSharper disable once CppDFAConstantConditions
209 if (aligned_ptr == nullptr) {
210 // ReSharper disable once CppDFAUnreachableCode
211 throw std::bad_alloc();
212 }
213
214 return aligned_ptr;
215 }
