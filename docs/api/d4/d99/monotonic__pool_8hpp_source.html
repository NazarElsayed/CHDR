100 HOT uint8_t* expand(size_t _bytes, size_t _alignment) {
101
102 assert(_bytes > 0U && "Allocation size must be greater than zero.");
103 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
104
105 uint8_t* result { nullptr };
106
107 try {
108
109 if (m_active_block_index + 1U < m_blocks.size()) { // Reuse an existing block:
110
111 auto& recycled_block = m_blocks[++m_active_block_index];
112
113 m_block_width = recycled_block.size;
114 result = recycled_block.data;
115 }
116 else { // Allocate a new, larger block:
117
118 m_block_width = utils::max(
119 m_initial_block_width,
120 utils::max(utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size), _bytes)
121 );
122
123 result = static_cast<uint8_t*>(operator new(m_block_width, static_cast<std::align_val_t>(_alignment)));
124
125 m_blocks.emplace_back(
126 m_block_width,
127 _alignment,
128 result
129 );
130
131 m_active_block_index = m_blocks.size() - 1U;
132 }
133
134 m_block_write = 0U; // Reset write head to beginning.
135 }
136 catch (...) {
137
138 /* Catch any errors that occur during allocation. */
139
140 // ReSharper disable once CppDFAConstantConditions
141 if (result != nullptr) {
142 // ReSharper disable once CppDFAUnreachableCode
143 operator delete(result, static_cast<std::align_val_t>(_alignment));
144 result = nullptr;
145
146 m_blocks.pop_back();
147 }
148 }
149
150 return result;
151 }
152
165 void cleanup() noexcept {
166 for (const auto& item : m_blocks) {
167 operator delete(item.data, static_cast<std::align_val_t>(item.alignment));
168 }
169 }

197 [[nodiscard]] virtual HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
198
199 assert(_bytes > 0U && "Allocation size must be greater than zero.");
200 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
201
202 uint8_t* aligned_ptr { nullptr };
203
204 // Attempt to allocate from the stack block:
205 if (m_stack_write < StackSize) {
206
207 if (const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
208 aligned_bytes < MaxStackAllocationSize &&
209 m_stack_write + aligned_bytes <= s_stack_block_size
210 ) {
211 aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
212 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
213 }
214 }
215
216 if (aligned_ptr == nullptr) {
217
218 // If the stack block is exhausted, fall back to dynamic blocks:
219 if (LIKELY(!m_blocks.empty() && m_blocks[m_active_block_index].size >= m_block_width)) {
220
221 auto* raw_ptr = static_cast<void*>(static_cast<uint8_t*>(m_blocks[m_active_block_index].data) + m_block_write);
222 size_t space = m_blocks[m_active_block_index].size - m_block_write;
223
224 aligned_ptr = static_cast<uint8_t*>(std::align(_alignment, _bytes, raw_ptr, space));
225 if (aligned_ptr != nullptr) {
226 m_block_write = static_cast<size_t>(static_cast<uint8_t*>(aligned_ptr) - static_cast<uint8_t*>(m_blocks[m_active_block_index].data)) + _bytes;
227 }
228 }
229 else {
230 aligned_ptr = nullptr;
231 }
232
233 // Expand if no valid candidate for allocation was found.
234 if (UNLIKELY(aligned_ptr == nullptr)) {
235 aligned_ptr = expand(_bytes, _alignment);
236 }
237
238 // Update write position.
239 m_block_write += static_cast<size_t>(aligned_ptr + _bytes - (m_blocks[m_active_block_index].data + m_block_write));
240 }
241
242 // ReSharper disable once CppDFAConstantConditions
243 if (UNLIKELY(aligned_ptr == nullptr)) {
244 // ReSharper disable once CppDFAUnreachableCode
245 throw std::bad_alloc();
246 }
247
248 PREFETCH(aligned_ptr, _MM_HINT_T0);
249
250#if CHDR_DIAGNOSTICS == 1
251
252 __diagnostic_data. num_allocated += _bytes;
253 __diagnostic_data.peak_allocated = utils::max(__diagnostic_data.peak_allocated, __diagnostic_data.num_allocated);
254
255#endif //CHDR_DIAGNOSTICS == 1
256
257 return aligned_ptr;
258 }
