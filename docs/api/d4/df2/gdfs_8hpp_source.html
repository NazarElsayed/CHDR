73 template <typename open_set_t, typename closed_set_t>
74 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
75
76 const auto s = utils::to_1d(_params.start, _params.size);
77 const auto e = utils::to_1d(_params.end, _params.size);
78
79 _open.emplace(s);
80 _closed.emplace(s);
81
82 // Main loop:
83 while (LIKELY(!_open.empty())) { // SEARCH FOR SOLUTION...
84
85 auto curr(std::move(_open.front()));
86 _open.pop();
87
88 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
89
90 node* RESTRICT curr_ptr(nullptr);
91
92 for (const auto& n_data: _params.maze.get_neighbours(curr.m_index)) {
93
94 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
95
96 // Check if node is not already visited:
97 if (!_closed.contains(n.index)) {
98 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
99
100 if (curr_ptr == nullptr) {
101 curr_ptr = new (_params.homogeneous_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
102 }
103
104 _open.emplace(n.index, curr_ptr);
105 }
106 }
107 }
108
109 if (curr_ptr == nullptr) {
110 curr.expunge(_params.homogeneous_pmr);
111 }
112 }
113 else { // SOLUTION REACHED ...
114
115 _open = open_set_t{};
116 _closed = closed_set_t{};
117
118 return solver_t::solver_utils::rbacktrack(curr, _params.size);
119 }
120 }
121
122 return std::vector<coord_t>{};
123 }
124
125 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
126
127 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
128
129 existence_set closed(_params.monotonic_pmr);
130 closed.reserve(capacity);
131
132 stack<node> open(_params.heterogeneous_pmr);
133
134 return solve_internal(open, closed, capacity, _params);
135 }
