341 [[maybe_unused]] inline void malloc_consolidate(const size_t _bytes = 4096U) {
342 assert(_bytes > 0U && "Allocation size must be greater than zero.");
343
344 auto* tmp = malloc(_bytes);
345 std::atomic_thread_fence(std::memory_order_seq_cst);
346 free(tmp);
347 }
