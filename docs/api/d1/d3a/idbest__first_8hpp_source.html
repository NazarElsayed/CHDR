131 template <typename open_set_t>
132 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, const params_t& _params) {
133
134 using neighbours_t = decltype(_params.maze.get_neighbours(std::declval<index_t>()));
135
136 const auto s = utils::to_1d(_params.start, _params.size);
137 const auto e = utils::to_1d(_params.end, _params.size);
138
139 auto bound = _params.h(_params.start, _params.end) * _params.weight;
140
141 _open.emplace_back(s, bound);
142
143 stack<state<neighbours_t>> stack{};
144
145 do {
146
147 stack.emplace(_open.back(), _params);
148
149 auto min = std::numeric_limits<scalar_t>::max();
150
151 // Main loop:
152 while (!stack.empty()) {
153
154 auto& _ = stack.top();
155 auto& curr = _.curr;
156
157 if (curr.m_hScore <= bound) {
158
159 if (_.neighbours_idx != _.neighbours.size()) {
160
161 if (const auto& n = solver_t::get_data(_.neighbours[(_.neighbours.size() - 1U) - (_.neighbours_idx++)], _params); n.active) {
162
163 if (std::none_of(_open.begin(), _open.end(), [&n](const auto& _item) ALWAYS_INLINE { return _item.m_index == n.index; })) {
164
165 auto h = _params.h(n.coord, _params.end) * _params.weight;
166
167 _open.emplace_back(n.index, h);
168
169 if (n.index != e) { // SEARCH FOR SOLUTION...
170 stack.emplace(_open.back(), _params);
171 }
172 else { // SOLUTION REACHED...
173 return solver_t::solver_utils::ibacktrack(_open, _params.size);
174 }
175 }
176 }
177 }
178 else {
179 _open.pop_back();
180 stack.pop();
181 }
182 }
183 else {
184 min = utils::min(min, curr.m_hScore);
185 _open.pop_back();
186 stack.pop();
187 }
188 }
189
190 if (_open.size() > 1U) {
191 _open.erase(_open.begin() + 1U, _open.end());
192 }
193 stack.clear();
194
195 bound = min;
196 }
197 while (bound != std::numeric_limits<scalar_t>::max());
198
199 return std::vector<coord_t>{};
200 }

202 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
203
204 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
205
206 std::pmr::vector<node> open(_params.heterogeneous_pmr);
207 try {
208 open.reserve(capacity / 8U);
209 }
210 catch (...) {} // NOLINT(*-empty-catch)
211
212 return solve_internal(open, _params);
213 }
