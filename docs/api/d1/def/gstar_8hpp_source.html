71 template <typename open_set_t, typename closed_set_t>
72 [[maybe_unused, nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
73
74 const auto s = utils::to_1d(_params.start, _params.size);
75 const auto e = utils::to_1d(_params.end, _params.size);
76
77 _open.emplace_nosort(s, static_cast<scalar_t>(0), _params.h(_params.start, _params.end) * _params.weight);
78 _closed.emplace(s);
79
80 // Main loop:
81 while (LIKELY(!_open.empty())) {
82
83 auto curr(std::move(_open.top()));
84 _open.pop();
85
86 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
87
88 node* RESTRICT curr_ptr(nullptr);
89
90 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
91
92 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
93
94 // Check if node is not already visited:
95 if (!_closed.contains(n.index)) {
96 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
97
98 if (curr_ptr == nullptr) {
99 curr_ptr = new (_params.homogeneous_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
100 }
101
102 if constexpr (params_t::lazy_sorting::value) {
103 _open.emplace_nosort(n.index, curr_ptr->m_gScore + n.distance, _params.h(n.coord, _params.end) * _params.weight, curr_ptr);
104 }
105 else {
106 _open.emplace(n.index, curr_ptr->m_gScore + n.distance, _params.h(n.coord, _params.end) * _params.weight, curr_ptr);
107 }
108 }
109 }
110 }
111
112 if (curr_ptr == nullptr) {
113 curr.expunge(_params.homogeneous_pmr);
114 }
115 }
116 else { // SOLUTION REACHED ...
117
118 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
119 _open.wipe();
120 }
121 else {
122 _open = {};
123 }
124 _closed = {};
125
126 return solver_t::solver_utils::rbacktrack(curr, _params.size, curr.m_gScore);
127 }
128 }
129
130 return std::vector<coord_t>{};
131 }
132
133 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
134
135 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
136
137 existence_set closed(_params.monotonic_pmr);
138 closed.reserve(capacity);
139
140 heap<node> open(_params.polytonic_pmr);
141 try {
142 open.reserve(capacity / 8U);
143 }
144 catch (...) {} // NOLINT(*-empty-catch)
145
146 return solve_internal(open, closed, capacity, _params);
147 }
