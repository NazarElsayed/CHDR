22 struct lowest_memory_usage {}; /** @brief Each item is represented by 1 bit in memory. */
23 struct high_performance {}; /** @brief Each item is represented by 8 bits in memory. */

25 template<typename>
26 struct alignment {};

28 template<> struct alignment<high_performance> { using type_t [[maybe_unused]] = char; };

30 /**
31 * @class existence_set
32 * @brief A set allowing for efficient existence checks without needing to store the original data in memory.
33 *
34 * @tparam alignment_type The alignment type used by the set.
35 *
36 * @see lowest_memory_usage
37 * @see low_memory_usage
38 * @see balanced
39 * @see highest_performance
40 */
41 template <typename alignment_type = high_performance>
42 class existence_set {
43
44 static_assert(
45 std::is_same_v<alignment_type, lowest_memory_usage> ||
46 std::is_same_v<alignment_type, high_performance>,
47 "alignment_type must be one of the following: "
48 "lowest_memory_usage, high_performance"
49 );
50
51 private:
52
53 using boolean_t = typename alignment<alignment_type>::type_t;
54
55 std::pmr::vector<boolean_t> c;
56
57 constexpr void enable(const size_t& _hash) {
58
59 if (_hash >= c.size()) {
60 resize(_hash + 1U);
61 }
62
63#if defined(GCC)
64#pragma GCC diagnostic push
65#pragma GCC diagnostic ignored "-Wstringop-overflow"
66#endif
67
68 c[_hash] = static_cast<boolean_t>(true);
69
70#if defined(GCC)
71#pragma GCC diagnostic pop
72#endif
73 }
74
75 constexpr void disable(const size_t& _hash) noexcept {
76
77 if (_hash < c.size()) {
78 c[_hash] = static_cast<boolean_t>(false);
79 }
80 }
81
82 public:
83
84 /**
85 * @brief Initialise set.
86 * @param[in] _resource (optional) Custom memory resource.
87 */
88 [[maybe_unused]] constexpr existence_set(std::pmr::memory_resource* _resource = std::pmr::get_default_resource()) noexcept : c(_resource) {}
89
90 /**
91 * @brief Initialise set.
92 * @param[in] _capacity Initial capacity of the set. Must be larger than 0.
93 * @param[in] _resource (optional) Custom memory resource.
94 */
95 [[maybe_unused]] constexpr explicit existence_set(const size_t& _capacity, std::pmr::memory_resource* _resource = std::pmr::get_default_resource()) : c(_resource) {
96 reserve(_capacity);
97 }
98
99 /**
100 * @brief Initialise set using a collection of items.
101 * @details Please note: Duplicate entries will be merged.
102 *
103 * @param[in] _items Items to construct the set using.
104 * @param[in] _capacity Initial capacity of the set. If a value less than 1 is assigned, it will use the size of the provided collection.
105 */
106 [[maybe_unused]] constexpr existence_set(const std::initializer_list<size_t>& _items, const size_t& _capacity = 0U) {
107
108 size_t autoCapacity = _capacity;
109
110 if (autoCapacity < 1U) {
111 autoCapacity = utils::max<size_t>(_items.size(), 1U);
112 }
113
114 reserve(autoCapacity);
115
116 for (const auto& item : _items) {
117 push(item);
118 }
119 }
120
121 /**
122 * @brief Preallocates memory based on the hash and bucket size.
123 * @param[in] _hash The value for which memory needs to be preallocated.
124 * @param[in] _increment Size of bucket to be considered for memory allocation.
125 * @param[in] _max_increment (optional) Maximum limit for the memory allocation.
126 */
127 constexpr void allocate(const size_t& _hash, const size_t& _increment, const size_t& _max_increment = std::numeric_limits<size_t>::max()) {
128 if (capacity() <= _hash) {
129 reserve(utils::min(capacity() + _increment, _max_increment));
130 }
131 }
132
133 /**
134 * @brief Add a hash to the set.
135 * @param[in] _hash The hash value to be added.
136 */
137 constexpr void push(const size_t& _hash) {
138 enable(_hash);
139 }
140
141 /**
142 * @brief Add a hash to the set.
143 * @param[in] _hash The hash value to be added.
144 */
145 template <typename T>
146 constexpr void emplace(T&& _hash) {
147
148 static_assert(std::is_integral_v<std::decay_t<T>>, "Hash must be an integral type");
149
150 enable(static_cast<size_t>(std::forward<T>(_hash)));
151 }
152
153 /**
154 * @brief remove a hash from the set.
155 * @param[in] _hash The hash value to be removed.
156 * @note Please note that this function does not resize the set.
157 *
158 * @see set::prune()
159 * @see set::clear()
160 */
161 [[maybe_unused]] constexpr void erase(const size_t& _hash) noexcept {
162 disable(_hash);
163 }
164
165 /**
166 * @brief Check if the given hash exists in the set.
167 * @param[in] _hash The hash value to check.
168 * @return True if the hash exists in the set, false otherwise.
169 */
170 [[maybe_unused, nodiscard]] constexpr bool contains(const size_t& _hash) const noexcept {
171 return _hash < size() && static_cast<bool>(c[_hash]);
172 }
173
174 /**
175 * @brief trim the set by removing trailing false values.
176 *
177 * @see set::clear()
178 */
179 [[maybe_unused]] constexpr void trim() {
180
181 auto it = c.rbegin();
182 while (it != c.rend() && !static_cast<boolean_t>(*it)) {
183 ++it;
184 }
185 c.erase(it.base(), c.end());
186 }
187
188 /**
189 * @brief Reserves memory for the existence_set.
190 *
191 * @param _newCapacity The new capacity to reserve.
192 * @note This method does not resize the set, only reserves memory for future elements.
193 *
194 * @see existence_set::capacity()
195 */
196 [[maybe_unused]] constexpr void reserve(const size_t& _newCapacity) { c.reserve(_newCapacity); }
197
198 /**
199 * @brief resize the existence_set.
200 *
201 * This method resizes the existence_set by calling the resize method on the internal vector.
202 *
203 * @param _newSize The new size to resize the set to.
204 * @param _newValue The optional value to fill the new elements with. (default is false).
205 * @note If the new size is smaller than the current size, the elements at the end will be removed.
206 * If the new size is greater than the current size, the new elements will be filled with the specified value.
207 *
208 * @see existence_set::reserve(const size_t&)
209 * @see existence_set::prune()
210 * @see existence_set::clear()
211 */
212 [[maybe_unused]] constexpr void resize(const size_t& _newSize, const boolean_t& _newValue = static_cast<boolean_t>(false)) {
213 c.resize(_newSize, _newValue);
214 }
215
216 /**
217 * @brief clear the content of the set.
218 * @details remove all elements from the set.
219 */
220 [[maybe_unused]] constexpr void clear() noexcept { c.clear(); }
221
222 /**
223 * @brief Trims unused elements from the end of the set.
224 * @details Shrinks the internal container of the set to reduce the structure's overall memory footprint.
225 */
226 [[maybe_unused]] constexpr void shrink_to_fit() { c.shrink_to_fit(); }
227
228 /**
229 * @brief Get the size of the set.
230 * @details Returns the number of elements in the set.
231 *
232 * @return The size of the set.
233 */
234 [[maybe_unused, nodiscard]] constexpr auto size() const noexcept { return c.size(); }
235

237 * @brief Get the capacity of the set.
238 * @details Returns the maximum number of elements that the set can hold.
239 *
240 * @return The capacity of the set.
241 */
242 [[maybe_unused, nodiscard]] constexpr auto capacity() const noexcept { return c.capacity(); }
