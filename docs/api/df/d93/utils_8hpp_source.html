194 template<typename coord_t>
195 [[nodiscard]] static constexpr auto to_nd(const typename std::decay_t<coord_t>::value_type& _index, const coord_t& _sizes) noexcept {
196
197 using T = typename std::decay_t<decltype(_index)>;
198 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
199
200 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
201 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
202
203 coord_t result{};
204
205 if constexpr (Kd == 1U) {
206 result = { _index };
207 }
208 else if constexpr (Kd == 2U) {
209 result = {
210 _index % _sizes[0U],
211 _index / _sizes[0U]
212 };
213 }
214 else if constexpr (Kd == 3U) {
215 const auto w1 = _sizes[0U] * _sizes[1U];
216 const auto w2 = _index % w1;
217
218 result = {
219 w2 % _sizes[1U],
220 w2 / _sizes[1U],
221 _index / w1
222 };
223 }
224 else if constexpr (Kd == 4U) {
225 const auto w1 = _sizes[2U] * _sizes[1U];
226 const auto w2 = w1 * _sizes[0U];
227 const auto w3 = _index % w2;
228
229 result = {
230 (w3 % w1) % _sizes[2U],
231 (w3 % w1) / _sizes[2U],
232 w3 / w1,
233 _index / w2,
234 };
235 }
236 else {
237 std::array<T, Kd> strides{};
238 strides[0U] = 1;
239
240 IVDEP
241 for (size_t i = 1U; i < Kd; ++i) {
242 strides[i] = strides[i - 1U] * _sizes[i - 1U];
243 }
244
245 // Please note this loop uses integer underflow to bypass a quirk of reverse for-loops.
246 auto idx = _index;
247 IVDEP
248 for (size_t i = Kd; i != 0U; --i) {
249 result[i - 1U] = idx / strides[i - 1U];
250 idx %= strides[i - 1U];
251 }
252 }
253
254 return result;
255 }

278 template<typename coord_t>
279 [[nodiscard]] static constexpr auto to_1d(const coord_t& _indices, const coord_t& _sizes) noexcept {
280
281 using T = typename std::decay_t<coord_t>::value_type;
282 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
283
284 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
285 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
286
287 T result{};
288
289 if constexpr (Kd == 1U) {
290 result = _indices[0U];
291 }
292 else if constexpr (Kd == 2U) {
293 result = (_indices[1U] * _sizes[0U]) + _indices[0U];
294 }
295 else if constexpr (Kd == 3U) {
296 result = (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
297 (_indices[1U] * _sizes[0U]) +
298 _indices[0U];
299 }
300 else if constexpr (Kd == 4U) {
301 result = (_indices[3U] * (_sizes[2U] * _sizes[1U] * _sizes[0U])) +
302 (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
303 (_indices[1U] * _sizes[0U]) +
304 _indices[0U];
305 }
306 else {
307 result = 0;
308
309 IVDEP
310 for (size_t i = Kd; i != 0U; --i) {
311 result = (result * _sizes[i - 1UL]) + _indices[i - 1UL];
312 }
313 }
314
315 return result;
316 }

338 template <typename return_t, typename T>
339 [[nodiscard]] static constexpr return_t sign(const T& _val) noexcept {
340
341 constexpr auto zero_v = static_cast<T>(0);
342 constexpr auto one_v = static_cast<T>(1);
343
344 if constexpr (std::is_signed_v<T>) {
345 static_assert(std::is_invocable_r_v<bool, decltype(std::less()), T, T>, "Type T must support the less-than operator.");
346 static_assert(!std::is_same_v<return_t, bool> && "return_t cannot be bool if T is signed.");
347
348 return static_cast<return_t>(zero_v < _val) - (_val < zero_v);
349 }
350 else {
351 static_assert(std::is_invocable_r_v<bool, decltype(std::equal_to()), T, T>, "Type T must support the equality operator.");
352
353 return static_cast<return_t>(_val == zero_v ? zero_v : one_v);
354 }
355 }

378 template <typename T>
379 static constexpr T powui(T _base, T _exp) noexcept {
380
381 static_assert(std::is_unsigned_v<T> && std::is_integral_v<T> && std::is_arithmetic_v<T>,
382 "powui requires an unsigned integral supporting arithmetic operations.");
383
384 T result { 1U };
385
386 while (_exp > 0U) {
387 if (_exp % 2U == 1U) {
388 assert(result <= std::numeric_limits<T>::max() / _base && "Multiplication would cause overflow.");
389 result *= _base;
390 }
391 assert(_base <= std::numeric_limits<T>::max() / _base && "Multiplication would cause overflow.");
392 _base *= _base;
393 _exp /= 2U;
394 }
395
396 return result;
397 }

440 template <typename T>
441 HOT static constexpr T sqrt(const T& _value) noexcept {
442
443 static_assert(std::is_arithmetic_v<T>, "Type T must be arithmetic.");
444
445 T result{};
446
447 if (__builtin_is_constant_evaluated()) {
448
449 if (_value < static_cast<T>(0)) {
450 result = std::numeric_limits<T>::quiet_NaN();
451 }
452 else {
453
454 if (_value == static_cast<T>(0) ||
455 _value == static_cast<T>(1)
456 ) {
457 result = _value; // sqrt(0) = 0, sqrt(1) = 1
458 }
459 else {
460
461 T x { _value };
462 T last { 0 };
463
464 while (x != last) {
465 last = x;
466 x = (x + _value / x) / static_cast<T>(2);
467 }
468
469 result = x;
470 }
471 }
472 }
473 else {
474 result = static_cast<T>(std::sqrt(_value));
475 }
476
477 return result;
478 }

562 [[nodiscard]] static std::string to_string(long double _duration, long double _scale = std::numeric_limits<long double>::epsilon()) {
563
564 constexpr std::array<const char* const, 4U> units { "s", "ms", "s", "ns" };
565
566 size_t i = 0U;
567 while (i < units.size() && abs(_duration) < 1.0L) {
568 _duration *= 1000.0L;
569 ++i;
570 }
571
572 _scale = clamp(_scale, 0.0L, 1.0L);
573
574 return trim_trailing_zeros(
575 std::to_string(
576 std::floor((_duration / _scale) + 0.5L) * _scale
577 )
578 ) + units[i];
579 }
