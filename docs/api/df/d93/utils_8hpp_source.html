152 template<typename coord_t>
153 [[nodiscard]] static constexpr auto to_nd(const typename std::decay_t<coord_t>::value_type& _index, const coord_t& _sizes) noexcept {
154
155 using T = typename std::decay_t<decltype(_index)>;
156 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
157
158 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
159 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
160
161 coord_t result{};
162
163 if constexpr (Kd == 1U) {
164 result = { _index };
165 }
166 else if constexpr (Kd == 2U) {
167 result = {
168 _index % _sizes[0U],
169 _index / _sizes[0U]
170 };
171 }
172 else if constexpr (Kd == 3U) {
173 const auto w1 = _sizes[0U] * _sizes[1U],
174 w2 = _index % w1;
175
176 result = {
177 w2 % _sizes[1U],
178 w2 / _sizes[1U],
179 _index / w1
180 };
181 }
182 else if constexpr (Kd == 4U) {
183 const auto w1 = _sizes[2U] * _sizes[1U],
184 w2 = w1 * _sizes[0U],
185 w3 = _index % w2;
186
187 result = {
188 (w3 % w1) % _sizes[2U],
189 (w3 % w1) / _sizes[2U],
190 w3 / w1,
191 _index / w2,
192 };
193 }
194 else if constexpr (Kd > 4U) {
195 std::array<T, Kd> strides{};
196 strides[0U] = 1;
197
198 IVDEP
199 for (size_t i = 1U; i < Kd; ++i) {
200 strides[i] = strides[i - 1U] * _sizes[i - 1U];
201 }
202
203 // Please note this loop uses integer underflow to bypass a quirk of reverse for-loops.
204 auto idx = _index;
205 IVDEP
206 for (size_t i = Kd; i != 0U; --i) {
207 result[i - 1U] = idx / strides[i - 1U];
208 idx %= strides[i - 1U];
209 }
210 }
211
212 return result;
213 }

236 template<typename coord_t>
237 [[nodiscard]] static constexpr auto to_1d(const coord_t& _indices, const coord_t& _sizes) noexcept {
238
239 using T = typename std::decay_t<coord_t>::value_type;
240 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
241
242 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
243 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
244
245 T result{};
246
247 if constexpr (Kd == 1U) {
248 result = _indices[0U];
249 }
250 else if constexpr (Kd == 2U) {
251 result = (_indices[1U] * _sizes[0U]) + _indices[0U];
252 }
253 else if constexpr (Kd == 3U) {
254 result = (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
255 (_indices[1U] * _sizes[0U]) +
256 _indices[0U];
257 }
258 else if constexpr (Kd == 4U) {
259 result = (_indices[3U] * (_sizes[2U] * _sizes[1U] * _sizes[0U])) +
260 (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
261 (_indices[1U] * _sizes[0U]) +
262 _indices[0U];
263 }
264 else if constexpr (Kd > 4U) {
265 result = 0;
266
267 IVDEP
268 for (size_t i = Kd; i != 0U; --i) {
269 result = (result * _sizes[i - 1UL]) + _indices[i - 1UL];
270 }
271 }
272
273 return result;
274 }

296 template <typename return_t, typename T>
297 [[nodiscard]] static constexpr return_t sign(const T& _val) noexcept {
298
299 constexpr auto zero_v = static_cast<T>(0);
300 constexpr auto one_v = static_cast<T>(1);
301
302 if constexpr (std::is_signed_v<T>) {
303 static_assert(std::is_invocable_r_v<bool, decltype(std::less<>()), T, T>, "Type T must support the less-than operator.");
304 static_assert(!std::is_same_v<return_t, bool> && "return_t cannot be bool if T is signed.");
305
306 return static_cast<return_t>(zero_v < _val) - (_val < zero_v);
307 }
308 else {
309 static_assert(std::is_invocable_r_v<bool, decltype(std::equal_to<>()), T, T>, "Type T must support the equality operator.");
310
311 return static_cast<return_t>(_val == zero_v ? zero_v : one_v);
312 }
313 }

334 template <typename T>
335 static constexpr T powui(T _base, T _exp) noexcept {
336
337 static_assert(std::is_unsigned_v<T> && std::is_integral_v<T> && std::is_arithmetic_v<T>,
338 "powui requires an unsigned integral supporting arithmetic operations.");
339
340 T result { 1U };
341
342 while (_exp > 0U) {
343 if (_exp % 2U == 1U) {
344 result *= _base;
345 }
346 _base *= _base;
347 _exp /= 2U;
348 }
349
350 return result;
351 }

393 template <typename T>
394 HOT static constexpr T sqrt(const T& _value) noexcept {
395
396 static_assert(std::is_arithmetic_v<T>, "Type T must be arithmetic.");
397
398 if (__builtin_is_constant_evaluated()) {
399
400 if (_value >= static_cast<T>(0)) {
401
402 if (_value != static_cast<T>(0) && _value != static_cast<T>(1)) {
403
404 T x { _value };
405 T last { 0 };
406
407 while (x != last) {
408 last = x;
409 x = (x + _value / x) / static_cast<T>(2);
410 }
411
412 return x;
413 }
414
415 return _value; // sqrt(0) = 0, sqrt(1) = 1
416 }
417
418 return std::numeric_limits<T>::quiet_NaN();
419 }
420 else {
421 return static_cast<T>(std::sqrt(_value));
422 }
423 }

503 [[nodiscard]] static std::string to_string(long double _duration, long double _scale = std::numeric_limits<long double>::epsilon()) {
504
505 static std::array<std::string, 4U> units = { "s", "ms", "µs", "ns" };
506
507 auto result = _duration;
508
509 size_t i = 0U;
510 while (i < units.size() && abs(result) < 1.0l) {
511 result *= 1000.0l;
512 ++i;
513 }
514
515 _scale = clamp(_scale, 0.0l, 1.0l);
516
517 return trim_trailing_zeros(
518 std::to_string(
519 std::floor(result / _scale + 0.5l) * _scale
520 )
521 ) + units[i];
522 }
