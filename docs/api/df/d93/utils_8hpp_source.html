159 template<typename coord_t>
160 [[nodiscard]] static constexpr auto to_nd(const typename std::decay_t<coord_t>::value_type& _index, const coord_t& _sizes) noexcept {
161
162 using T = typename std::decay_t<decltype(_index)>;
163 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
164
165 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
166 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
167
168 coord_t result{};
169
170 if constexpr (Kd == 1U) {
171 result = { _index };
172 }
173 else if constexpr (Kd == 2U) {
174 result = {
175 _index % _sizes[0U],
176 _index / _sizes[0U]
177 };
178 }
179 else if constexpr (Kd == 3U) {
180 const auto w1 = _sizes[0U] * _sizes[1U];
181 const auto w2 = _index % w1;
182
183 result = {
184 w2 % _sizes[1U],
185 w2 / _sizes[1U],
186 _index / w1
187 };
188 }
189 else if constexpr (Kd == 4U) {
190 const auto w1 = _sizes[2U] * _sizes[1U];
191 const auto w2 = w1 * _sizes[0U];
192 const auto w3 = _index % w2;
193
194 result = {
195 (w3 % w1) % _sizes[2U],
196 (w3 % w1) / _sizes[2U],
197 w3 / w1,
198 _index / w2,
199 };
200 }
201 else {
202 std::array<T, Kd> strides{};
203 strides[0U] = 1;
204
205 IVDEP
206 for (size_t i = 1U; i < Kd; ++i) {
207 strides[i] = strides[i - 1U] * _sizes[i - 1U];
208 }
209
210 // Please note this loop uses integer underflow to bypass a quirk of reverse for-loops.
211 auto idx = _index;
212 IVDEP
213 for (size_t i = Kd; i != 0U; --i) {
214 result[i - 1U] = idx / strides[i - 1U];
215 idx %= strides[i - 1U];
216 }
217 }
218
219 return result;
220 }

243 template<typename coord_t>
244 [[nodiscard]] static constexpr auto to_1d(const coord_t& _indices, const coord_t& _sizes) noexcept {
245
246 using T = typename std::decay_t<coord_t>::value_type;
247 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
248
249 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
250 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
251
252 T result{};
253
254 if constexpr (Kd == 1U) {
255 result = _indices[0U];
256 }
257 else if constexpr (Kd == 2U) {
258 result = (_indices[1U] * _sizes[0U]) + _indices[0U];
259 }
260 else if constexpr (Kd == 3U) {
261 result = (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
262 (_indices[1U] * _sizes[0U]) +
263 _indices[0U];
264 }
265 else if constexpr (Kd == 4U) {
266 result = (_indices[3U] * (_sizes[2U] * _sizes[1U] * _sizes[0U])) +
267 (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
268 (_indices[1U] * _sizes[0U]) +
269 _indices[0U];
270 }
271 else {
272 result = 0;
273
274 IVDEP
275 for (size_t i = Kd; i != 0U; --i) {
276 result = (result * _sizes[i - 1UL]) + _indices[i - 1UL];
277 }
278 }
279
280 return result;
281 }

303 template <typename return_t, typename T>
304 [[nodiscard]] static constexpr return_t sign(const T& _val) noexcept {
305
306 constexpr auto zero_v = static_cast<T>(0);
307 constexpr auto one_v = static_cast<T>(1);
308
309 if constexpr (std::is_signed_v<T>) {
310 static_assert(std::is_invocable_r_v<bool, decltype(std::less()), T, T>, "Type T must support the less-than operator.");
311 static_assert(!std::is_same_v<return_t, bool> && "return_t cannot be bool if T is signed.");
312
313 return static_cast<return_t>(zero_v < _val) - (_val < zero_v);
314 }
315 else {
316 static_assert(std::is_invocable_r_v<bool, decltype(std::equal_to()), T, T>, "Type T must support the equality operator.");
317
318 return static_cast<return_t>(_val == zero_v ? zero_v : one_v);
319 }
320 }

343 template <typename T>
344 static constexpr T powui(T _base, T _exp) noexcept {
345
346 static_assert(std::is_unsigned_v<T> && std::is_integral_v<T> && std::is_arithmetic_v<T>,
347 "powui requires an unsigned integral supporting arithmetic operations.");
348
349 T result { 1U };
350
351 while (_exp > 0U) {
352 if (_exp % 2U == 1U) {
353 assert(result <= std::numeric_limits<T>::max() / _base && "Multiplication would cause overflow.");
354 result *= _base;
355 }
356 assert(_base <= std::numeric_limits<T>::max() / _base && "Multiplication would cause overflow.");
357 _base *= _base;
358 _exp /= 2U;
359 }
360
361 return result;
362 }

405 template <typename T>
406 HOT static constexpr T sqrt(const T& _value) noexcept {
407
408 static_assert(std::is_arithmetic_v<T>, "Type T must be arithmetic.");
409
410 T result{};
411
412 if (__builtin_is_constant_evaluated()) {
413
414 if (_value < static_cast<T>(0)) {
415 result = std::numeric_limits<T>::quiet_NaN();
416 }
417 else {
418
419 if (_value == static_cast<T>(0) ||
420 _value == static_cast<T>(1)
421 ) {
422 result = _value; // sqrt(0) = 0, sqrt(1) = 1
423 }
424 else {
425
426 T x { _value };
427 T last { 0 };
428
429 while (x != last) {
430 last = x;
431 x = (x + _value / x) / static_cast<T>(2);
432 }
433
434 result = x;
435 }
436 }
437 }
438 else {
439 result = static_cast<T>(std::sqrt(_value));
440 }
441
442 return result;
443 }

527 [[nodiscard]] static std::string to_string(long double _duration, long double _scale = std::numeric_limits<long double>::epsilon()) {
528
529 constexpr std::array<const char* const, 4U> units { "s", "ms", "µs", "ns" };
530
531 size_t i = 0U;
532 while (i < units.size() && abs(_duration) < 1.0L) {
533 _duration *= 1000.0L;
534 ++i;
535 }
536
537 _scale = clamp(_scale, 0.0L, 1.0L);
538
539 return trim_trailing_zeros(
540 std::to_string(
541 std::floor((_duration / _scale) + 0.5L) * _scale
542 )
543 ) + units[i];
544 }
