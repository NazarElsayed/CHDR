144 template<typename coord_t>
145 [[nodiscard]] static constexpr auto to_nd(const typename std::decay_t<coord_t>::value_type& _index, const coord_t& _sizes) noexcept {
146
147 using T = typename std::decay_t<decltype(_index)>;
148 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
149
150 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
151 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
152
153 coord_t result{};
154
155 if constexpr (Kd == 1U) {
156 result = { _index };
157 }
158 else if constexpr (Kd == 2U) {
159 result = {
160 _index % _sizes[0U],
161 _index / _sizes[0U]
162 };
163 }
164 else if constexpr (Kd == 3U) {
165 const auto w1 = _sizes[0U] * _sizes[1U],
166 w2 = _index % w1;
167
168 result = {
169 w2 % _sizes[1U],
170 w2 / _sizes[1U],
171 _index / w1
172 };
173 }
174 else if constexpr (Kd == 4U) {
175 const auto w1 = _sizes[2U] * _sizes[1U],
176 w2 = w1 * _sizes[0U],
177 w3 = _index % w2;
178
179 result = {
180 (w3 % w1) % _sizes[2U],
181 (w3 % w1) / _sizes[2U],
182 w3 / w1,
183 _index / w2,
184 };
185 }
186 else if constexpr (Kd > 4U) {
187 std::array<T, Kd> strides{};
188 strides[0U] = 1;
189
190 IVDEP
191 for (size_t i = 1U; i < Kd; ++i) {
192 strides[i] = strides[i - 1U] * _sizes[i - 1U];
193 }
194
195 // Please note this loop uses integer underflow to bypass a quirk of reverse for-loops.
196 auto idx = _index;
197 IVDEP
198 for (size_t i = Kd; i != 0U; --i) {
199 result[i - 1U] = idx / strides[i - 1U];
200 idx %= strides[i - 1U];
201 }
202 }
203
204 return result;
205 }

227 template<typename coord_t>
228 [[nodiscard]] static constexpr auto to_1d(const coord_t& _indices, const coord_t& _sizes) noexcept {
229
230 using T = typename std::decay_t<coord_t>::value_type;
231 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
232
233 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
234 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
235
236 T result{};
237
238 if constexpr (Kd == 1U) {
239 result = _indices[0U];
240 }
241 else if constexpr (Kd == 2U) {
242 result = (_indices[1U] * _sizes[0U]) + _indices[0U];
243 }
244 else if constexpr (Kd == 3U) {
245 result = (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
246 (_indices[1U] * _sizes[0U]) +
247 _indices[0U];
248 }
249 else if constexpr (Kd == 4U) {
250 result = (_indices[3U] * (_sizes[2U] * _sizes[1U] * _sizes[0U])) +
251 (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
252 (_indices[1U] * _sizes[0U]) +
253 _indices[0U];
254 }
255 else if constexpr (Kd > 4U) {
256 result = 0;
257
258 IVDEP
259 for (size_t i = Kd; i != 0U; --i) {
260 result = (result * _sizes[i - 1UL]) + _indices[i - 1UL];
261 }
262 }
263
264 return result;
265 }

286 template <typename return_t, typename T>
287 [[nodiscard]] static constexpr return_t sign(const T& _val) noexcept {
288
289 constexpr auto zero_v = static_cast<T>(0);
290 constexpr auto one_v = static_cast<T>(1);
291
292 if constexpr (std::is_signed_v<T>) {
293 static_assert(std::is_invocable_r_v<bool, decltype(std::less<>()), T, T>, "Type T must support the less-than operator.");
294 static_assert(!std::is_same_v<return_t, bool> && "return_t cannot be bool if T is signed.");
295
296 return static_cast<return_t>(zero_v < _val) - (_val < zero_v);
297 }
298 else {
299 static_assert(std::is_invocable_r_v<bool, decltype(std::equal_to<>()), T, T>, "Type T must support the equality operator.");
300
301 return static_cast<return_t>(_val == zero_v ? zero_v : one_v);
302 }
303 }

323 template <typename T>
324 static constexpr T powui(T _base, T _exp) noexcept {
325
326 static_assert(std::is_unsigned_v<T> && std::is_integral_v<T> && std::is_arithmetic_v<T>,
327 "powui requires an unsigned integral supporting arithmetic operations.");
328
329 T result { 1U };
330
331 while (_exp > 0U) {
332 if (_exp % 2U == 1U) {
333 result *= _base;
334 }
335 _base *= _base;
336 _exp /= 2U;
337 }
338
339 return result;
340 }

380 template <typename T>
381 HOT static constexpr T sqrt(const T& _value) noexcept {
382
383 static_assert(std::is_arithmetic_v<T>, "Type T must be arithmetic.");
384
385 if (__builtin_is_constant_evaluated()) {
386
387 if (_value >= static_cast<T>(0)) {
388
389 if (_value != static_cast<T>(0) && _value != static_cast<T>(1)) {
390
391 T x { _value };
392 T last { 0 };
393
394 while (x != last) {
395 last = x;
396 x = (x + _value / x) / static_cast<T>(2);
397 }
398
399 return x;
400 }
401
402 return _value; // sqrt(0) = 0, sqrt(1) = 1
403 }
404
405 return std::numeric_limits<T>::quiet_NaN();
406 }
407 else {
408 return static_cast<T>(std::sqrt(_value));
409 }
410 }

485 [[nodiscard]] static std::string to_string(long double _duration, long double _scale = std::numeric_limits<long double>::epsilon()) {
486
487 static std::array<std::string, 4U> units = { "s", "ms", "µs", "ns" };
488
489 auto result = _duration;
490
491 size_t i = 0U;
492 while (i < units.size() && abs(result) < 1.0l) {
493 result *= 1000.0l;
494 ++i;
495 }
496
497 _scale = clamp(_scale, 0.0l, 1.0l);
498
499 return trim_trailing_zeros(
500 std::to_string(
501 std::floor(result / _scale + 0.5l) * _scale
502 )
503 ) + units[i];
504 }

519 [[nodiscard]] static std::string trim_trailing_zeros(std::string _str) {
520
521 _str.erase(_str.find_last_not_of('0') + 1U, std::string::npos);
522
523 if (_str.back() == '.') {
