156 template<typename coord_t>
157 [[nodiscard]] static constexpr auto to_nd(const typename std::decay_t<coord_t>::value_type& _index, const coord_t& _sizes) noexcept {
158
159 using T = typename std::decay_t<decltype(_index)>;
160 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
161
162 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
163 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
164
165 coord_t result{};
166
167 if constexpr (Kd == 1U) {
168 result = { _index };
169 }
170 else if constexpr (Kd == 2U) {
171 result = {
172 _index % _sizes[0U],
173 _index / _sizes[0U]
174 };
175 }
176 else if constexpr (Kd == 3U) {
177 const auto w1 = _sizes[0U] * _sizes[1U];
178 const auto w2 = _index % w1;
179
180 result = {
181 w2 % _sizes[1U],
182 w2 / _sizes[1U],
183 _index / w1
184 };
185 }
186 else if constexpr (Kd == 4U) {
187 const auto w1 = _sizes[2U] * _sizes[1U];
188 const auto w2 = w1 * _sizes[0U];
189 const auto w3 = _index % w2;
190
191 result = {
192 (w3 % w1) % _sizes[2U],
193 (w3 % w1) / _sizes[2U],
194 w3 / w1,
195 _index / w2,
196 };
197 }
198 else {
199 std::array<T, Kd> strides{};
200 strides[0U] = 1;
201
202 IVDEP
203 for (size_t i = 1U; i < Kd; ++i) {
204 strides[i] = strides[i - 1U] * _sizes[i - 1U];
205 }
206
207 // Please note this loop uses integer underflow to bypass a quirk of reverse for-loops.
208 auto idx = _index;
209 IVDEP
210 for (size_t i = Kd; i != 0U; --i) {
211 result[i - 1U] = idx / strides[i - 1U];
212 idx %= strides[i - 1U];
213 }
214 }
215
216 return result;
217 }

240 template<typename coord_t>
241 [[nodiscard]] static constexpr auto to_1d(const coord_t& _indices, const coord_t& _sizes) noexcept {
242
243 using T = typename std::decay_t<coord_t>::value_type;
244 static_assert(std::is_integral_v<T>, "Only integer types are allowed.");
245
246 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
247 static_assert(Kd > 0U, "The number of dimensions must be greater than 0.");
248
249 T result{};
250
251 if constexpr (Kd == 1U) {
252 result = _indices[0U];
253 }
254 else if constexpr (Kd == 2U) {
255 result = (_indices[1U] * _sizes[0U]) + _indices[0U];
256 }
257 else if constexpr (Kd == 3U) {
258 result = (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
259 (_indices[1U] * _sizes[0U]) +
260 _indices[0U];
261 }
262 else if constexpr (Kd == 4U) {
263 result = (_indices[3U] * (_sizes[2U] * _sizes[1U] * _sizes[0U])) +
264 (_indices[2U] * (_sizes[1U] * _sizes[0U])) +
265 (_indices[1U] * _sizes[0U]) +
266 _indices[0U];
267 }
268 else {
269 result = 0;
270
271 IVDEP
272 for (size_t i = Kd; i != 0U; --i) {
273 result = (result * _sizes[i - 1UL]) + _indices[i - 1UL];
274 }
275 }
276
277 return result;
278 }

300 template <typename return_t, typename T>
301 [[nodiscard]] static constexpr return_t sign(const T& _val) noexcept {
302
303 constexpr auto zero_v = static_cast<T>(0);
304 constexpr auto one_v = static_cast<T>(1);
305
306 if constexpr (std::is_signed_v<T>) {
307 static_assert(std::is_invocable_r_v<bool, decltype(std::less()), T, T>, "Type T must support the less-than operator.");
308 static_assert(!std::is_same_v<return_t, bool> && "return_t cannot be bool if T is signed.");
309
310 return static_cast<return_t>(zero_v < _val) - (_val < zero_v);
311 }
312 else {
313 static_assert(std::is_invocable_r_v<bool, decltype(std::equal_to()), T, T>, "Type T must support the equality operator.");
314
315 return static_cast<return_t>(_val == zero_v ? zero_v : one_v);
316 }
317 }

338 template <typename T>
339 static constexpr T powui(T _base, T _exp) noexcept {
340
341 static_assert(std::is_unsigned_v<T> && std::is_integral_v<T> && std::is_arithmetic_v<T>,
342 "powui requires an unsigned integral supporting arithmetic operations.");
343
344 T result { 1U };
345
346 while (_exp > 0U) {
347 if (_exp % 2U == 1U) {
348 result *= _base;
349 }
350 _base *= _base;
351 _exp /= 2U;
352 }
353
354 return result;
355 }

398 template <typename T>
399 HOT static constexpr T sqrt(const T& _value) noexcept {
400
401 static_assert(std::is_arithmetic_v<T>, "Type T must be arithmetic.");
402
403 if (__builtin_is_constant_evaluated()) {
404
405 if (_value >= static_cast<T>(0)) {
406
407 if (_value != static_cast<T>(0) && _value != static_cast<T>(1)) {
408
409 T x { _value };
410 T last { 0 };
411
412 while (x != last) {
413 last = x;
414 x = (x + _value / x) / static_cast<T>(2);
415 }
416
417 return x;
418 }
419
420 return _value; // sqrt(0) = 0, sqrt(1) = 1
421 }
422
423 return std::numeric_limits<T>::quiet_NaN();
424 }
425
426 return static_cast<T>(std::sqrt(_value));
427 }

506 [[nodiscard]] static std::string to_string(const long double _duration, long double _scale = std::numeric_limits<long double>::epsilon()) {
507
508 constexpr std::array<const char* const, 4U> units { "s", "ms", "s", "ns" };
509
510 auto result = _duration;
511
512 size_t i = 0U;
513 while (i < units.size() && abs(result) < 1.0L) {
514 result *= 1000.0L;
515 ++i;
516 }
517
518 _scale = clamp(_scale, 0.0L, 1.0L);
519
520 return trim_trailing_zeros(
521 std::to_string(
522 std::floor((result / _scale) + 0.5L) * _scale
523 )
524 ) + units[i];
525 }
