95 [[maybe_unused, nodiscard]] HOT static constexpr auto sqr_euclidean_distance(const coord_t& _a, const coord_t& _b) noexcept {
96
97 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
98
99 scalar_t result{0};
100
101 IVDEP
102 for (size_t i = 0U; i < Kd; ++i) {
103 const auto val = static_cast<scalar_t>(utils::abs(static_cast<signed>(_b[i]) - static_cast<signed>(_a[i])));
104 result += val * val;
105 }
106
107 return result;

165 template <typename scalar_t, typename coord_t>
166 [[maybe_unused, nodiscard]] HOT static constexpr auto canberra_distance(const coord_t& _a, const coord_t& _b) noexcept {
167
168 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
169
170 scalar_t result{0};
171
172 IVDEP
173 for (size_t i = 0U; i < Kd; ++i) {
174
175 if (const auto denominator = static_cast<scalar_t>(utils::abs(_a[i]) + utils::abs(_b[i])); denominator != 0) {
176 const auto numerator = static_cast<scalar_t>(utils::abs(static_cast<signed>(_b[i]) - static_cast<signed>(_a[i])));
177
178 result += numerator / denominator;
179 }
180 }
181
182 return result;
183 }

192 template <typename scalar_t, typename coord_t>
193 [[maybe_unused, nodiscard]] HOT static constexpr auto cosine_distance(const coord_t& _a, const coord_t& _b) noexcept {
194
195 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
196
197 scalar_t dot_product {0};
198 scalar_t norm_a {0};
199 scalar_t norm_b {0};
200
201 IVDEP
202 for (size_t i = 0; i < Kd; ++i) {
203
204 const scalar_t val_a = static_cast<scalar_t>(_a[i]);
205 const scalar_t val_b = static_cast<scalar_t>(_b[i]);
206
207 dot_product += val_a * val_b;
208 norm_a += val_a * val_a;
209 norm_b += val_b * val_b;
210 }
211
212 const auto norm_product = utils::sqrt(norm_a) * utils::sqrt(norm_b);
213
214 return (norm_product > 0) ? (1 - (dot_product / norm_product)) : static_cast<scalar_t>(1);
215 }
