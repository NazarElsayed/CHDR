145 template <typename scalar_t, typename coord_t>
146 [[maybe_unused, nodiscard]] HOT static constexpr auto canberra_distance(const coord_t& _a, const coord_t& _b) noexcept {
147
148 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
149
150 scalar_t result{0};
151
152 IVDEP
153 for (size_t i = 0U; i < Kd; ++i) {
154
155 if (const auto denominator = static_cast<scalar_t>(utils::abs(_a[i]) + utils::abs(_b[i])); denominator != 0) {
156 const auto numerator = static_cast<scalar_t>(utils::abs(static_cast<signed>(_b[i]) - static_cast<signed>(_a[i])));
157
158 result += numerator / denominator;
159 }
160 }
161
162 return result;

173 template <typename scalar_t, typename coord_t>
174 [[maybe_unused, nodiscard]] HOT static constexpr auto cosine_distance(const coord_t& _a, const coord_t& _b) noexcept {
175
176 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
177
178 scalar_t dot_product {0};
179 scalar_t norm_a {0};
180 scalar_t norm_b {0};
181
182 IVDEP
183 for (size_t i = 0U; i < Kd; ++i) {
184
185 const scalar_t val_a = static_cast<scalar_t>(_a[i]);
186 const scalar_t val_b = static_cast<scalar_t>(_b[i]);
187
188 dot_product += val_a * val_b;
189 norm_a += val_a * val_a;
190 norm_b += val_b * val_b;
191 }
192
193 const auto norm_product = utils::sqrt(norm_a) * utils::sqrt(norm_b);
194
195 return (norm_product > 0) ? (1 - (dot_product / norm_product)) : static_cast<scalar_t>(1);
196 }
