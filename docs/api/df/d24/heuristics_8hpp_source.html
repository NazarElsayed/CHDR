169 template <typename scalar_t, typename coord_t>
170 [[maybe_unused, nodiscard]] HOT static constexpr auto canberra_distance(const coord_t& _a, const coord_t& _b) noexcept {
171
172 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
173
174 scalar_t result{0};
175
176 IVDEP
177 for (size_t i = 0U; i < Kd; ++i) {
178
179 if (const auto denominator = static_cast<scalar_t>(utils::abs(_a[i]) + utils::abs(_b[i])); denominator != 0) {
180 const auto numerator = static_cast<scalar_t>(utils::abs(static_cast<signed>(_b[i]) - static_cast<signed>(_a[i])));
181
182 result += numerator / denominator;
183 }
184 }
185
186 return result;

196 template <typename scalar_t, typename coord_t>
197 [[maybe_unused, nodiscard]] HOT static constexpr auto cosine_distance(const coord_t& _a, const coord_t& _b) noexcept {
198
199 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
200
201 scalar_t dot_product {0};
202 scalar_t norm_a {0};
203 scalar_t norm_b {0};
204
205 IVDEP
206 for (size_t i = 0; i < Kd; ++i) {
207
208 const scalar_t val_a = static_cast<scalar_t>(_a[i]);
209 const scalar_t val_b = static_cast<scalar_t>(_b[i]);
210
211 dot_product += val_a * val_b;
212 norm_a += val_a * val_a;
213 norm_b += val_b * val_b;
214 }
215
216 const auto norm_product = utils::sqrt(norm_a) * utils::sqrt(norm_b);
217
218 return (norm_product > 0) ? (1 - (dot_product / norm_product)) : static_cast<scalar_t>(1);
219 }
