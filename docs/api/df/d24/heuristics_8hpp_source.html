96 [[maybe_unused, nodiscard]] HOT static constexpr auto octile_distance(const coord_t& _a, const coord_t& _b) noexcept {
97
98 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
99
100 scalar_t result = chebyshev_distance<scalar_t, coord_t>(_a, _b);
101
102 bool straight = false;
103 for (size_t i = 0U; i < Kd; ++i) {
104
105 if (_a[0] == _b[0]) {
106 straight = true;
107
108 if constexpr (Kd > 4U) { break; }
109 }
110 }

113 constexpr auto sqrt2 = utils::sqrt<scalar_t>(2.0);
114 result *= sqrt2;
115 }

179 [[maybe_unused, nodiscard]] HOT static constexpr auto canberra_distance(const coord_t& _a, const coord_t& _b) noexcept {
180
181 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
182
183 scalar_t result{0};
184
185 IVDEP
186 for (size_t i = 0U; i < Kd; ++i) {
187
188 if (const auto denominator = static_cast<scalar_t>(utils::abs(_a[i]) + utils::abs(_b[i])); denominator != 0) {
189 const auto numerator = static_cast<scalar_t>(utils::abs(static_cast<signed>(_b[i]) - static_cast<signed>(_a[i])));

191 result += numerator / denominator;

206 template <typename scalar_t, typename coord_t>
207 [[maybe_unused, nodiscard]] HOT static constexpr auto cosine_distance(const coord_t& _a, const coord_t& _b) noexcept {
208
209 constexpr auto Kd = std::tuple_size_v<std::decay_t<coord_t>>;
210
211 scalar_t dot_product {0};
212 scalar_t norm_a {0};
213 scalar_t norm_b {0};
214
215 IVDEP
216 for (size_t i = 0U; i < Kd; ++i) {
217
218 const scalar_t val_a = static_cast<scalar_t>(_a[i]);
219 const scalar_t val_b = static_cast<scalar_t>(_b[i]);
220
221 dot_product += val_a * val_b;
222 norm_a += val_a * val_a;
223 norm_b += val_b * val_b;
224 }
225
226 const auto norm_product = utils::sqrt(norm_a) * utils::sqrt(norm_b);
227
228 return (norm_product > 0) ? (1 - (dot_product / norm_product)) : static_cast<scalar_t>(1);
229 }
