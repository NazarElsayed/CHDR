405 template<size_t Index>
406 constexpr void compute_single_diagonal(const coord_t& _id, neighbour_t& _output) const noexcept {
407
408 constexpr size_t sampleIndex = (Index >= s_neighbour_count / 2U) ? (Index + 1U) : Index;
409 constexpr coord_t direction = utils::to_nd(sampleIndex, coord_t { 3U });
410
411 bool oob = false;
412 coord_t coord = _id;
413
414 for (size_t j = 0U; j < s_rank; ++j) {
415 coord[j] += (direction[j] - 1U);
416
417 if (coord[j] >= m_size[j]) {
418 oob = true;
419 break;
420 }
421 }
422
423 _output = { !oob && operator[](coord).is_active(), coord };
424 }
425
426 template<size_t Index>
427 constexpr void compute_single_axis(const coord_t& _id, neighbour_t& _negative, neighbour_t& _positive) const noexcept {
428
429 coord_t nCoord = _id; // Negative
430 coord_t pCoord = _id; // Positive
431
432 --nCoord[Index];
433 ++pCoord[Index];
434
435 _negative = { _id[Index] > 0U && operator[](utils::to_1d(nCoord, m_size)).is_active(), nCoord };
436 _positive = { _id[Index] < m_size[Index] - 1U && operator[](utils::to_1d(pCoord, m_size)).is_active(), pCoord };
437 }
