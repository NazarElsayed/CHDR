268 neighbour_t output{};

430 template<size_t Index>
431 constexpr void compute_single_diagonal(const coord_t& _id, neighbour_t& _output) const noexcept {
432
433 constexpr size_t sampleIndex = (Index >= s_neighbour_count / 2U) ? (Index + 1U) : Index; // Skips the middle value
434 constexpr coord_t direction = utils::to_nd(sampleIndex, coord_t { 3U });
435
436 bool oob = false;
437 coord_t coord = _id;
438
439 for (size_t i = 0U; i < s_rank; ++i) {
440 coord[i] += (direction[i] - 1U);
441
442 if (coord[i] >= m_size[i]) {
443 oob = true;
444
445 if constexpr (s_rank > 4U) {
446 break;
447 }
448 }
449 }
450
451 _output = { !oob && operator[](coord).is_active(), coord };
452 }

454 template<size_t Index>
455 constexpr void compute_single_axis(const coord_t& _id, neighbour_t& _negative, neighbour_t& _positive) const noexcept {
456
457 coord_t nCoord = _id; // Negative
458 coord_t pCoord = _id; // Positive
459
460 --nCoord[Index];
461 ++pCoord[Index];
462
463 _negative = { _id[Index] > 0U && operator[](nCoord).is_active(), nCoord };
464 _positive = { _id[Index] < m_size[Index] - 1U && operator[](pCoord).is_active(), pCoord };
