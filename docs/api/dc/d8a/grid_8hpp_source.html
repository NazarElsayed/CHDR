406 template<size_t Index>
407 constexpr void compute_single_diagonal(const coord_t& _id, neighbour_t& _output) const noexcept {
408
409 constexpr size_t sampleIndex = (Index >= s_neighbour_count / 2U) ? (Index + 1U) : Index;
410 constexpr coord_t direction = utils::to_nd(sampleIndex, coord_t { 3U });
411
412 bool oob = false;
413 coord_t coord = _id;
414
415 for (size_t j = 0U; j < s_rank; ++j) {
416 coord[j] += (direction[j] - 1U);
417 oob |= (coord[j] >= m_size[j]);
418
419 if constexpr (s_rank > 4U) {
420 if (oob) { break; }
421 }
422 }
423
424 _output = { !oob && at(coord).is_active(), coord };
425 }
426
427 template<size_t Index>
428 constexpr void compute_single_axis(const coord_t& _id, neighbour_t& _negative, neighbour_t& _positive) const noexcept {
429
430 coord_t nCoord = _id; // Negative
431 coord_t pCoord = _id; // Positive
432
433 --nCoord[Index];
434 ++pCoord[Index];
435
436 _negative = { _id[Index] > 0U && at(utils::to_1d(nCoord, m_size)).is_active(), nCoord };
437 _positive = { _id[Index] < m_size[Index] - 1U && at(utils::to_1d(pCoord, m_size)).is_active(), pCoord };
438 }
