471 template<size_t Index>
472 constexpr void compute_single_diagonal(const coord_t& _id, neighbour_t& _output) const noexcept {
473
474 constexpr size_t sampleIndex = (Index >= s_neighbour_count / 2U) ? (Index + 1U) : Index; // Skips the middle value
475 constexpr coord_t direction = utils::to_nd(sampleIndex, utils::make_kernel<coord_t>());
476
477 bool oob = false;
478 coord_t coord = _id;
479
480 for (size_t i = 0U; i < s_rank; ++i) {
481 coord[i] += (direction[i] - static_cast<typename coord_t::value_type>(1U));
482
483 if (coord[i] >= m_size[i]) {
484 oob = true;
485
486 if constexpr (s_rank > 4U) {
487 break;
488 }
489 }
490 }
491
492 _output = { !oob && operator[](coord).is_active(), coord };
493 }
494
495 template<size_t Index>
496 constexpr void compute_single_axis(const coord_t& _id, neighbour_t& _negative, neighbour_t& _positive) const noexcept {
497
498 coord_t nCoord = _id; // Negative
499 coord_t pCoord = _id; // Positive
500
501 --nCoord[Index];
502 ++pCoord[Index];
503
504 _negative = { _id[Index] > static_cast<typename coord_t::value_type>(0U) && operator[](nCoord).is_active(), nCoord };
505 _positive = { _id[Index] < m_size[Index] - static_cast<typename coord_t::value_type>(1U) && operator[](pCoord).is_active(), pCoord };
506 }
