122 template <typename open_set_t, typename closed_set_t>
123 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
124
125 const auto s = utils::to_1d(_params.start, _params.size);
126 const auto e = utils::to_1d(_params.end, _params.size);
127
128 _open.emplace_nosort(s, static_cast<scalar_t>(0), _params.h(_params.start, _params.end) * _params.weight);
129 _closed.emplace(s);
130
131 // Main loop:
132 while (LIKELY(!_open.empty())) {
133
134 auto curr(std::move(_open.top()));
135 _open.pop();
136
137 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
138
139 node* RESTRICT curr_ptr(nullptr);
140
141 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
142
143 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
144
145 // Check if node is not already visited:
146 if (!_closed.contains(n.index)) {
147 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
148
149 if (curr_ptr == nullptr) {
150 curr_ptr = new (_params.monotonic_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
151 }
152
153 if constexpr (params_t::lazy_sorting::value) {
154 _open.emplace_nosort(n.index, curr_ptr->m_gScore + n.distance, _params.h(n.coord, _params.end) * _params.weight, curr_ptr);
155 }
156 else {
157 _open.emplace(n.index, curr_ptr->m_gScore + n.distance, _params.h(n.coord, _params.end) * _params.weight, curr_ptr);
158 }
159 }
160 }
161 }
162 }
163 else { // SOLUTION REACHED ...
164
165 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
166 _open.wipe();
167 }
168 else {
169 _open = open_set_t{};
170 }
171 _closed = closed_set_t{};
172
173 return solver_t::solver_utils::rbacktrack(curr, _params.size, curr.m_gScore);
174 }
175 }
176
177 return std::vector<coord_t>{};
178 }

180 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
181
182 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
183
184 existence_set closed(_params.monotonic_pmr);
185 closed.reserve(capacity);
186
187 heap<node> open(_params.heterogeneous_pmr);
188 try {
189 open.reserve(capacity / 8U);
190 }
191 catch (...) {} // NOLINT(*-empty-catch)
192
193 return solve_internal(open, closed, capacity, _params);
194 }
