95 template <typename open_set_t, typename closed_set_t>
96 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
97
98 const auto s = utils::to_1d(_params.start, _params.size);
99 const auto e = utils::to_1d(_params.end, _params.size);
100
101 _open.emplace_nosort(s, _params.h(_params.start, _params.end));
102 _closed.emplace(s);
103
104 // Main loop:
105 while (LIKELY(!_open.empty())) {
106
107 auto curr(std::move(_open.top()));
108 _open.pop();
109
110 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
111
112 node* RESTRICT curr_ptr(nullptr);
113
114 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
115
116 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
117
118 // Check if node is not already visited:
119 if (!_closed.contains(n.index)) {
120 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
121
122 if (curr_ptr == nullptr) {
123 curr_ptr = new (_params.monotonic_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
124 }
125
126 if constexpr (params_t::lazy_sorting::value) {
127 _open.emplace_nosort(n.index, _params.h(n.coord, _params.end), curr_ptr);
128 }
129 else {
130 _open.emplace(n.index, _params.h(n.coord, _params.end), curr_ptr);
131 }
132 }
133 }
134 }
135 }
136 else { // SOLUTION REACHED ...
137
138 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
139 _open.wipe();
140 }
141 else {
142 _open = {};
143 }
144 _closed = {};
145
146 return solver_t::solver_utils::rbacktrack(curr, _params.size);
147 }
148 }
149
150 return std::vector<coord_t>{};
151 }

153 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
154
155 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
156
157 existence_set closed(_params.monotonic_pmr);
158 closed.reserve(capacity);
159
160 heap<node> open(_params.polytonic_pmr);
161 try {
162 open.reserve(capacity / 8U);
163 }
164 catch (...) {} // NOLINT(*-empty-catch)
165
166 return solve_internal(open, closed, capacity, _params);
167 }
