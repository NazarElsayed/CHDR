56 [[nodiscard]] constexpr node(index_t _index, const direction_t& _direction, scalar_t _gScore, scalar_t _hScore, node* RESTRICT const _parent = nullptr) noexcept : managed_node<index_t, node>(_index, _parent),

61 [[nodiscard]] HOT friend constexpr bool operator < (const node& _a, const node& _b) noexcept {
62 return _a.m_fScore == _b.m_fScore ?
63 _a.m_gScore > _b.m_gScore :
64 _a.m_fScore > _b.m_fScore;

68 static constexpr rotation_t s_identity { 0U, 1U, 2U,
69 3U, 4U,
70 5U, 6U, 7U };
71
72 static constexpr rotation_t s_rotate_l { 2U, 4U, 7U,
73 1U, 6U,
74 0U, 3U, 5U };
75
76 static constexpr rotation_t s_rotate_2 { 7U, 6U, 5U,
77 4U, 3U,
78 2U, 1U, 0U };
79
80 static constexpr rotation_t s_rotate_r { 5U, 3U, 0U,
81 6U, 1U,
82 7U, 4U, 2U };
83
84 static constexpr direction_t zero_direction_v { 3U };
85
86 static constexpr std::array<rotation_t, 9U> s_lookup {
87 /* { -1, -1 } : 0 */ s_rotate_2, // 0
88 /* { 0, -1 } : 1 */ s_rotate_r, // 1
89 /* { -1, 0 } : 2 */ s_rotate_2, // 2
90 /* { 0, 0 } : 3 */ s_identity, // 3
91 /* { 1, 0 } : 4 */ s_identity, // 4
92 /* { 0, 1 } : 5 */ s_rotate_l, // 5
93 /* { 1, 1 } : 6 */ s_identity, // 6
94 /* { 1, -1 } : 2 */ s_rotate_r, // 7
95 /* { -1, 1 } : 4 */ s_rotate_l, // 8
96 };

98 static constexpr bool is_straight(const direction_t& _direction) noexcept {
99 return _direction == 1U ||
100 _direction == 2U ||
101 _direction == 4U ||
102 _direction == 5U;
103 }

105 [[nodiscard]] static constexpr auto get_direction(const coord_t& _from, const coord_t& _to) {
106
107 coord_t dir;
108 for (size_t i = 0U; i < Kd; ++i) {
109 dir[i] = static_cast<index_t>(utils::sign<signed>(static_cast<signed>(_to[i]) - static_cast<signed>(_from[i])) + 1);
110 }
111
112 direction_t result{};
113
114 if (dir[0U] == 2U && dir[1U] == 0U) { result = 7U; }
115 else if (dir[0U] == 0U && dir[1U] == 2U) { result = 8U; }
116 else {
117 result = static_cast<direction_t>(utils::to_1d(dir, { Kd }));
118 }
119
120 return result;
121 }

123 template <typename maze_t>
124 [[nodiscard]] static constexpr neighbours_t go_find_jump_points(const maze_t& _maze, const coord_t& _current, const direction_t _direction, const coord_t& _end) {
125
126 constexpr auto null_v = std::make_pair(false, coord_t{});
127
128 const auto& neighbours = _maze.template get_neighbours<true>(_current);
129
130 if (UNLIKELY(_direction == zero_direction_v)) { // START NODE:
131
132 return {
133 neighbours[0U].first ? jump(_maze, neighbours[0U].second, _current, _end) : null_v, // FORCED
134 neighbours[1U].first ? jump(_maze, neighbours[1U].second, _current, _end) : null_v, // FORCED
135 neighbours[2U].first ? jump(_maze, neighbours[2U].second, _current, _end) : null_v, // FORCED
136 neighbours[3U].first ? jump(_maze, neighbours[3U].second, _current, _end) : null_v, // FORCED
137 neighbours[4U].first ? jump(_maze, neighbours[4U].second, _current, _end) : null_v, // FORCED
138 neighbours[5U].first ? jump(_maze, neighbours[5U].second, _current, _end) : null_v, // FORCED
139 neighbours[6U].first ? jump(_maze, neighbours[6U].second, _current, _end) : null_v, // FORCED
140 neighbours[7U].first ? jump(_maze, neighbours[7U].second, _current, _end) : null_v // FORCED
141 };
142 }
143 else {
144
145 const auto& map = s_lookup[static_cast<size_t>(_direction)];
146
147 if (is_straight(_direction)) { // STRAIGHT:
148
149 return {
150 neighbours[map[2U]].first && !neighbours[map[1U]].first ? jump(_maze, neighbours[map[2U]].second, _current, _end) : null_v, // FORCED
151 neighbours[map[7U]].first && !neighbours[map[6U]].first ? jump(_maze, neighbours[map[7U]].second, _current, _end) : null_v, // FORCED
152 neighbours[map[4U]].first ? jump(_maze, neighbours[map[4U]].second, _direction, _end) : null_v, // NATURAL
153 null_v, // NULL
154 null_v, // NULL
155 null_v, // NULL
156 null_v, // NULL
157 null_v // NULL
158 };
159 }
160 else if (neighbours[map[1U]].first || neighbours[map[3U]].first) { // DIAGONAL (if not blocked):
161
162 return {
163 neighbours[map[2U]].first && !neighbours[map[1U]].first ? jump(_maze, neighbours[map[2U]].second, _current, _end) : null_v, // FORCED
164 neighbours[map[5U]].first && !neighbours[map[3U]].first ? jump(_maze, neighbours[map[5U]].second, _current, _end) : null_v, // FORCED
165 neighbours[map[4U]].first ? jump(_maze, neighbours[map[4U]].second, _current, _end) : null_v, // NATURAL
166 neighbours[map[6U]].first ? jump(_maze, neighbours[map[6U]].second, _current, _end) : null_v, // NATURAL
167 neighbours[map[7U]].first ? jump(_maze, neighbours[map[7U]].second, _direction, _end) : null_v, // NATURAL
168 null_v, // NULL
169 null_v, // NULL
170 null_v // NULL
171 };
172 }
173 else {
174 return { null_v };
175 }
176 }
177 }

180 [[nodiscard]] static constexpr neighbour_t jump(const maze_t& _maze, const coord_t& _current, const coord_t& _previous, const coord_t& _end) {
181 return jump(_maze, _current, get_direction(_previous, _current), _end);

184 template <typename maze_t>
185 [[nodiscard]] static constexpr neighbour_t jump(const maze_t& _maze, const coord_t& _current, const direction_t& _direction, const coord_t& _end) {
186
187 if (UNLIKELY(_current == _end)) { // SOLUTION REACHED...
188 return { true, _current };
189 }
190 else { // SEARCH FOR SOLUTION...
191
192 const auto& neighbours = _maze.template get_neighbours<true>(_current);
193 const auto& map = s_lookup[static_cast<size_t>(_direction)];
194
195 const auto check_forced = [&neighbours, &map](size_t _a, size_t _b) ALWAYS_INLINE {
196 return neighbours[map[_a]].first && !neighbours[map[_b]].first;
197 };
198
199 if (is_straight(_direction)) { // STRAIGHT...
200
201 if (check_forced(2U, 1U) || check_forced(7U, 6U)) { // FORCED:
202 return { true, _current };
203 }
204 else if (neighbours[map[4U]].first) { // NATURAL:
205 return jump(_maze, neighbours[map[4U]].second, _direction, _end);
206 }
207 }
208 else if (neighbours[map[1U]].first || neighbours[map[3U]].first) { // DIAGONAL (if not blocked)...
209
210 if (check_forced(2U, 1U) || check_forced(5U, 3U)) { // FORCED:
211 return { true, _current };
212 }
213 else { // NATURAL:
214
215 for (const auto& i : { 4U, 6U }) {
216
217 if (neighbours[map[i]].first) {
218 if (jump(_maze, neighbours[map[i]].second, _current, _end).first) {
219 return { true, _current };
220 }
221 }
222 }
223
224 if (neighbours[map[7U]].first) {
225 return jump(_maze, neighbours[map[7U]].second, _direction, _end);
226 }
227 }
228 }
229 }
230
231 return { false, _current };
232 }

234 template <typename open_set_t, typename closed_set_t>
235 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
236
237 static_assert(std::is_base_of_v<mazes::grid<coord_t, weight_t>, std::remove_cv_t<std::remove_reference_t<decltype(_params.maze)>>>,
238 "GJPS only supports grid mazes.");
239
240 if constexpr (Kd == 2U) {
241
242 const auto s = utils::to_1d(_params.start, _params.size);
243 const auto e = utils::to_1d(_params.end , _params.size);
244
245 _open.emplace_nosort(s, zero_direction_v, static_cast<scalar_t>(0), _params.h(_params.start, _params.end) * _params.weight);
246 _closed.emplace(s);
247
248 // Main loop:
249 while (LIKELY(!_open.empty())) {
250
251 auto curr(std::move(_open.top()));
252 _open.pop();
253
254 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
255
256 node* RESTRICT curr_ptr(nullptr);
257
258 const auto coord = utils::to_nd(curr.m_index, _params.size);
259
260 for (const auto& n_data : go_find_jump_points(_params.maze, coord, curr.m_direction, _params.end)) {
261
262 if (const auto& [nActive, nCoord] = n_data; nActive) {
263
264 const auto n = utils::to_1d(nCoord, _params.size);
265
266 constexpr scalar_t nDistance{1};
267
268 if (!_closed.contains(n)) {
269 solver_t::solver_utils::preallocate_emplace(_closed, n, _capacity, _params.maze.count());
270
271 if (curr_ptr == nullptr) {
272 curr_ptr = new (_params.pool_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
273 }
274
275 if constexpr (params_t::lazy_sorting::value) {
276 _open.emplace_nosort(n, get_direction(coord, nCoord), curr_ptr->m_gScore + nDistance, _params.h(nCoord, _params.end) * _params.weight, curr_ptr);
277 }
278 else {
279 _open.emplace(n, get_direction(coord, nCoord), curr_ptr->m_gScore + nDistance, _params.h(nCoord, _params.end) * _params.weight, curr_ptr);
280 }
281 }
282 }
283 }
284
285 if (curr_ptr == nullptr) {
286 curr.expunge(_params.pool_pmr);
287 }
288 }
289 else { // SOLUTION REACHED ...
290
291 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
292 _open.wipe();
293 }
294 else {
295 _open = {};
296 }
297 _closed = {};
298
299 return solver_t::solver_utils::rbacktrack(curr, _params.size, curr.m_gScore);
300 }
301 }
302 }
303
304 return std::vector<coord_t>{};
305 }
306
307 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
308
309 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
310
311 existence_set closed(_params.monotonic_pmr);
312 closed.reserve(capacity);
313
314 heap<node> open(_params.polytonic_pmr);
315 try {
316 open.reserve(capacity / 8U);
317 }
318 catch (...) {} // NOLINT(*-empty-catch)
319
320 return solve_internal(open, closed, capacity, _params);
321 }
322 };
