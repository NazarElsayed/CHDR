103 template <typename open_set_t, typename closed_set_t>
104 [[maybe_unused, nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
105
106 const auto s = utils::to_1d(_params.start, _params.size);
107 const auto e = utils::to_1d(_params.end, _params.size);
108
109 _open.emplace_nosort(s, _params.h(_params.start, _params.end));
110 _closed.emplace(s);
111
112 // Main loop:
113 while (LIKELY(!_open.empty())) {
114
115 auto curr(std::move(_open.top()));
116 _open.pop();
117
118 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
119
120 node* RESTRICT curr_ptr(nullptr);
121
122 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
123
124 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
125
126 // Check if node is not already visited:
127 if (!_closed.contains(n.index)) {
128 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
129
130 if (curr_ptr == nullptr) {
131 curr_ptr = new (_params.homogeneous_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
132 }
133
134 if constexpr (params_t::lazy_sorting::value) {
135 _open.emplace_nosort(n.index, _params.h(n.coord, _params.end), curr_ptr);
136 }
137 else {
138 _open.emplace(n.index, _params.h(n.coord, _params.end), curr_ptr);
139 }
140 }
141 }
142 }
143
144 if (curr_ptr == nullptr) {
145 curr.expunge(_params.homogeneous_pmr);
146 }
147 }
148 else { // SOLUTION REACHED...
149
150 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
151 _open.wipe();
152 }
153 else {
154 _open = open_set_t{};
155 }
156 _closed = closed_set_t{};
157
158 return solver_t::solver_utils::rbacktrack(curr, _params.size);
159 }
160 }
161
162 return std::vector<coord_t>{};
163 }

165 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
166
167 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
168
169 existence_set closed(_params.monotonic_pmr);
170 closed.reserve(capacity);
171
172 heap<node> open(_params.heterogeneous_pmr);
173 try {
174 open.reserve(capacity / 8U);
175 }
176 catch (...) {} // NOLINT(*-empty-catch)
177
178 return solve_internal(open, closed, capacity, _params);
179 }
