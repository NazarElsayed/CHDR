77 template <typename open_set_t, typename closed_set_t>
78 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
79
80 const auto s = utils::to_1d(_params.start, _params.size);
81 const auto e = utils::to_1d(_params.end, _params.size);
82
83 _open.emplace(s);
84 _closed.emplace(s);
85
86 // Main loop:
87 while (LIKELY(!_open.empty())) {
88
89 auto curr(std::move(_open.top()));
90 _open.pop();
91
92 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
93
94 node* RESTRICT curr_ptr(nullptr);
95
96 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
97
98 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
99
100 // Check if node is not already visited:
101 if (!_closed.contains(n.index)) {
102 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
103
104 if (curr_ptr == nullptr) {
105 curr_ptr = new (_params.monotonic_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
106 }
107
108 _open.emplace(n.index, curr_ptr);
109 }
110 }
111 }
112 }
113 else { // SOLUTION REACHED...
114
115 _open = open_set_t{};
116 _closed = closed_set_t{};
117
118 return solver_t::solver_utils::rbacktrack(curr, _params.size);
119 }
120 }
121
122 return std::vector<coord_t>{};
123 }

125 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
126
127 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
128
129 existence_set closed(_params.monotonic_pmr);
130 closed.reserve(capacity);
131
132 stack<node> open(_params.heterogeneous_pmr);
133
134 return solve_internal(open, closed, capacity, _params);
135 }
