160 template <typename open_set_t>
161 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, const params_t& _params) {
162
163 const auto e = utils::to_1d(_params.end, _params.size);
164
165 _open.emplace_nosort(e, static_cast<scalar_t>(0));
166
167 multi_result result(_params.size, _params.capacity);
168
169 // Main loop:
170 while (LIKELY(!_open.empty())) {
171
172 auto _ = std::move(_open.top());
173 _open.pop();
174
175 auto& curr = result.m_data.emplace(_.m_index, std::move(_)).first->second;
176
177 for (const auto& n_data : _params.maze.template get_neighbours<params_t::octile_neighbours::value>(curr.m_index)) {
178
179 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
180
181 auto g = curr.m_gScore + n.distance;
182
183 if (const auto search = result.m_data.find(n.index); search != result.m_data.end()) {
184
185 if (auto& child = search->second; g < child.m_gScore) {
186
187 child.m_gScore = g;
188 child.m_parent = &curr;
189
190 if constexpr (params_t::lazy_sorting::value) {
191 _open.emplace_nosort(child);
192 }
193 else {
194 _open.emplace(child);
195 }
196 }
197 }
198 else {
199 if constexpr (params_t::lazy_sorting::value) {
200 _open.emplace_nosort(n.index, g, &curr);
201 }
202 else {
203 _open.emplace(n.index, g, &curr);
204 }
205 }
206 }
207 }
208 }
209
210 // TODO: return multi_result instead of std::vector<coord_t>.
211 return result.get(_params.start);
212 }
