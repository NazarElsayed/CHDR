137 explicit graph(const grid<coord_t, weight_t>& _grid, std::pmr::memory_resource* _resource = std::pmr::get_default_resource()) : m_entries(_resource) {
138
139 const auto size = _grid.size();
140
141 if constexpr (!Prune) {
142
143 index_t index {0};
144 for (auto& element : _grid) {
145
146 if (element.is_active()) {
147
148 for (auto& neighbour : _grid.get_neighbours(index)) {
149
150 if (const auto& [nActive, nCoord] = neighbour; nActive) {
151 add(index, std::make_pair(utils::to_1d(nCoord, size), static_cast<scalar_t>(1)));
152 }
153 }
154 }
155
156 ++index;
157 }
158 }
159 else {
160
161 // TODO: Add support for directed graphs.
162 std::mutex mtx;
163
164 const auto worker = [&](index_t _start, index_t _end) ALWAYS_INLINE {
165
166 std::pmr::unsynchronized_pool_resource pool;
167 stack<edge_t> stack(&pool);
168
169 std::pmr::unordered_set<index_t> global_closed(&pool);
170 std::pmr::unordered_set<index_t> local_closed(&pool);
171
172 std::pmr::unordered_map<index_t, neighbours_t> thread_connections(&pool);
173
174 for (auto index = _start; index < _end; ++index) {
175
176 if (const auto& element = _grid.at(index); element.is_active()) {
177
178 global_closed.clear();
179 global_closed.insert(index);
180
181 const auto& neighbours = _grid.get_neighbours(index);
182
183 if (!_grid.is_transitory(neighbours)) {
184
185 for (const auto& n1 : neighbours) {
186
187 if (const auto& [nActive1, nCoord1] = n1; nActive1) {
188
189 const auto nIdx1 = static_cast<index_t>(utils::to_1d(nCoord1, size));
190
191 local_closed.clear();
192 stack.emplace(std::make_pair(nIdx1, static_cast<scalar_t>(1)));
193
194 while (!stack.empty()) {
195
196 const auto [currIdx, currDistance] = std::move(stack.top());
197 stack.pop();
198
199 if (local_closed.find(currIdx) == local_closed.end()) {
200
201 local_closed.insert(currIdx);
202 global_closed.insert(currIdx);
203
204 for (const auto& n3 : _grid.get_neighbours(currIdx)) {
205
206 if (const auto& [sActive3, sCoord3] = n3; sActive3) {
207
208 const auto s = static_cast<index_t>(utils::to_1d(sCoord3, size));
209
210 if (global_closed.find(s) == global_closed.end()) {
211
212 const edge_t next = std::make_pair(s, currDistance + static_cast<scalar_t>(1));
213
214 if (_grid.is_transitory(s)) {
215 stack.emplace(std::move(next));
216 }
217 else {
218 thread_connections[nIdx1].emplace_back(std::move(next));
219 stack.clear();
220 break;
221 }
222 }
223 }
224 }
225 }
226 }
227
228 for (const auto& item : thread_connections[nIdx1]) {
229 const std::lock_guard lock(mtx);
230 add(index, item);
231 }
232 }
233 }
234 }
235 }
236 }
237 };
238
239 // Create a thread pool to perform work in parallel:
240 const size_t count = _grid.count();
241 const size_t numThreads = utils::clamp(std::thread::hardware_concurrency(), 1U, 6U);
242
243 const auto chunkSize = static_cast<index_t>((count + numThreads - 1U) / numThreads);
244
245 std::vector<std::future<void>> futures;
246 futures.reserve(numThreads);
247
248 for (size_t i = 0U; i < numThreads; ++i) {
249
250 const index_t start = static_cast<index_t>(i) * chunkSize;
251 const index_t end = utils::min(start + chunkSize, static_cast<index_t>(count));
252
253 futures.emplace_back(std::async(std::launch::async, worker, start, end));
254 }
255
256 for (auto& fut : futures) {
257 fut.get();
258 }
259
260 // Finalise the pruning process on a single thread.
261 prune<ConsolidateAfterPrune>();
262 }
263 }

364 template <bool ConsolidateAfterPrune = true>
365 [[maybe_unused]] void prune() {
366
367 // TODO: Extend support for pruning directed graphs.
368
369 std::vector<index_t> nodesToRemove;
370
371 bool dirty = false;
372
373 do {
374 nodesToRemove.clear();
375
376 for (const auto& entry : m_entries) {
377
378 if (m_entries.size() > 2U) {
379
380 if (auto& [node, neighbours] = entry; neighbours.size() == 2U) {
381
382 dirty = ConsolidateAfterPrune;
383
384 auto it = neighbours.begin();
385
386 auto& [n1_id, n1_cost] = *it;
387 auto& [n2_id, n2_cost] = *(++it);
388
389 auto s1 = m_entries.find(n1_id);
390 auto s2 = m_entries.find(n2_id);
391
392 // Merge the connections from the current node with its neighbours:
393
394 if (s1 != m_entries.end() && n1_id != n2_id) {
395 auto& set = s1->second;
396 auto sn = std::find(set.begin(), set.end(), std::make_pair(node, n1_cost));
397
398 if (sn != set.end()) {
399 neighbour_t new_vertex{n2_id, sn->second + n2_cost};
400 if (std::find(set.begin(), set.end(), new_vertex) == set.end()) {
401 set.emplace_back(new_vertex);
402 }
403 }
404 }
405
406 if (s2 != m_entries.end() && n1_id != n2_id) {
407 auto& set = s2->second;
408 auto sn = std::find(set.begin(), set.end(), std::make_pair(node, n2_cost));
409
410 if (sn != set.end()) {
411 neighbour_t new_vertex{n1_id, sn->second + n1_cost};
412 set.erase(sn);
413 if (std::find(set.begin(), set.end(), new_vertex) == set.end()) {
414 set.emplace_back(new_vertex);
415 }
416 }
417 }
418
419 // Remove the current node from the graph.
420 nodesToRemove.emplace_back(node);
421 }
422 }
423 else {
424 break;
425 }
426 }
427 for (const auto& node : nodesToRemove) {
428 m_entries.erase(node);
429 }
430 }
431 while (!nodesToRemove.empty() && m_entries.size() > 2U);
432
433 if (dirty) {
434 chdr::malloc_consolidate();
435 }
436 }
