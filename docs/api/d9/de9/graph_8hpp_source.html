143 explicit graph(const grid<coord_t, weight_t>& _grid, std::pmr::memory_resource* _resource = std::pmr::get_default_resource()) : m_entries(_resource) {
144
145 const auto size = _grid.size();
146
147 if constexpr (!Prune) {
148
149 index_t index {0};
150 for (auto& element : _grid) {
151
152 if (element.is_active()) {
153
154 for (auto& neighbour : _grid.get_neighbours(index)) {
155
156 if (const auto& [nActive, nCoord] = neighbour; nActive) {
157 add(index, std::make_pair(utils::to_1d(nCoord, size), static_cast<scalar_t>(1)));
158 }
159 }
160 }
161
162 ++index;
163 }
164 }
165 else {
166
167 // TODO: Add support for directed graphs.
168 std::mutex mtx;
169
170 const auto worker = [&](index_t _start, index_t _end) ALWAYS_INLINE {
171
172 std::pmr::unsynchronized_pool_resource pool;
173 stack<edge_t> stack(&pool);
174
175 std::pmr::unordered_set<index_t> global_closed(&pool);
176 std::pmr::unordered_set<index_t> local_closed(&pool);
177
178 std::pmr::unordered_map<index_t, neighbours_t> thread_connections(&pool);
179
180 for (auto index = _start; index < _end; ++index) {
181
182 if (const auto& element = _grid[index]; element.is_active()) {
183
184 global_closed.clear();
185 global_closed.insert(index);
186
187 if (const auto& neighbours = _grid.get_neighbours(index); !_grid.is_transitory(neighbours)) {
188
189 for (const auto& n1 : neighbours) {
190
191 if (const auto& [nActive1, nCoord1] = n1; nActive1) {
192
193 const auto nIdx1 = static_cast<index_t>(utils::to_1d(nCoord1, size));
194
195 local_closed.clear();
196 stack.emplace(std::make_pair(nIdx1, static_cast<scalar_t>(1)));
197
198 while (!stack.empty()) {
199
200 const auto [currIdx, currDistance] = std::move(stack.top());
201 stack.pop();
202
203 if (local_closed.find(currIdx) == local_closed.end()) {
204
205 local_closed.insert(currIdx);
206 global_closed.insert(currIdx);
207
208 for (const auto& n3 : _grid.get_neighbours(currIdx)) {
209
210 if (const auto& [sActive3, sCoord3] = n3; sActive3) {
211
212 if (const auto s = static_cast<index_t>(utils::to_1d(sCoord3, size));
213 global_closed.find(s) == global_closed.end()
214 ) {
215
216 const edge_t next = std::make_pair(s, currDistance + static_cast<scalar_t>(1));
217
218 if (_grid.is_transitory(s)) {
219 stack.emplace(std::move(next));
220 }
221 else {
222 thread_connections[nIdx1].emplace_back(std::move(next));
223 stack.clear();
224 break;
225 }
226 }
227 }
228 }
229 }
230 }
231
232 for (const auto& item : thread_connections[nIdx1]) {
233 const std::lock_guard lock(mtx);
234 add(index, item);
235 }
236 }
237 }
238 }
239 }
240 }
241 };
242
243 // Create a thread pool to perform work in parallel:
244 const size_t count = _grid.count();
245 const size_t numThreads = utils::clamp(std::thread::hardware_concurrency(), 1U, 6U);
246
247 const auto chunkSize = static_cast<index_t>((count + numThreads - 1U) / numThreads);
248
249 std::array<std::future<void>, 6U> futures;
250
251 for (size_t i = 0U; i < numThreads; ++i) {
252 assert(i < numThreads && "Error: Index 'i' exceeds the number of initialised threads.");
253
254 const index_t start = static_cast<index_t>(i) * chunkSize;
255 const index_t end = utils::min(start + chunkSize, static_cast<index_t>(count));
256
257 futures[i] = std::async(std::launch::async, worker, start, end);
258 }
259
260 for (size_t i = 0U; i < numThreads; ++i) {
261 assert(i < numThreads && "Error: Index 'i' exceeds the number of initialised threads.");
262
263 futures[i].get();
264 }
265
266 // Finalise the pruning process on a single thread.
267 prune<ConsolidateAfterPrune>();
268 }

359 template <bool ConsolidateAfterPrune = true>
360 [[maybe_unused]] void prune() {
361
362 // TODO: Extend support for pruning directed graphs.
363
364 std::vector<index_t> nodesToRemove;
365
366 bool dirty = false;
367
368 do {
369 nodesToRemove.clear();
370
371 for (const auto& entry : m_entries) {
372
373 if (m_entries.size() > 2U) {
374
375 if (auto& [node, neighbours] = entry; neighbours.size() == 2U) {
376
377 dirty = ConsolidateAfterPrune;
378
379 auto it = neighbours.begin();
380
381 auto& [n1_id, n1_cost] = *it;
382 auto& [n2_id, n2_cost] = *(++it);
383
384 auto s1 = m_entries.find(n1_id);
385 auto s2 = m_entries.find(n2_id);
386
387 // Merge the connections from the current node with its neighbours:
388
389 if (s1 != m_entries.end() && n1_id != n2_id) {
390 auto& set = s1->second;
391 auto sn = std::find(set.begin(), set.end(), std::make_pair(node, n1_cost));
392
393 if (sn != set.end()) {
394
395 if (neighbour_t new_vertex { n2_id, sn->second + n2_cost };
396 std::find(set.begin(), set.end(), new_vertex) == set.end()
397 ) {
398 set.emplace_back(new_vertex);
399 }
400 }
401 }
402
403 if (s2 != m_entries.end() && n1_id != n2_id) {
404 auto& set = s2->second;
405 auto sn = std::find(set.begin(), set.end(), std::make_pair(node, n2_cost));
406
407 if (sn != set.end()) {
408 neighbour_t new_vertex{n1_id, sn->second + n1_cost};
409 set.erase(sn);
410 if (std::find(set.begin(), set.end(), new_vertex) == set.end()) {
411 set.emplace_back(new_vertex);
412 }
413 }
414 }
415
416 // Remove the current node from the graph.
417 nodesToRemove.emplace_back(node);
418 }
419 }
420 else {
421 break;
422 }
423 }
424 for (const auto& node : nodesToRemove) {
425 m_entries.erase(node);
426 }
427 }
428 while (!nodesToRemove.empty() && m_entries.size() > 2U);
429
430 if (dirty) {
431 chdr::malloc_consolidate();
432 }
433 }
