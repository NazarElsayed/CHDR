141 explicit graph(const grid<coord_t, weight_t>& _grid, std::pmr::memory_resource* _resource = std::pmr::get_default_resource()) : m_entries(_resource) {
142
143 const auto size = _grid.size();
144
145 if constexpr (!Prune) {
146
147 index_t index {0};
148 for (auto& element : _grid) {
149
150 if (element.is_active()) {
151
152 for (auto& neighbour : _grid.get_neighbours(index)) {
153
154 if (const auto& [nActive, nCoord] = neighbour; nActive) {
155 add(index, std::make_pair(utils::to_1d(nCoord, size), static_cast<scalar_t>(1)));
156 }
157 }
158 }
159
160 ++index;
161 }
162 }
163 else {
164
165 // TODO: Add support for directed graphs.
166 std::mutex mtx;
167
168 const auto worker = [&](index_t _start, index_t _end) ALWAYS_INLINE {
169
170 std::pmr::unsynchronized_pool_resource pool;
171 stack<edge_t> stack(&pool);
172
173 std::pmr::unordered_set<index_t> global_closed(&pool);
174 std::pmr::unordered_set<index_t> local_closed(&pool);
175
176 std::pmr::unordered_map<index_t, neighbours_t> thread_connections(&pool);
177
178 for (auto index = _start; index < _end; ++index) {
179
180 if (const auto& element = _grid.at(index); element.is_active()) {
181
182 global_closed.clear();
183 global_closed.insert(index);
184
185 const auto& neighbours = _grid.get_neighbours(index);
186
187 if (!_grid.is_transitory(neighbours)) {
188
189 for (const auto& n1 : neighbours) {
190
191 if (const auto& [nActive1, nCoord1] = n1; nActive1) {
192
193 const auto nIdx1 = static_cast<index_t>(utils::to_1d(nCoord1, size));
194
195 local_closed.clear();
196 stack.emplace(std::make_pair(nIdx1, static_cast<scalar_t>(1)));
197
198 while (!stack.empty()) {
199
200 const auto [currIdx, currDistance] = std::move(stack.top());
201 stack.pop();
202
203 if (local_closed.find(currIdx) == local_closed.end()) {
204
205 local_closed.insert(currIdx);
206 global_closed.insert(currIdx);
207
208 for (const auto& n3 : _grid.get_neighbours(currIdx)) {
209
210 if (const auto& [sActive3, sCoord3] = n3; sActive3) {
211
212 const auto s = static_cast<index_t>(utils::to_1d(sCoord3, size));
213
214 if (global_closed.find(s) == global_closed.end()) {
215
216 const edge_t next = std::make_pair(s, currDistance + static_cast<scalar_t>(1));
217
218 if (_grid.is_transitory(s)) {
219 stack.emplace(std::move(next));
220 }
221 else {
222 thread_connections[nIdx1].emplace_back(std::move(next));
223 stack.clear();
224 break;
225 }
226 }
227 }
228 }
229 }
230 }
231
232 for (const auto& item : thread_connections[nIdx1]) {
233 const std::lock_guard lock(mtx);
234 add(index, item);
235 }
236 }
237 }
238 }
239 }
240 }
241 };
242
243 // Create a thread pool to perform work in parallel:
244 const size_t count = _grid.count();
245 const size_t numThreads = utils::clamp(std::thread::hardware_concurrency(), 1U, 6U);
246
247 const auto chunkSize = static_cast<index_t>((count + numThreads - 1U) / numThreads);
248
249 std::vector<std::future<void>> futures;
250 futures.reserve(numThreads);
251
252 for (size_t i = 0U; i < numThreads; ++i) {
253
254 const index_t start = static_cast<index_t>(i) * chunkSize;
255 const index_t end = utils::min(start + chunkSize, static_cast<index_t>(count));
256
257 futures.emplace_back(std::async(std::launch::async, worker, start, end));
258 }
259
260 for (auto& fut : futures) {
261 fut.get();
262 }
263
264 // Finalise the pruning process on a single thread.
265 prune<ConsolidateAfterPrune>();
266 }
267 }

368 template <bool ConsolidateAfterPrune = true>
369 [[maybe_unused]] void prune() {
370
371 // TODO: Extend support for pruning directed graphs.
372
373 std::vector<index_t> nodesToRemove;
374
375 bool dirty = false;
376
377 do {
378 nodesToRemove.clear();
379
380 for (const auto& entry : m_entries) {
381
382 if (m_entries.size() > 2U) {
383
384 if (auto& [node, neighbours] = entry; neighbours.size() == 2U) {
385
386 dirty = ConsolidateAfterPrune;
387
388 auto it = neighbours.begin();
389
390 auto& [n1_id, n1_cost] = *it;
391 auto& [n2_id, n2_cost] = *(++it);
392
393 auto s1 = m_entries.find(n1_id);
394 auto s2 = m_entries.find(n2_id);
395
396 // Merge the connections from the current node with its neighbours:
397
398 if (s1 != m_entries.end() && n1_id != n2_id) {
399 auto& set = s1->second;
400 auto sn = std::find(set.begin(), set.end(), std::make_pair(node, n1_cost));
401
402 if (sn != set.end()) {
403 neighbour_t new_vertex{n2_id, sn->second + n2_cost};
404 if (std::find(set.begin(), set.end(), new_vertex) == set.end()) {
405 set.emplace_back(new_vertex);
406 }
407 }
408 }
409
410 if (s2 != m_entries.end() && n1_id != n2_id) {
411 auto& set = s2->second;
412 auto sn = std::find(set.begin(), set.end(), std::make_pair(node, n2_cost));
413
414 if (sn != set.end()) {
415 neighbour_t new_vertex{n1_id, sn->second + n1_cost};
416 set.erase(sn);
417 if (std::find(set.begin(), set.end(), new_vertex) == set.end()) {
418 set.emplace_back(new_vertex);
419 }
420 }
421 }
422
423 // Remove the current node from the graph.
424 nodesToRemove.emplace_back(node);
425 }
426 }
427 else {
428 break;
429 }
430 }
431 for (const auto& node : nodesToRemove) {
432 m_entries.erase(node);
433 }
434 }
435 while (!nodesToRemove.empty() && m_entries.size() > 2U);
436
437 if (dirty) {
438 chdr::malloc_consolidate();
439 }
440 }
