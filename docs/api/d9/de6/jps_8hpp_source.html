187 [[nodiscard]] static constexpr auto get_direction(const coord_t& _from, const coord_t& _to) {
188
189 coord_t dir{};
190 for (size_t i = 0U; i < Kd; ++i) {
191 dir[i] = static_cast<index_t>(utils::sign<signed>(static_cast<signed>(_to[i]) - static_cast<signed>(_from[i])) + 1);
192 }
193
194 direction_t result{};
195
196 if (dir[0U] == 0U && dir[1U] == 0U) { result = TL; }
197 else if (dir[0U] == 1U && dir[1U] == 0U) { result = TM; }
198 else if (dir[0U] == 2U && dir[1U] == 0U) { result = TR; }
199 else if (dir[0U] == 0U && dir[1U] == 1U) { result = ML; }
200 else if (dir[0U] == 2U && dir[1U] == 1U) { result = MR; }
201 else if (dir[0U] == 0U && dir[1U] == 2U) { result = BL; }
202 else if (dir[0U] == 1U && dir[1U] == 2U) { result = BM; }
203 else if (dir[0U] == 2U && dir[1U] == 2U) { result = BR; }
204 else {
205 result = ZZ;
206 }
207
208 return result;

212 [[nodiscard]] static constexpr neighbours_t go_find_jump_points(const maze_t& _maze, const coord_t& _current, const direction_t _direction, const coord_t& _end) {
213
214 constexpr auto null_v = std::make_pair(false, coord_t{});
215
216 const auto& neighbours = _maze.template get_neighbours<true>(_current);
217
218 if (UNLIKELY(_direction == ZZ)) { // START NODE:
219
220 return {
221 neighbours[TL].first ? jump(_maze, neighbours[TL].second, _current, _end) : null_v, // FORCED
222 neighbours[TM].first ? jump(_maze, neighbours[TM].second, _current, _end) : null_v, // FORCED
223 neighbours[TR].first ? jump(_maze, neighbours[TR].second, _current, _end) : null_v, // FORCED
224 neighbours[ML].first ? jump(_maze, neighbours[ML].second, _current, _end) : null_v, // FORCED
225 neighbours[MR].first ? jump(_maze, neighbours[MR].second, _current, _end) : null_v, // FORCED
226 neighbours[BL].first ? jump(_maze, neighbours[BL].second, _current, _end) : null_v, // FORCED
227 neighbours[BM].first ? jump(_maze, neighbours[BM].second, _current, _end) : null_v, // FORCED
228 neighbours[BR].first ? jump(_maze, neighbours[BR].second, _current, _end) : null_v // FORCED
229 };
230 }
231 else {
232
233 const auto& map = s_lookup[static_cast<size_t>(_direction)];
234
235 if (is_straight(_direction)) { // STRAIGHT:
236
237 return {
238 neighbours[map[TR]].first && !neighbours[map[TM]].first ? jump(_maze, neighbours[map[TR]].second, _current, _end) : null_v, // FORCED
239 neighbours[map[BR]].first && !neighbours[map[BM]].first ? jump(_maze, neighbours[map[BR]].second, _current, _end) : null_v, // FORCED
240 neighbours[map[MR]].first ? jump(_maze, neighbours[map[MR]].second, _direction, _end) : null_v, // NATURAL
241 null_v, // NULL
242 null_v, // NULL
243 null_v, // NULL
244 null_v, // NULL
245 null_v // NULL
246 };
247 }
248 else if (neighbours[map[TM]].first || neighbours[map[ML]].first) { // DIAGONAL (if not blocked):
249
250 return {
251 neighbours[map[TR]].first && !neighbours[map[TM]].first ? jump(_maze, neighbours[map[TR]].second, _current, _end) : null_v, // FORCED
252 neighbours[map[BL]].first && !neighbours[map[ML]].first ? jump(_maze, neighbours[map[BL]].second, _current, _end) : null_v, // FORCED
253 neighbours[map[MR]].first ? jump(_maze, neighbours[map[MR]].second, _current, _end) : null_v, // NATURAL
254 neighbours[map[BM]].first ? jump(_maze, neighbours[map[BM]].second, _current, _end) : null_v, // NATURAL
255 neighbours[map[BR]].first ? jump(_maze, neighbours[map[BR]].second, _direction, _end) : null_v, // NATURAL
256 null_v, // NULL
257 null_v, // NULL
258 null_v // NULL
259 };
260 }
261 else {

272 template <typename maze_t>
273 [[nodiscard]] static constexpr neighbour_t jump(const maze_t& _maze, const coord_t& _current, const direction_t& _direction, const coord_t& _end) {
274
275 if (UNLIKELY(_current == _end)) { // SOLUTION REACHED...
276 return { true, _current };
277 }
278 else { // SEARCH FOR SOLUTION...
279
280 const auto& neighbours = _maze.template get_neighbours<true>(_current);
281 const auto& map = s_lookup[static_cast<size_t>(_direction)];
282
283 const auto check_forced = [&neighbours, &map](const size_t _a, const size_t _b) ALWAYS_INLINE {
284 return neighbours[map[_a]].first && !neighbours[map[_b]].first;
285 };
286
287 if (is_straight(_direction)) { // STRAIGHT...
288
289 if (check_forced(TR, TM) || check_forced(BR, BM)) { // FORCED:
290 return { true, _current };
291 }
292 else if (neighbours[map[MR]].first) { // NATURAL:
293 return jump(_maze, neighbours[map[MR]].second, _direction, _end);
294 }
295 }
296 else if (neighbours[map[TM]].first || neighbours[map[ML]].first) { // DIAGONAL (if not blocked)...
297
298 if (check_forced(TR, TM) || check_forced(BL, ML)) { // FORCED:
299 return { true, _current };
300 }
301 else { // NATURAL:
302
303 for (const auto& i : { MR, BM }) {
304
305 if (neighbours[map[i]].first) {
306 if (jump(_maze, neighbours[map[i]].second, _current, _end).first) {
307 return { true, _current };
308 }
309 }
310 }
311
312 if (neighbours[map[BR]].first) {
313 return jump(_maze, neighbours[map[BR]].second, _direction, _end);
314 }
315 }
316 }
317 }
318
319 return { false, _current };
320 }

322 template <typename open_set_t, typename closed_set_t>
323 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
324
325 static_assert(std::is_base_of_v<mazes::grid<coord_t, weight_t>, std::remove_cv_t<std::remove_reference_t<decltype(_params.maze)>>>,
326 "JPS only supports grid mazes.");
327
328 __CHDR_UTILS_STATIC_WARNING_NIF(params_t::octile_neighbours::value, "This implementations of JPS only supports octile grids.");
329
330 if constexpr (Kd == 2U) {
331
332 const auto& end = params_t::reverse_equivalence::value ? _params.start : _params.end;
333 const auto& start = params_t::reverse_equivalence::value ? _params.end : _params.start;
334
335 const auto s = utils::to_1d(start, _params.size);
336 const auto e = utils::to_1d(end, _params.size);
337
338 _open.emplace_nosort(s, ZZ, static_cast<scalar_t>(0), _params.h(start, end) * _params.weight);
339 _closed.emplace(s);
340
341 // Main loop:
342 while (LIKELY(!_open.empty())) {
343
344 auto curr(std::move(_open.top()));
345 _open.pop();
346
347 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
348
349 node* RESTRICT curr_ptr(nullptr);
350
351 const auto coord = utils::to_nd(curr.m_index, _params.size);
352
353 for (const auto& n_data : go_find_jump_points(_params.maze, coord, curr.m_direction, end)) {
354
355 if (const auto& [nActive, nCoord] = n_data; nActive) {
356
357 const auto n = utils::to_1d(nCoord, _params.size);
358
359 const scalar_t nDistance = heuristics::octile_distance<scalar_t>(coord, nCoord);
360
361 if (!_closed.contains(n)) {
362 solver_t::solver_utils::preallocate_emplace(_closed, n, _capacity, _params.maze.count());
363
364 if (curr_ptr == nullptr) {
365 curr_ptr = new (_params.monotonic_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
366 }
367
368 if constexpr (params_t::lazy_sorting::value) {
369 _open.emplace_nosort(n, get_direction(coord, nCoord), curr_ptr->m_gScore + nDistance, _params.h(nCoord, end) * _params.weight, curr_ptr);
370 }
371 else {
372 _open.emplace(n, get_direction(coord, nCoord), curr_ptr->m_gScore + nDistance, _params.h(nCoord, end) * _params.weight, curr_ptr);
373 }
374 }
375 }
376 }
377 }
378 else { // SOLUTION REACHED...
379
380 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
381 _open.wipe();
382 }
383 else {
384 _open = open_set_t{};
385 }
386 _closed = closed_set_t{};
387
388 if constexpr (params_t::reverse_equivalence::value) {
389 return solver_t::solver_utils::rbacktrack(curr, _params.size, static_cast<size_t>(curr.m_gScore));
390 }
391 else {
392 return solver_t::solver_utils::rbacktrack_noreverse(curr, _params.size, static_cast<size_t>(curr.m_gScore));
393 }
394 }
395 }
396 }
397
398 return std::vector<coord_t>{};
399 }

401 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
402
403 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
404
405 existence_set closed(_params.monotonic_pmr);
406 closed.reserve(capacity);
407
408 heap<node> open(_params.heterogeneous_pmr);
409 try {
410 open.reserve(capacity / 8U);
411 }
412 catch (...) {} // NOLINT(*-empty-catch)
413
414 return solve_internal(open, closed, capacity, _params);
415 }
