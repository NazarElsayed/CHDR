212 [[nodiscard]] static constexpr auto get_direction(const coord_t& _from, const coord_t& _to) {
213
214 coord_t dir{};
215 for (size_t i = 0U; i < Kd; ++i) {
216 dir[i] = static_cast<index_t>(utils::sign<signed>(static_cast<signed>(_to[i]) - static_cast<signed>(_from[i])) + 1);
217 }
218
219 direction_t result{};
220
221 if (dir[0U] == 0U && dir[1U] == 0U) { result = TL; }
222 else if (dir[0U] == 1U && dir[1U] == 0U) { result = TM; }
223 else if (dir[0U] == 2U && dir[1U] == 0U) { result = TR; }
224 else if (dir[0U] == 0U && dir[1U] == 1U) { result = ML; }
225 else if (dir[0U] == 2U && dir[1U] == 1U) { result = MR; }
226 else if (dir[0U] == 0U && dir[1U] == 2U) { result = BL; }
227 else if (dir[0U] == 1U && dir[1U] == 2U) { result = BM; }
228 else if (dir[0U] == 2U && dir[1U] == 2U) { result = BR; }
229 else {
230 result = ZZ;
231 }
232
233 return result;

297 template <typename maze_t>
298 [[nodiscard]] static constexpr neighbour_t jump(const maze_t& _maze, const coord_t& _current, const direction_t& _direction, const coord_t& _end) {
299
300 if (UNLIKELY(_current == _end)) { // SOLUTION REACHED...
301 return { true, _current };
302 }
303 else { // SEARCH FOR SOLUTION...
304
305 const auto& neighbours = _maze.template get_neighbours<true>(_current);
306 const auto& map = s_lookup[static_cast<size_t>(_direction)];
307
308 const auto check_forced = [&neighbours, &map](const size_t _a, const size_t _b) ALWAYS_INLINE {
309 return neighbours[map[_a]].first && !neighbours[map[_b]].first;
310 };
311
312 if (is_straight(_direction)) { // STRAIGHT...
313
314 if (check_forced(TR, TM) || check_forced(BR, BM)) { // FORCED:
315 return { true, _current };
316 }
317 else if (neighbours[map[MR]].first) { // NATURAL:
318 return jump(_maze, neighbours[map[MR]].second, _direction, _end);
319 }
320 }
321 else if (neighbours[map[TM]].first || neighbours[map[ML]].first) { // DIAGONAL (if not blocked)...
322
323 if (check_forced(TR, TM) || check_forced(BL, ML)) { // FORCED:
324 return { true, _current };
325 }
326 else { // NATURAL:
327
328 for (const auto& i : { MR, BM }) {
329
330 if (neighbours[map[i]].first) {
331 if (jump(_maze, neighbours[map[i]].second, _current, _end).first) {
332 return { true, _current };
333 }
334 }
335 }
336
337 if (neighbours[map[BR]].first) {
338 return jump(_maze, neighbours[map[BR]].second, _direction, _end);
339 }
340 }
341 }
342 }
343
344 return { false, _current };
345 }

348 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
349
350 static_assert(std::is_base_of_v<mazes::grid<coord_t, weight_t>, std::remove_cv_t<std::remove_reference_t<decltype(_params.maze)>>>,
351 "JPS only supports grid mazes.");
352
353 if constexpr (Kd == 2U) {
354
355 const auto s = utils::to_1d(_params.start, _params.size);
356 const auto e = utils::to_1d(_params.end , _params.size);
357
358 _open.emplace_nosort(s, ZZ, static_cast<scalar_t>(0), _params.h(_params.start, _params.end) * _params.weight);
359 _closed.emplace(s);
360
361 // Main loop:
362 while (LIKELY(!_open.empty())) {
363
364 auto curr(std::move(_open.top()));
365 _open.pop();
366
367 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
368
369 node* RESTRICT curr_ptr(nullptr);
370
371 const auto coord = utils::to_nd(curr.m_index, _params.size);
372
373 for (const auto& n_data : go_find_jump_points(_params.maze, coord, curr.m_direction, _params.end)) {
374
375 if (const auto& [nActive, nCoord] = n_data; nActive) {
376
377 const auto n = utils::to_1d(nCoord, _params.size);
378
379 const scalar_t nDistance = _params.h(coord, nCoord) * _params.weight;
380
381 if (!_closed.contains(n)) {
382 solver_t::solver_utils::preallocate_emplace(_closed, n, _capacity, _params.maze.count());
383
384 if (curr_ptr == nullptr) {
385 curr_ptr = new (_params.monotonic_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
386 }
387
388 // if constexpr (params_t::lazy_sorting::value) {
389 // _open.emplace_nosort(n, get_direction(coord, nCoord), curr_ptr->m_gScore + nDistance, _params.h(nCoord, _params.end) * _params.weight, curr_ptr);
390 // }
391 // else {
392 _open.emplace(n, get_direction(coord, nCoord), curr_ptr->m_gScore + nDistance, _params.h(nCoord, _params.end) * _params.weight, curr_ptr);
393 // }
394 }
395 }
396 }
397 }
398 else { // SOLUTION REACHED ...
399
400 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
401 _open.wipe();
402 }
403 else {
404 _open = open_set_t{};
405 }
406 _closed = closed_set_t{};
407
408 return solver_t::solver_utils::rbacktrack(curr, _params.size, curr.m_gScore);
409 }
410 }
411 }
412
413 return std::vector<coord_t>{};

416 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
417
418 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
419
420 existence_set closed(_params.monotonic_pmr);
421 closed.reserve(capacity);
422
423 heap<node> open(_params.heterogeneous_pmr);
424 try {
425 open.reserve(capacity / 8U);
426 }
427 catch (...) {} // NOLINT(*-empty-catch)
428
429 return solve_internal(open, closed, capacity, _params);
430 }
