109 HOT uint8_t* expand(size_t _bytes, size_t _alignment) noexcept {
110
111 assert(_bytes > 0U && "Allocation size must be greater than zero.");
112 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
113
114 const auto allocate_size = utils::max(m_block_width, _bytes);
115
116 uint8_t* result { nullptr };
117
118 try {
119 result = static_cast<uint8_t*>(operator new(allocate_size, static_cast<std::align_val_t>(_alignment)));
120
121 m_blocks.emplace_back(
122 allocate_size,
123 _alignment,
124 result
125 );
126
127 // If the block is not entirely consumed, add the remaining space to the free list.
128 if (const size_t remaining_size = allocate_size - _bytes;
129 remaining_size != 0U
130 ) {
131
132 /*
133 * Attempt to insert new block into free list.
134 * Repeat with a smaller key upon collision.
135 * If insertion is impossible, throw.
136 */
137 bool success = false;
138 for (size_t i = 0U; i < remaining_size; ++i) {
139
140 if (m_free.try_emplace(remaining_size - i, remaining_size, _alignment, result + utils::max(_bytes, _alignment)).second) {
141 success = true;
142 break;
143 }
144 }
145
146 if (UNLIKELY(!success)) {
147 throw std::bad_alloc();
148 }
149 }
150
151 m_block_width = utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size);
152 }
153 catch (...) {
154
155 /* Catch any errors that occur during allocation. */
156
157 // ReSharper disable once CppDFAConstantConditions
158 if (result != nullptr) {
159 // ReSharper disable once CppDFAUnreachableCode
160 operator delete(result, static_cast<std::align_val_t>(_alignment));
161 result = nullptr;
162
163 m_blocks.pop_back();
164 }
165 }
166 return result;
167 }

169 HOT uint8_t* allocate_from_free(size_t _bytes) {
170
171 assert(_bytes > 0U && "Allocation size must be greater than zero.");
172
173 uint8_t* result = nullptr;
174
175 // Find the smallest free block that fits:
176 auto it = m_free.lower_bound(_bytes);
177
178 if (it != m_free.end() && it->second.size >= _bytes) {
179
180 // Extract the matching block without performing a full tree rebalance:
181 typename std::map<size_t, block>::node_type node = m_free.extract(it);
182
183 result = node.mapped().data;
184
185 // Consume block or adjust the remaining size:
186 size_t remaining_size = node.mapped().size - _bytes;
187 if (remaining_size != 0U) {
188
189 // Update with adjusted size:
190 node.key() = remaining_size;
191 node.mapped().size = remaining_size;
192 node.mapped().data += utils::max(_bytes, node.mapped().alignment);
193
194 m_free.insert(std::move(node)); // Reinsert (node is consumed otherwise).
195 }
196 }
197
198 return result;
199 }

247 [[nodiscard]] virtual HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
248
249 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
250
251 if (_bytes == 0U) {
252 return nullptr; // No-op. (See C++17 standard.)
253 }
254 else {
255 const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
256
257 uint8_t* aligned_ptr { nullptr };
258
259 // Attempt to allocate from the stack block:
260 if (aligned_bytes < MaxStackAllocationSize && m_stack_write + aligned_bytes <= s_stack_block_size) {
261
262 aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
263 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
264 }
265
266 if (aligned_ptr == nullptr) {
267
268 // Attempt to find a free block, or create one otherwise:
269 aligned_ptr = allocate_from_free(aligned_bytes);
270 if (UNLIKELY(aligned_ptr == nullptr)) {
271 aligned_ptr = expand(_bytes, _alignment);
272 }
273 }
274
275 // ReSharper disable once CppDFAConstantConditions
276 if (UNLIKELY(aligned_ptr == nullptr)) {
277 // ReSharper disable once CppDFAUnreachableCode
278 throw std::bad_alloc();
279 }
280
281 PREFETCH(aligned_ptr, 1, 1);
282
283#if CHDR_DIAGNOSTICS == 1
284
285 __diagnostic_data. num_allocated += _bytes;
286 __diagnostic_data.peak_allocated = utils::max(__diagnostic_data.peak_allocated, __diagnostic_data.num_allocated);
287
288#endif //CHDR_DIAGNOSTICS == 1
289
290 return aligned_ptr;
291 }

317 virtual HOT void do_deallocate(void* _p, size_t _bytes, size_t _alignment) override {
318
319 assert(_bytes > 0U && "Allocation size must be greater than zero.");
320 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
321
322 if (_bytes != 0U) {
323 // No-op. (See C++17 standard.)
324 }
325 else {
326 if (((m_stack_write + (_bytes + _alignment - 1U)) & ~(_alignment - 1U)) > s_stack_block_size) {
327
328 if constexpr (Coalescing) {
329
330 auto it = m_free.lower_bound(_bytes);
331
332 // Merge with next block if adjacent:
333 if (it != m_free.end() && _bytes + static_cast<uint8_t*>(_p) == it->second.data) {
334 _bytes += it->second.size;
335 _alignment = utils::max(it->second.alignment, _alignment);
336
337 it = m_free.erase(it);
338 }
339
340 // Merge with previous block if adjacent:
341 if (it != m_free.begin()) {
342 --it;
343
344 if (it->second.data + it->second.size == _p) {
345 _p = it->second.data;
346 _bytes += it->second.size;
347 _alignment = utils::max(it->second.alignment, _alignment);
348
349 m_free.erase(it);
350 }
351 }
352 }
353
354 m_free.try_emplace(_bytes, _bytes, _alignment, static_cast<uint8_t*>(_p));
355 }
356
357#if CHDR_DIAGNOSTICS == 1
358
359 __diagnostic_data.num_allocated = (_bytes > __diagnostic_data.num_allocated) ? 0U : (__diagnostic_data.num_allocated - _bytes);
360
361#endif //CHDR_DIAGNOSTICS == 1
362 }
