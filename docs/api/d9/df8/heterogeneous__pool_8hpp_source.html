95 HOT uint8_t* expand(size_t _bytes, size_t _alignment) noexcept {
96
97 const auto allocate_size = utils::max(m_block_width, _bytes);
98
99 uint8_t* result { nullptr };
100
101 try {
102 result = static_cast<uint8_t*>(operator new(allocate_size, static_cast<std::align_val_t>(_alignment)));
103
104 m_blocks.emplace_back(
105 allocate_size,
106 _alignment,
107 result
108 );
109
110 // If the block is not entirely consumed, add the remaining space to the free list.
111 if (const size_t remaining_size = allocate_size - _bytes;
112 remaining_size != 0U
113 ) {
114
115 /*
116 * Attempt to insert new block into free list.
117 * Repeat with a smaller key upon collision.
118 * If insertion is impossible, throw.
119 */
120 bool success = false;
121 for (size_t i = 0U; i < remaining_size; ++i) {
122 if (m_free.try_emplace(remaining_size - i, remaining_size, _alignment, result + utils::max(_bytes, _alignment)).second) {
123 success = true;
124 break;
125 }
126 }
127
128 if (!success) {
129 throw std::bad_alloc();
130 }
131 }
132
133 m_block_width = utils::min((m_block_width * 3U) / 2U, s_max_heap_block_size);
134 }
135 catch (...) {
136
137 /* Catch any errors that occur during allocation. */
138
139 // ReSharper disable once CppDFAConstantConditions
140 if (result != nullptr) {
141 // ReSharper disable once CppDFAUnreachableCode
142 operator delete(result, static_cast<std::align_val_t>(_alignment));
143 result = nullptr;
144
145 m_blocks.pop_back();
146 }
147 }
148 return result;
149 }

151 HOT uint8_t* allocate_from_free(size_t _bytes) {
152
153 uint8_t* result;
154
155 // Find the smallest free block that fits:
156 auto it = m_free.lower_bound(_bytes);
157 if (it != m_free.end() && it->second.size >= _bytes) {
158
159 result = it->second.data;
160
161 // Consume the block, or modify it in-place:
162 auto remaining_size = it->second.size - _bytes;
163 if (remaining_size != 0U) {
164
165 it->second.size = remaining_size;
166 it->second.data += utils::max(_bytes, it->second.alignment);
167 }
168 else {
169 m_free.erase(it);
170 }
171 }
172 else {
173 result = nullptr;
174 }
175
176 return result;
177 }

191 void cleanup() noexcept {
192 for (const auto& item : m_blocks) {
193 operator delete(item.data, static_cast<std::align_val_t>(item.alignment));
194 }
195 }

217 [[nodiscard]] virtual HOT void* do_allocate(const size_t _bytes, const size_t _alignment) override {
218
219 assert(_bytes > 0U && "Allocation size must be greater than zero.");
220 assert((_alignment & (_alignment - 1U)) == 0U && "Alignment must be a power of two.");
221
222 const size_t aligned_bytes = (_bytes + _alignment - 1U) & ~(_alignment - 1U);
223
224 uint8_t* aligned_ptr { nullptr };
225
226 // Attempt to allocate from the stack block:
227 if (aligned_bytes < MaxStackAllocationSize && m_stack_write + aligned_bytes <= s_stack_block_size) {
228
229 aligned_ptr = m_stack_block + ((m_stack_write + _alignment - 1U) & ~(_alignment - 1U));
230 m_stack_write = static_cast<size_t>(aligned_ptr - m_stack_block) + aligned_bytes;
231 }
232 else {
233 // Attempt to find a free block, or create one otherwise:
234 aligned_ptr = allocate_from_free(aligned_bytes);
235 if (aligned_ptr == nullptr) {
236 aligned_ptr = expand(_bytes, _alignment);
237 }
238 }
239
240 if (aligned_ptr == nullptr) {
241 throw std::bad_alloc();
242 }
243
244 return aligned_ptr;
245 }

265 virtual HOT void do_deallocate(void* _p, size_t _bytes, size_t _alignment) override {
266 assert(_p != nullptr && "Cannot deallocate null pointer.");
267
268 if (((m_stack_write + (_bytes + _alignment - 1U)) & ~(_alignment - 1U)) > s_stack_block_size) {
269
270 if constexpr (Coalescing) {
271
272 auto it = m_free.lower_bound(_bytes);
273
274 // Merge with next block if adjacent:
275 if (it != m_free.end() && _bytes + static_cast<uint8_t*>(_p) == it->second.data) {
276 _bytes += it->second.size;
277 _alignment = utils::max(it->second.alignment, _alignment);
278
279 it = m_free.erase(it);
280 }
281
282 // Merge with previous block if adjacent:
283 if (it != m_free.begin()) {
284 --it;
285
286 if (it->second.data + it->second.size == _p) {
287 _p = it->second.data;
288 _bytes += it->second.size;
289 _alignment = utils::max(it->second.alignment, _alignment);
290
291 m_free.erase(it);
292 }
293 }
294 }
295
296 m_free.try_emplace(_bytes, _bytes, _alignment, static_cast<uint8_t*>(_p));
297 }
