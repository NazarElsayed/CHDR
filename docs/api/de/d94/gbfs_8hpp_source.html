57 template <typename open_set_t, typename closed_set_t>
58 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
59
60 const auto s = utils::to_1d(_params.start, _params.size);
61 const auto e = utils::to_1d(_params.end, _params.size);
62
63 _open.emplace(s);
64 _closed.emplace(s);
65
66 // Main loop:
67 while (LIKELY(!_open.empty())) {
68
69 auto curr(std::move(_open.front()));
70 _open.pop();
71
72 if (curr.m_index != e) { // SEARCH FOR SOLUTION...
73
74 node* RESTRICT curr_ptr(nullptr);
75
76 for (const auto& n_data: _params.maze.get_neighbours(curr.m_index)) {
77
78 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
79
80 // Check if node is not already visited:
81 if (!_closed.contains(n.index)) {
82 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
83
84 if (curr_ptr == nullptr) {
85 curr_ptr = new (_params.homogeneous_pmr->allocate(sizeof(node), alignof(node))) node(std::move(curr));
86 }
87
88 _open.emplace(n.index, curr_ptr);
89 }
90 }
91 }
92
93 if (curr_ptr == nullptr) {
94 curr.expunge(_params.homogeneous_pmr);
95 }
96 }
97 else { // SOLUTION REACHED ...
98
99 _open = {};
100 _closed = {};
101
102 return solver_t::solver_utils::rbacktrack(curr, _params.size);
103 }
104 }
105
106 return std::vector<coord_t>{};
107 }

109 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
110
111 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
112
113 existence_set closed(_params.monotonic_pmr);
114 closed.reserve(capacity);
115
116 queue<node> open(_params.polytonic_pmr);
117
118 return solve_internal(open, closed, capacity, _params);
119 }
