120 template <typename open_set_t, typename closed_set_t>
121 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, closed_set_t& _closed, size_t _capacity, const params_t& _params) {
122
123 using neighbours_t = decltype(_params.maze.get_neighbours(std::declval<index_t>()));
124
125 const auto s = utils::to_1d(_params.start, _params.size);
126 const auto e = utils::to_1d(_params.end, _params.size);
127
128 _open.emplace_back(s, 0U);
129
130 stack<state<neighbours_t>> stack(_params.heterogeneous_pmr);
131
132 for (size_t bound = 0U; bound < std::numeric_limits<size_t>::max(); ++bound) {
133
134 stack.emplace(_open.back(), _params);
135
136 // Main loop:
137 while (!stack.empty()) {
138
139 auto& _ = stack.top();
140 auto& curr = _open.back();
141
142 if (curr.m_depth <= bound && _.neighbours_idx != _.neighbours.size()) {
143
144 if (const auto& n = solver_t::get_data(_.neighbours[(_.neighbours.size() - 1U) - (_.neighbours_idx++)], _params); n.active) {
145
146 if (!_closed.contains(n.index)) {
147 solver_t::solver_utils::preallocate_emplace(_closed, n.index, _capacity, _params.maze.count());
148
149 _open.emplace_back(n.index, curr.m_depth + 1U);
150
151 if (n.index != e) { // SEARCH FOR SOLUTION...
152 stack.emplace(_open.back(), _params);
153 }
154 else { // SOLUTION REACHED...
155
156 _closed = closed_set_t{};
157
158 return solver_t::solver_utils::ibacktrack(_open, _params.size);
159 }
160 }
161 }
162 }
163 else {
164 _open.pop_back();
165 stack.pop();
166 _closed.erase(curr.m_index);
167 }
168 }
169
170
171 if (_open.size() > 1U) {
172 _open.resize(1U);
173
174 stack.clear();
175 _closed.clear();
176
177 solver_t::solver_utils::reset_resources(_params);
178 }
179 else {
180 break;
181 }
182 }
183
184 return std::vector<coord_t>{};
185 }

187 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
188
189 const auto capacity = solver_t::solver_utils::determine_capacity(_params);
190
191 existence_set closed(_params.monotonic_pmr);
192 closed.reserve(capacity);
193
194 std::pmr::vector<node> open(_params.heterogeneous_pmr);
195 try {
196 open.reserve(capacity / 8U);
197 }
198 catch (...) {} // NOLINT(*-empty-catch)
199
200 return solve_internal(open, closed, capacity, _params);
201 }
