80 using transposition_table_t = std::unordered_map<index_t, scalar_t>;
81
82 template <typename open_set_t>
83 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, const params_t& _params) {
84
85 using neighbours_t = decltype(_params.maze.get_neighbours(std::declval<index_t>()));
86
87 const auto s = utils::to_1d(_params.start, _params.size);
88 const auto e = utils::to_1d(_params.end, _params.size);
89
90 auto min = std::numeric_limits<scalar_t>::max();
91
92 const auto bound = _params.h(_params.start, _params.end) * _params.weight;
93
94 _open.emplace_back(s, bound);
95
96 stack<state<neighbours_t>> stack;
97 stack.emplace(_open.back(), bound, _params);
98
99 transposition_table_t transposition_table;
100 transposition_table[_open.back().m_index] = bound;
101
102 // Main loop:
103 while (!stack.empty()) {
104
105 auto& _ = stack.top();
106 auto& curr = _.curr;
107
108 if (_.neighbours_idx != _.neighbours.size()) {
109 const auto& n_data = _.neighbours[_.neighbours_idx++];
110
111 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
112
113 auto h = _params.h(n.coord, _params.end) * _params.weight;
114
115 auto search = transposition_table.find(n.index);
116 if (!(search != transposition_table.end() && h >= search->second)) {
117 transposition_table[n.index] = h;
118
119 _open.emplace_back(n.index, h);
120
121 if (n.index != e) { // SEARCH FOR SOLUTION...
122 stack.emplace(_open.back(), _.bound, _params);
123 }
124 else { // SOLUTION REACHED ...
125
126 transposition_table = {};
127
128 return solver_utils::ibacktrack(_open, _params.size);
129 }
130 }
131 }
132 }
133 else {
134 min = utils::min(min, curr.m_hScore);
135
136 _open.pop_back();
137 stack.pop();
138 }
139 }
140
141 return std::vector<coord_t>{};
142 }
143
144 [[maybe_unused, nodiscard]] static auto execute(const params_t& _params) {
145
146 const auto capacity = solver_t::determine_capacity(_params);
147
148 std::pmr::vector<node> open(_params.polytonic_pmr);
149 try {
150 open.reserve(capacity / 8U);
151 }
152 catch (...) {} // NOLINT(*-empty-catch)
153
154 return solve_internal(open, _params);
155 }
