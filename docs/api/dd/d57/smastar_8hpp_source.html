113 template <typename open_set_t, typename all_nodes_t>
114 static void remove_weakest(open_set_t& _open, all_nodes_t& _all_nodes, index_t _forbidden = null_v) {
115
116 assert(!_all_nodes.empty() && "_all_nodes is empty!");
117
118 // Find least-useful node.
119 //const auto weakest_open = std::prev(_open.end());
120
121 // Find deepest node (maximum depth) with highest f-score.
122 auto weakest_all = _all_nodes.begin();
123 for (auto it = _all_nodes.begin(); it != _all_nodes.end(); ++it) {
124 if (it->second.m_index != _forbidden.m_index &&
125 (it->second.m_depth > weakest_all->second.m_depth ||
126 (it->second.m_depth == weakest_all->second.m_depth &&
127 it->second.m_fScore > weakest_all->second.m_fScore))
128 ) {
129 weakest_all = it;
130 }
131 }
132
133 if (weakest_all != _all_nodes.end()) {
134 std::cout << "Memory usage: " << memory_usage(_open, _all_nodes) << "\tRemoving: " << weakest_all->second.m_index << "\n";
135
136 auto weakest = weakest_all->second;
137
138 auto it = std::find_if(
139 _open.begin(),
140 _open.end(),
141 [&weakest](const auto& item) { return item == weakest.m_index; }
142 );
143 if (it != _open.end()) { _open.erase(it); }
144
145 backup_f_values(weakest_all->second, _all_nodes);
146 _all_nodes.erase(weakest_all);
147 }
148 }

150 template <typename all_nodes_t>
151 static void backup_f_values(const node& _removed_node, all_nodes_t& _all_nodes) {
152
153 auto current_parent = _removed_node.m_parent;
154 while (current_parent != null_v) {
155
156 auto parent_it = _all_nodes.find(current_parent);
157 if (parent_it == _all_nodes.end()) {
158 break;
159 }
160
161 auto& parent_node = parent_it->second;
162 auto min_child_f = std::numeric_limits<scalar_t>::max();
163 bool has_children = false;
164
165 // Find minimum f-value amongst children.
166 for (const auto& child_idx : parent_node.m_successors) {
167 auto child_it = _all_nodes.find(child_idx);
168 if (child_it != _all_nodes.end()) {
169 min_child_f = std::min(min_child_f, child_it->second.m_fScore);
170 has_children = true;
171 }
172 }
173
174 // Update parent's f-value if we found children.
175 if (has_children && min_child_f > parent_node.m_fScore) {
176 parent_node.m_fScore = min_child_f;
177 current_parent = parent_node.m_parent; // Move up the chain
178 }
179 else {
180 break;
181 }
182 }

185 [[nodiscard]] HOT static constexpr auto solve_internal(const params_t& _params) {
186
187 const auto s = utils::to_1d(_params.start, _params.size);
188 const auto e = utils::to_1d(_params.end, _params.size);
189
190 std::unordered_map<index_t, node> all_nodes;
191
192 const static auto node_comparator = [&all_nodes](index_t _a, index_t _b) {
193
194 auto it_a = all_nodes.find(_a);
195 auto it_b = all_nodes.find(_b);
196
197 assert(!(it_a == all_nodes.end() || it_b == all_nodes.end()) && "Node lookup failed in comparator");
198
199 return !(it_a->second < it_b->second);
200 };
201
202 std::set<index_t, decltype(node_comparator)> open(node_comparator);
203
204 // Add start node
205 all_nodes[s] = node(
206 static_cast<index_t>(0),
207 static_cast<index_t>(s),
208 static_cast<scalar_t>(0),
209 _params.h(_params.start, _params.end) * _params.weight
210 );
211 open.emplace(static_cast<index_t>(s));
212
213 // Main loop:
214 while (LIKELY(!open.empty())) {
215
216 std::cout << "Memory usage: " << memory_usage(open, all_nodes) << "\n";
217
218 // Handle memory management:
219 while (open.size() > 1U && memory_usage(open, all_nodes) > _params.memory_limit) {
220 remove_weakest(open, all_nodes);
221 }
222
223 // Get current node index and remove from open set
224 auto curr_idx = *open.begin();
225 open.erase(open.begin());
226
227 auto& curr = all_nodes[curr_idx];
228
229 if (curr.m_index == e) { // SOLUTION REACHED...
230
231 std::vector<coord_t> result(curr.m_depth);
232
233 size_t i = 0U;
234 for (auto p = curr.m_parent; p != null_v; p = all_nodes[p].m_parent) {
235 result[(result.size() - 1U) - i] = utils::to_nd(all_nodes[p].m_index, _params.size);
236 ++i;
237 }
238
239 std::cout << "Final path length: " << result.size() << "\n";
240
241 return result;
242 }
243 else { // SEARCH FOR SOLUTION...
244
245 // Expand successors if not already:
246 bool expanded = !curr.m_successors.empty() || curr.m_fScore == std::numeric_limits<scalar_t>::max();
247 if (!expanded) {
248
249 if (curr.m_depth == _params.memory_limit) {
250 curr.m_fScore = std::numeric_limits<scalar_t>::max();
251 }
252 else {
253
254 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
255
256 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
257
258 curr.m_successors.push_back(n.index);
259
260 const auto g = curr.m_gScore + n.distance;
261 const auto h = _params.h(n.coord, _params.end) * _params.weight;
262
263 // Create or update successor node:
264 auto child_search = all_nodes.find(n.index);
265 if (child_search == all_nodes.end()) {
266
267 // Create room for new node if necessary:
268 while (memory_usage(open, all_nodes) > _params.memory_limit) {
269 remove_weakest(open, all_nodes, curr_idx);
270 }
271
272 // Create new node:
273 all_nodes[n.index] = node(curr.m_depth + 1U, n.index, g, h, curr.m_index);
274 open.emplace(n.index);
275 }
276 else {
277
278 // Update existing node and reinsert:
279 auto& child_node = child_search->second;
280 if (g < child_node.m_gScore) {
281
282 open.erase(n.index);
283
284 child_node.m_depth = curr.m_depth + 1U;
285 child_node.m_gScore = g;
286 child_node.m_fScore = std::max(g + h, curr.m_fScore);
287 child_node.m_parent = curr.m_index;
288
289 open.emplace(n.index);
290 }
291 }
292 }
293 }
294 }
295 }
296
297 // Handle case where node has no successors (dead end):
298 if (curr.m_successors.empty()) {
299 curr.m_fScore = std::numeric_limits<scalar_t>::max();
300 backup_f_values(curr, all_nodes);
301 }
302 else if (!expanded && memory_usage(open, all_nodes) < _params.memory_limit) {
303 open.emplace(curr_idx);
304 }
305 }
306 }
307
308 return std::vector<coord_t>{};
309 }
310
311 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
312
313 if (_params.memory_limit > 0U) {
314 return solve_internal(_params);
315 }
316 else {
317 return std::vector<coord_t>{};
318 }
319 }
320 };
