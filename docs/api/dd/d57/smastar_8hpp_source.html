107 template <typename open_set_t>
108 [[nodiscard]] HOT static constexpr auto solve_internal(open_set_t& _open, const params_t& _params) {
109
110 const auto s = utils::to_1d(_params.start, _params.size);
111 const auto e = utils::to_1d(_params.end, _params.size);
112
113 std::unordered_map<index_t, node> generated;
114
115 {
116 node start(true, static_cast<index_t>(0), s, static_cast<scalar_t>(0), _params.h(_params.start, _params.end) * _params.weight);
117 _open.emplace(start);
118 auto [it, success] = generated.try_emplace(s, start);
119 it->second.m_parents.clear();
120 }
121
122 // Main loop:
123 while (LIKELY(!_open.empty())) {
124 node curr(std::move(_open.extract(_open.begin()).value()));
125
126 if (curr.m_index == e) { // SOLUTION REACHED ...
127
128 if constexpr (std::is_same_v<std::decay_t<decltype(_open)>, heap<node>>) {
129 _open.wipe();
130 }
131 else {
132 _open = open_set_t{};
133 }
134
135 return solver_t::solver_utils::rbacktrack(curr, _params.size, curr.m_gScore);
136 }
137 else { // SEARCH FOR SOLUTION...
138
139 // Expand (generate successors of curr):
140 if (curr.m_successors.empty()) {
141 for (const auto& n_data : _params.maze.get_neighbours(curr.m_index)) {
142 if (const auto& n = solver_t::get_data(n_data, _params); n.active) {
143 node* n_ptr(nullptr);
144
145 auto search = generated.find(n.index);
146 if (search == generated.end()) {
147 auto [it, success] = generated.try_emplace(n.index, false, curr.m_depth + 1U, n.index, curr.m_gScore + n.distance,
148 _params.h(n.coord, _params.end) * _params.weight, &curr);
149 n_ptr = &(it->second);
150 n_ptr->m_parents.push_back(&curr);
151 }
152 else {
153 n_ptr = &(search->second);
154 if (std::find(n_ptr->m_parents.begin(), n_ptr->m_parents.end(), &curr) == n_ptr->m_parents.end()) {
155 n_ptr->m_parents.push_back(&curr);
156 }
157 }
158
159 curr.m_successors.push_back(n_ptr);
160 }
161 }
162 }
163
164 node* succ(nullptr);
165 for (auto* successor : curr.m_successors) {
166 if (!successor->m_enqueued) {
167 succ = successor;
168 break;
169 }
170 }
171
172 if (succ != nullptr) {
173 if (succ->m_index != e && succ->m_depth == _params.memory_limit) {
174 succ->m_fScore = std::numeric_limits<scalar_t>::infinity();
175 }
176 else {
177 const auto sCoord = utils::to_nd(succ->m_index, _params.size);
178 succ->m_fScore = std::max(succ->m_fScore, succ->m_gScore + _params.h(sCoord, _params.end) * _params.weight);
179 }
180 }
181 else {
182 // Update current node's f-cost to minimum of successor f-costs.
183 if (!curr.m_successors.empty()) {
184 auto min_successor_f = std::numeric_limits<scalar_t>::infinity();
185 for (const auto* successor : curr.m_successors) {
186 min_successor_f = std::min(min_successor_f, successor->m_fScore);
187 }
188 curr.m_fScore = std::max(curr.m_fScore, min_successor_f);
189 }
190
191 // Propagate cost updates to parent nodes if needed.
192 for (auto* parent : curr.m_parents) {
193 // Calculate what the parent's f-cost should be based on current node's updated cost.
194 scalar_t new_parent_f = std::numeric_limits<scalar_t>::infinity();
195
196 // Find minimum f-cost among all of parent's successors:
197 for (const auto* successor : parent->m_successors) {
198 new_parent_f = std::min(new_parent_f, successor->m_fScore);
199 }
200
201 // Update parent's f-cost if the new minimum is higher than current:
202 if (new_parent_f > parent->m_fScore) {
203 // Find the parent in the open set:
204 auto parent_it = _open.find(*parent);
205 if (parent_it != _open.end()) {
206 // Remove from set, and re-insert with updated cost.
207 _open.erase(parent_it);
208 parent->m_fScore = new_parent_f;
209 _open.emplace(*parent);
210 }
211 else {
212 // Parent not in open set, just update the cost.
213 parent->m_fScore = new_parent_f;
214 }
215 }
216 }
217 }
218
219 bool all_successors_in_queue = true;
220 for (auto* successor : curr.m_successors) {
221 if (!successor->m_enqueued) {
222 all_successors_in_queue = false;
223 break;
224 }
225 }
226 if (all_successors_in_queue) {
227 curr.m_enqueued = false;
228 }
229
230 // memory management
231 if (_open.size() >= _params.memory_limit) {
232
233 // Find shallowest node with highest f-cost.
234 // Remove it and update parents:
235
236 const auto& bad = *(--_open.end());
237 const auto& bad_node = bad;
238
239 for (auto* parent : bad_node.m_parents) {
240 parent->m_successors.erase(
241 std::remove_if(
242 parent->m_successors.begin(),
243 parent->m_successors.end(),
244 [&bad_node](const node* n) { return n == &bad_node; }
245 ),
246 parent->m_successors.end()
247 );
248
249 if (parent->m_successors.empty()) {
250 _open.emplace(*parent);
251 }
252 }
253
254 _open.erase(bad);
255 }
256
257 if (succ != nullptr && !succ->m_enqueued) {
258 succ->m_enqueued = true;
259 _open.emplace(*succ);
260 }
261 }
262 }
263
264 return std::vector<coord_t>{};
265 }

267 [[maybe_unused, nodiscard]] static auto invoke(const params_t& _params) {
268
269 std::pmr::set<node> open(_params.heterogeneous_pmr);
270
271 return solve_internal(open, _params);
272 }
