22 template <typename index_t, typename derived = void>
23 struct managed_node : bnode<index_t> {
24
25 using node_t = std::conditional_t<std::is_void_v<derived>, managed_node<index_t>, derived>;
26
27 managed_node* RESTRICT m_parent;
28 unsigned char m_successors;
29
30 /**
31 * @brief Constructs an uninitialized node.
32 *
33 * This constructor creates a node with uninitialized members.
34 */
35 [[nodiscard]] constexpr managed_node() noexcept : // NOLINT(*-pro-type-member-init, *-use-equals-default)
36 m_parent(nullptr),
37 m_successors(0U) {}
38 ~managed_node() noexcept = default;
39
40 [[nodiscard]] constexpr managed_node(const index_t& _index, managed_node* RESTRICT const _parent = nullptr) noexcept : bnode<index_t>(_index),
41 m_parent(_parent),
42 m_successors(0U)
43 {
44 if (_parent != nullptr) {
45 incr();
46 }
47 }

59 template <typename pmr_t>
60 void expunge(pmr_t* _pmr) noexcept {
61
62 while (m_parent != nullptr) {
63 decr();
64
65 if (m_parent->m_successors != 0U) {
66 break;
67 }
68
69 auto* const RESTRICT d = m_parent;
70 m_parent = m_parent->m_parent;
71 _pmr->deallocate(d, sizeof(managed_node), alignof(managed_node));
72 }
73 }
74
75 constexpr void decr() const noexcept {
76 assert(m_parent != nullptr && "Dereferencing of nullptr.");
77 assert(m_parent->m_successors != static_cast<decltype(m_successors)>(0U) && "Underflow detected!");
78
79 --m_parent->m_successors;
80 }
81
82 constexpr void incr() const noexcept {
83 assert(m_parent != nullptr && "Dereferencing of nullptr.");
84 assert(m_parent->m_successors != static_cast<decltype(m_successors)>(-1U) && "Overflow detected!");
85
86 ++m_parent->m_successors;
87 }
88
89 [[nodiscard]] friend constexpr bool operator < (const managed_node& _a, const managed_node& _b) noexcept {
90 return _a.m_fScore == _b.m_fScore ?
91 _a.m_gScore > _b.m_gScore :
92 _a.m_fScore > _b.m_fScore;
93 }
94 };
